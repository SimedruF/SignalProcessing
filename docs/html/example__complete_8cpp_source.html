29void acquire_ecg_data(SignalProcessing *sp, double duration_sec,
30 double sampling_rate, double heart_rate)
31{
32 int total_samples = (int)(duration_sec * sampling_rate);
33 double beat_interval = 60.0 / heart_rate; // seconds per beat
34 int samples_per_beat = (int)(beat_interval * sampling_rate);
35
36 printf("Simulating ECG acquisition:\n");
37 printf(" Duration: %.1f seconds\n", duration_sec);
38 printf(" Sampling rate: %.0f Hz\n", sampling_rate);
39 printf(" Heart rate: %.0f BPM\n", heart_rate);
40 printf(" Total samples: %d\n\n", total_samples);
41
42 struct timespec start_time;
43 clock_gettime(CLOCK_REALTIME, &start_time);
44
45 for (int i = 0; i < total_samples; ++i)
46 {
47 // Baseline wander (breathing artifact)
48 double baseline = 0.3 * sin(2.0 * M_PI * i / (sampling_rate * 3.0));
49
50 // P-wave, QRS complex, T-wave
51 int phase = i % samples_per_beat;
52 double ecg_value = baseline;
53
54 // P-wave (small bump before QRS)
55 if (phase >= samples_per_beat * 0.15 && phase <= samples_per_beat * 0.25)
56 {
57 double p_pos = (phase - samples_per_beat * 0.2) / (samples_per_beat * 0.05);
58 ecg_value += 0.3 * exp(-p_pos * p_pos / 0.5);
59 }
60
61 // QRS complex (sharp peak)
62 if (phase >= samples_per_beat * 0.45 && phase <= samples_per_beat * 0.55)
63 {
64 double qrs_pos = (phase - samples_per_beat * 0.5) / (samples_per_beat * 0.02);
65 // Q-wave (small negative)
66 if (qrs_pos < -1.0)
67 ecg_value -= 0.2 * exp(-(qrs_pos + 1.5) * (qrs_pos + 1.5) / 0.3);
68 // R-wave (large positive)
69 else if (qrs_pos >= -1.0 && qrs_pos <= 0.0)
70 ecg_value += 5.0 * exp(-qrs_pos * qrs_pos / 0.1);
71 // S-wave (negative)
72 else
73 ecg_value -= 0.5 * exp(-(qrs_pos - 1.0) * (qrs_pos - 1.0) / 0.3);
74 }
75
76 // T-wave (rounded bump after QRS)
77 if (phase >= samples_per_beat * 0.65 && phase <= samples_per_beat * 0.85)
78 {
79 double t_pos = (phase - samples_per_beat * 0.75) / (samples_per_beat * 0.1);
80 ecg_value += 0.5 * exp(-t_pos * t_pos / 1.0);
81 }
82
83 // Add noise
84 ecg_value += 0.05 * (rand() % 100 - 50) / 100.0;
85
86 // Create timestamp
87 struct timespec ts = start_time;
88 long nsec_offset = (long)((i / sampling_rate) * 1e9);
89 ts.tv_sec += nsec_offset / 1000000000L;
90 ts.tv_nsec += nsec_offset % 1000000000L;
91 if (ts.tv_nsec >= 1000000000L) {
92 ts.tv_sec++;
93 ts.tv_nsec -= 1000000000L;
94 }
95
96 sp->AddValueWithTimestamp(ecg_value, ts);
97 }
98
99 printf("✓ Data acquisition complete: %d samples\n\n", sp->GetIndex());
100}

131void detect_r_peaks(SignalProcessing *sp, double sampling_rate)
132{
133 printf("Detecting R-peaks:\n");
134
135 // Calculate adaptive threshold
136 double mean = sp->GetMean();
137 double std = sp->GetStandardDeviation();
138 double threshold = mean + 1.5 * std;
139 printf(" Threshold: %.3f (mean + 1.5*std)\n", threshold);
140
141 // Method 1: Threshold-based detection
142 int peaks_threshold[100];
143 int count_threshold = sp->DetectPeaksWithThreshold(threshold, peaks_threshold, 100);
144 printf(" Method 1 (Threshold): %d peaks detected\n", count_threshold);
145
146 // Method 2: Distance-based detection (more robust)
147 int min_rr_distance = (int)(sampling_rate * 0.4); // Min 0.4s between beats (150 BPM max)
148 int peaks_distance[100];
149 int count_distance = sp->DetectPeaksWithDistance(min_rr_distance, peaks_distance, 100);
150 printf(" Method 2 (Distance): %d peaks detected\n", count_distance);
151
152 // Use distance-based peaks for analysis
153 if (count_distance >= 2)
154 {
155 printf("\n R-peak analysis:\n");
156
157 // Calculate RR intervals
158 double rr_intervals[100];
159 double sum_rr = 0;
160
161 for (int i = 1; i < count_distance; ++i)
162 {
163 rr_intervals[i-1] = (peaks_distance[i] - peaks_distance[i-1]) / sampling_rate;
164 sum_rr += rr_intervals[i-1];
165 printf(" RR[%d]: %.3f s (%.1f BPM)\n",
166 i, rr_intervals[i-1], 60.0 / rr_intervals[i-1]);
167 }
168
169 // Average heart rate
170 double avg_rr = sum_rr / (count_distance - 1);
171 double avg_hr = 60.0 / avg_rr;
172 printf("\n ✓ Average Heart Rate: %.1f BPM\n", avg_hr);
173
174 // Heart Rate Variability (SDNN)
175 double hrv_sum = 0;
176 for (int i = 0; i < count_distance - 1; ++i)
177 {
178 double diff = rr_intervals[i] - avg_rr;
179 hrv_sum += diff * diff;
180 }
181 double sdnn = sqrt(hrv_sum / (count_distance - 1));
182 printf(" ✓ HRV (SDNN): %.3f s\n", sdnn);
183
184 // Peak amplitudes
185 printf("\n R-peak amplitudes:\n");
186 for (int i = 0; i < count_distance; ++i)
187 {
188 double amplitude = sp->GetPeakValue(peaks_distance[i]);
189 printf(" Peak[%d] at sample %d: %.3f\n",
190 i+1, peaks_distance[i], amplitude);
191 }
192 }
193 else
194 {
195 printf(" ⚠ Not enough peaks for analysis\n");
196 }
197
198 printf("\n");
199}
