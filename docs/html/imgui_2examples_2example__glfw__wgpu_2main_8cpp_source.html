295static WGPUAdapter RequestAdapter(wgpu::Instance& instance)
296{
297 wgpu::Adapter acquired_adapter;
298 wgpu::RequestAdapterOptions adapter_options;
299 auto onRequestAdapter = [&](wgpu::RequestAdapterStatus status, wgpu::Adapter adapter, wgpu::StringView message)
300 {
301 if (status != wgpu::RequestAdapterStatus::Success)
302 {
303 printf("Failed to get an adapter: %s\n", message.data);
304 return;
305 }
306 acquired_adapter = std::move(adapter);
307 };
308
309 // Synchronously (wait until) acquire Adapter
310 wgpu::Future waitAdapterFunc { instance.RequestAdapter(&adapter_options, wgpu::CallbackMode::WaitAnyOnly, onRequestAdapter) };
311 wgpu::WaitStatus waitStatusAdapter = instance.WaitAny(waitAdapterFunc, UINT64_MAX);
312 IM_ASSERT(acquired_adapter != nullptr && waitStatusAdapter == wgpu::WaitStatus::Success && "Error on Adapter request");
313 return acquired_adapter.MoveToCHandle();
314}

322 );
323 device_desc.SetUncapturedErrorCallback(
324 [](const wgpu::Device&, wgpu::ErrorType type, wgpu::StringView msg) { fprintf(stderr, "%s error: %s\n", ImGui_ImplWGPU_GetErrorTypeName((WGPUErrorType)type), msg.data); }

329 {
330 if (status != wgpu::RequestDeviceStatus::Success)

332 printf("Failed to get an device: %s\n", message.data);
333 return;

335 acquired_device = std::move(local_device);
336 };
337
338 // Synchronously (wait until) get Device

409static bool InitWGPU(GLFWwindow* window)
410{
411 WGPUTextureFormat preferred_fmt = WGPUTextureFormat_Undefined; // acquired from SurfaceCapabilities
412
413 // Google DAWN backend: Adapter and Device acquisition, Surface creation
414#if defined(IMGUI_IMPL_WEBGPU_BACKEND_DAWN)
415 wgpu::InstanceDescriptor instance_desc = {};
416 static constexpr wgpu::InstanceFeatureName timedWaitAny = wgpu::InstanceFeatureName::TimedWaitAny;
417 instance_desc.requiredFeatureCount = 1;
418 instance_desc.requiredFeatures = &timedWaitAny;
419 wgpu::Instance instance = wgpu::CreateInstance(&instance_desc);
420
421 wgpu::Adapter adapter = RequestAdapter(instance);
422 ImGui_ImplWGPU_DebugPrintAdapterInfo(adapter.Get());
423
424 wgpu_device = RequestDevice(instance, adapter);
425
426 // Create the surface.
427#ifdef __EMSCRIPTEN__
428 wgpu::EmscriptenSurfaceSourceCanvasHTMLSelector canvas_desc = {};
429 canvas_desc.selector = "#canvas";
430
431 wgpu::SurfaceDescriptor surface_desc = {};
432 surface_desc.nextInChain = &canvas_desc;
433 wgpu_surface = instance.CreateSurface(&surface_desc).MoveToCHandle();
434#else
435 wgpu_surface = CreateWGPUSurface(instance.Get(), window);
436#endif
437 if (!wgpu_surface)
438 return false;
439
440 // Moving Dawn objects into WGPU handles
441 wgpu_instance = instance.MoveToCHandle();
442
443 WGPUSurfaceCapabilities surface_capabilities = {};
444 wgpuSurfaceGetCapabilities(wgpu_surface, adapter.Get(), &surface_capabilities);
445
446 preferred_fmt = surface_capabilities.formats[0];
447
448 // WGPU backend: Adapter and Device acquisition, Surface creation
449#elif defined(IMGUI_IMPL_WEBGPU_BACKEND_WGPU)
450 wgpu_instance = wgpuCreateInstance(nullptr);
451
452#ifdef __EMSCRIPTEN__
453 getAdapterAndDeviceViaJS();
454
455 wgpu_device = emscripten_webgpu_get_device();
456 IM_ASSERT(wgpu_device != nullptr && "Error creating the Device");
457

459 html_surface_desc.chain.sType = WGPUSType_SurfaceDescriptorFromCanvasHTMLSelector;
460 html_surface_desc.selector = "#canvas";
461
462 WGPUSurfaceDescriptor surface_desc = {};
463 surface_desc.nextInChain = &html_surface_desc.chain;
464
465 // Create the surface.
466 wgpu_surface = wgpuInstanceCreateSurface(wgpu_instance, &surface_desc);
467 preferred_fmt = wgpuSurfaceGetPreferredFormat(wgpu_surface, {} /* adapter */);
468#else // __EMSCRIPTEN__
469 wgpuSetLogCallback(
470 [](WGPULogLevel level, WGPUStringView msg, void* userdata) { fprintf(stderr, "%s: %.*s\n", ImGui_ImplWGPU_GetLogLevelName(level), (int)msg.length, msg.data); }, nullptr
471 );

473
474 WGPUAdapter adapter = RequestAdapter(wgpu_instance);
475 ImGui_ImplWGPU_DebugPrintAdapterInfo(adapter);
476
477 wgpu_device = RequestDevice(adapter);
478
479 // Create the surface.
480 wgpu_surface = CreateWGPUSurface(wgpu_instance, window);
481 if (!wgpu_surface)
482 return false;
483
484 WGPUSurfaceCapabilities surface_capabilities = {};
485 wgpuSurfaceGetCapabilities(wgpu_surface, adapter, &surface_capabilities);
486
487 preferred_fmt = surface_capabilities.formats[0];
488#endif // __EMSCRIPTEN__
489#endif // IMGUI_IMPL_WEBGPU_BACKEND_WGPU
490
491 wgpu_surface_configuration.presentMode = WGPUPresentMode_Fifo;
492 wgpu_surface_configuration.alphaMode = WGPUCompositeAlphaMode_Auto;
493 wgpu_surface_configuration.usage = WGPUTextureUsage_RenderAttachment;
494 wgpu_surface_configuration.width = wgpu_surface_width;
495 wgpu_surface_configuration.height = wgpu_surface_height;
496 wgpu_surface_configuration.device = wgpu_device;
497 wgpu_surface_configuration.format = preferred_fmt;
498
499 wgpuSurfaceConfigure(wgpu_surface, &wgpu_surface_configuration);
500 wgpu_queue = wgpuDeviceGetQueue(wgpu_device);
501
502 return true;
503}
