204// Helper functions to use DirectX11
205bool CreateDeviceD3D(HWND hWnd)
206{
207 // Setup swap chain
208 // This is a basic setup. Optimally could use e.g. DXGI_SWAP_EFFECT_FLIP_DISCARD and handle fullscreen mode differently. See #8979 for suggestions.
209 DXGI_SWAP_CHAIN_DESC sd;
210 ZeroMemory(&sd, sizeof(sd));
211 sd.BufferCount = 2;
212 sd.BufferDesc.Width = 0;
213 sd.BufferDesc.Height = 0;
214 sd.BufferDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
215 sd.BufferDesc.RefreshRate.Numerator = 60;
216 sd.BufferDesc.RefreshRate.Denominator = 1;
217 sd.Flags = DXGI_SWAP_CHAIN_FLAG_ALLOW_MODE_SWITCH;
218 sd.BufferUsage = DXGI_USAGE_RENDER_TARGET_OUTPUT;
219 sd.OutputWindow = hWnd;
220 sd.SampleDesc.Count = 1;
221 sd.SampleDesc.Quality = 0;
222 sd.Windowed = TRUE;
223 sd.SwapEffect = DXGI_SWAP_EFFECT_DISCARD;
224
225 UINT createDeviceFlags = 0;
226 //createDeviceFlags |= D3D11_CREATE_DEVICE_DEBUG;
227 D3D_FEATURE_LEVEL featureLevel;
228 const D3D_FEATURE_LEVEL featureLevelArray[2] = { D3D_FEATURE_LEVEL_11_0, D3D_FEATURE_LEVEL_10_0, };
229 HRESULT res = D3D11CreateDeviceAndSwapChain(nullptr, D3D_DRIVER_TYPE_HARDWARE, nullptr, createDeviceFlags, featureLevelArray, 2, D3D11_SDK_VERSION, &sd, &g_pSwapChain, &g_pd3dDevice, &featureLevel, &g_pd3dDeviceContext);
230 if (res == DXGI_ERROR_UNSUPPORTED) // Try high-performance WARP software driver if hardware is not available.
231 res = D3D11CreateDeviceAndSwapChain(nullptr, D3D_DRIVER_TYPE_WARP, nullptr, createDeviceFlags, featureLevelArray, 2, D3D11_SDK_VERSION, &sd, &g_pSwapChain, &g_pd3dDevice, &featureLevel, &g_pd3dDeviceContext);
232 if (res != S_OK)
233 return false;
234
235 CreateRenderTarget();
236 return true;
237}
