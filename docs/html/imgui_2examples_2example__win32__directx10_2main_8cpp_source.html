198bool CreateDeviceD3D(HWND hWnd)
199{
200 // Setup swap chain
201 // This is a basic setup. Optimally could use handle fullscreen mode differently. See #8979 for suggestions.
202 DXGI_SWAP_CHAIN_DESC sd;
203 ZeroMemory(&sd, sizeof(sd));
204 sd.BufferCount = 2;
205 sd.BufferDesc.Width = 0;
206 sd.BufferDesc.Height = 0;
207 sd.BufferDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
208 sd.BufferDesc.RefreshRate.Numerator = 60;
209 sd.BufferDesc.RefreshRate.Denominator = 1;
210 sd.Flags = DXGI_SWAP_CHAIN_FLAG_ALLOW_MODE_SWITCH;
211 sd.BufferUsage = DXGI_USAGE_RENDER_TARGET_OUTPUT;
212 sd.OutputWindow = hWnd;
213 sd.SampleDesc.Count = 1;
214 sd.SampleDesc.Quality = 0;
215 sd.Windowed = TRUE;
216 sd.SwapEffect = DXGI_SWAP_EFFECT_DISCARD;
217
218 UINT createDeviceFlags = 0;
219 //createDeviceFlags |= D3D10_CREATE_DEVICE_DEBUG;
220 HRESULT res = D3D10CreateDeviceAndSwapChain(nullptr, D3D10_DRIVER_TYPE_HARDWARE, nullptr, createDeviceFlags, D3D10_SDK_VERSION, &sd, &g_pSwapChain, &g_pd3dDevice);
221 if (res == DXGI_ERROR_UNSUPPORTED) // Try high-performance WARP software driver if hardware is not available.
222 res = D3D10CreateDeviceAndSwapChain(nullptr, D3D10_DRIVER_TYPE_WARP, nullptr, createDeviceFlags, D3D10_SDK_VERSION, &sd, &g_pSwapChain, &g_pd3dDevice);
223 if (res != S_OK)
224 return false;
225
226 CreateRenderTarget();
227 return true;
228}

253// Win32 message handler
254// You can read the io.WantCaptureMouse, io.WantCaptureKeyboard flags to tell if dear imgui wants to use your inputs.
255// - When io.WantCaptureMouse is true, do not dispatch mouse input data to your main application, or clear/overwrite your copy of the mouse data.
256// - When io.WantCaptureKeyboard is true, do not dispatch keyboard input data to your main application, or clear/overwrite your copy of the keyboard data.
257// Generally you may always pass all inputs to dear imgui, and hide them from your application based on those two flags.
258LRESULT WINAPI WndProc(HWND hWnd, UINT msg, WPARAM wParam, LPARAM lParam)
259{
260 if (ImGui_ImplWin32_WndProcHandler(hWnd, msg, wParam, lParam))
261 return true;
262
263 switch (msg)
264 {
265 case WM_SIZE:
266 if (wParam == SIZE_MINIMIZED)
267 return 0;
268 g_ResizeWidth = (UINT)LOWORD(lParam); // Queue resize
269 g_ResizeHeight = (UINT)HIWORD(lParam);
270 return 0;
271 case WM_SYSCOMMAND:
272 if ((wParam & 0xfff0) == SC_KEYMENU) // Disable ALT application menu
273 return 0;
274 break;
275 case WM_DESTROY:
276 ::PostQuitMessage(0);
277 return 0;
278 }
279 return ::DefWindowProcW(hWnd, msg, wParam, lParam);
