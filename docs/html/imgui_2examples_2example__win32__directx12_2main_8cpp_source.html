36// Simple free list based allocator
37struct ExampleDescriptorHeapAllocator
38{
39 ID3D12DescriptorHeap* Heap = nullptr;
40 D3D12_DESCRIPTOR_HEAP_TYPE HeapType = D3D12_DESCRIPTOR_HEAP_TYPE_NUM_TYPES;
41 D3D12_CPU_DESCRIPTOR_HANDLE HeapStartCpu;
42 D3D12_GPU_DESCRIPTOR_HANDLE HeapStartGpu;
43 UINT HeapHandleIncrement;
44 ImVector<int> FreeIndices;
45
46 void Create(ID3D12Device* device, ID3D12DescriptorHeap* heap)
47 {
48 IM_ASSERT(Heap == nullptr && FreeIndices.empty());
49 Heap = heap;
50 D3D12_DESCRIPTOR_HEAP_DESC desc = heap->GetDesc();
51 HeapType = desc.Type;
52 HeapStartCpu = Heap->GetCPUDescriptorHandleForHeapStart();
53 HeapStartGpu = Heap->GetGPUDescriptorHandleForHeapStart();
54 HeapHandleIncrement = device->GetDescriptorHandleIncrementSize(HeapType);
55 FreeIndices.reserve((int)desc.NumDescriptors);
56 for (int n = desc.NumDescriptors; n > 0; n--)
57 FreeIndices.push_back(n - 1);
58 }
59 void Destroy()
60 {
61 Heap = nullptr;
62 FreeIndices.clear();
63 }
64 void Alloc(D3D12_CPU_DESCRIPTOR_HANDLE* out_cpu_desc_handle, D3D12_GPU_DESCRIPTOR_HANDLE* out_gpu_desc_handle)
65 {
66 IM_ASSERT(FreeIndices.Size > 0);
67 int idx = FreeIndices.back();
68 FreeIndices.pop_back();
69 out_cpu_desc_handle->ptr = HeapStartCpu.ptr + (idx * HeapHandleIncrement);
70 out_gpu_desc_handle->ptr = HeapStartGpu.ptr + (idx * HeapHandleIncrement);
71 }
72 void Free(D3D12_CPU_DESCRIPTOR_HANDLE out_cpu_desc_handle, D3D12_GPU_DESCRIPTOR_HANDLE out_gpu_desc_handle)
73 {
74 int cpu_idx = (int)((out_cpu_desc_handle.ptr - HeapStartCpu.ptr) / HeapHandleIncrement);
75 int gpu_idx = (int)((out_gpu_desc_handle.ptr - HeapStartGpu.ptr) / HeapHandleIncrement);
76 IM_ASSERT(cpu_idx == gpu_idx);
77 FreeIndices.push_back(cpu_idx);
78 }
79};

312bool CreateDeviceD3D(HWND hWnd)
313{
314 // Setup swap chain
315 // This is a basic setup. Optimally could handle fullscreen mode differently. See #8979 for suggestions.
316 DXGI_SWAP_CHAIN_DESC1 sd;
317 {
318 ZeroMemory(&sd, sizeof(sd));
319 sd.BufferCount = APP_NUM_BACK_BUFFERS;
320 sd.Width = 0;
321 sd.Height = 0;
322 sd.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
323 sd.Flags = DXGI_SWAP_CHAIN_FLAG_FRAME_LATENCY_WAITABLE_OBJECT;
324 sd.BufferUsage = DXGI_USAGE_RENDER_TARGET_OUTPUT;
325 sd.SampleDesc.Count = 1;
326 sd.SampleDesc.Quality = 0;
327 sd.SwapEffect = DXGI_SWAP_EFFECT_FLIP_DISCARD;
328 sd.AlphaMode = DXGI_ALPHA_MODE_UNSPECIFIED;
329 sd.Scaling = DXGI_SCALING_STRETCH;
330 sd.Stereo = FALSE;
331 }
332
333 // [DEBUG] Enable debug interface
334#ifdef DX12_ENABLE_DEBUG_LAYER
335 ID3D12Debug* pdx12Debug = nullptr;
336 if (SUCCEEDED(D3D12GetDebugInterface(IID_PPV_ARGS(&pdx12Debug))))
337 pdx12Debug->EnableDebugLayer();
338#endif
339
340 // Create device
341 D3D_FEATURE_LEVEL featureLevel = D3D_FEATURE_LEVEL_11_0;
342 if (D3D12CreateDevice(nullptr, featureLevel, IID_PPV_ARGS(&g_pd3dDevice)) != S_OK)
343 return false;
344
345 // [DEBUG] Setup debug interface to break on any warnings/errors
346#ifdef DX12_ENABLE_DEBUG_LAYER
347 if (pdx12Debug != nullptr)
348 {
349 ID3D12InfoQueue* pInfoQueue = nullptr;
350 g_pd3dDevice->QueryInterface(IID_PPV_ARGS(&pInfoQueue));
351 pInfoQueue->SetBreakOnSeverity(D3D12_MESSAGE_SEVERITY_ERROR, true);
352 pInfoQueue->SetBreakOnSeverity(D3D12_MESSAGE_SEVERITY_CORRUPTION, true);
353 pInfoQueue->SetBreakOnSeverity(D3D12_MESSAGE_SEVERITY_WARNING, true);
354
355 // Disable breaking on this warning because of a suspected bug in the D3D12 SDK layer, see #9084 for details.
356 const int D3D12_MESSAGE_ID_FENCE_ZERO_WAIT_ = 1424; // not in all copies of d3d12sdklayers.h
357 D3D12_MESSAGE_ID disabledMessages[] = { (D3D12_MESSAGE_ID)D3D12_MESSAGE_ID_FENCE_ZERO_WAIT_ };
358 D3D12_INFO_QUEUE_FILTER filter = {};
359 filter.DenyList.NumIDs = 1;
360 filter.DenyList.pIDList = disabledMessages;
361 pInfoQueue->AddStorageFilterEntries(&filter);
362
363 pInfoQueue->Release();
364 pdx12Debug->Release();
365 }
366#endif
367
368 {
369 D3D12_DESCRIPTOR_HEAP_DESC desc = {};
370 desc.Type = D3D12_DESCRIPTOR_HEAP_TYPE_RTV;
371 desc.NumDescriptors = APP_NUM_BACK_BUFFERS;
372 desc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_NONE;
373 desc.NodeMask = 1;
374 if (g_pd3dDevice->CreateDescriptorHeap(&desc, IID_PPV_ARGS(&g_pd3dRtvDescHeap)) != S_OK)
375 return false;
376
377 SIZE_T rtvDescriptorSize = g_pd3dDevice->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_RTV);
378 D3D12_CPU_DESCRIPTOR_HANDLE rtvHandle = g_pd3dRtvDescHeap->GetCPUDescriptorHandleForHeapStart();
379 for (UINT i = 0; i < APP_NUM_BACK_BUFFERS; i++)
380 {
381 g_mainRenderTargetDescriptor[i] = rtvHandle;
382 rtvHandle.ptr += rtvDescriptorSize;
383 }
384 }
385
386 {
387 D3D12_DESCRIPTOR_HEAP_DESC desc = {};
388 desc.Type = D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV;
389 desc.NumDescriptors = APP_SRV_HEAP_SIZE;
390 desc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_SHADER_VISIBLE;
391 if (g_pd3dDevice->CreateDescriptorHeap(&desc, IID_PPV_ARGS(&g_pd3dSrvDescHeap)) != S_OK)
392 return false;
393 g_pd3dSrvDescHeapAlloc.Create(g_pd3dDevice, g_pd3dSrvDescHeap);
394 }
395
396 {
397 D3D12_COMMAND_QUEUE_DESC desc = {};
398 desc.Type = D3D12_COMMAND_LIST_TYPE_DIRECT;
399 desc.Flags = D3D12_COMMAND_QUEUE_FLAG_NONE;
400 desc.NodeMask = 1;
401 if (g_pd3dDevice->CreateCommandQueue(&desc, IID_PPV_ARGS(&g_pd3dCommandQueue)) != S_OK)
402 return false;
403 }
404
405 for (UINT i = 0; i < APP_NUM_FRAMES_IN_FLIGHT; i++)
406 if (g_pd3dDevice->CreateCommandAllocator(D3D12_COMMAND_LIST_TYPE_DIRECT, IID_PPV_ARGS(&g_frameContext[i].CommandAllocator)) != S_OK)
407 return false;
408
409 if (g_pd3dDevice->CreateCommandList(0, D3D12_COMMAND_LIST_TYPE_DIRECT, g_frameContext[0].CommandAllocator, nullptr, IID_PPV_ARGS(&g_pd3dCommandList)) != S_OK ||
410 g_pd3dCommandList->Close() != S_OK)
411 return false;
412
413 if (g_pd3dDevice->CreateFence(0, D3D12_FENCE_FLAG_NONE, IID_PPV_ARGS(&g_fence)) != S_OK)
414 return false;
415
416 g_fenceEvent = CreateEvent(nullptr, FALSE, FALSE, nullptr);
417 if (g_fenceEvent == nullptr)
418 return false;
419
420 {
421 IDXGIFactory5* dxgiFactory = nullptr;
422 IDXGISwapChain1* swapChain1 = nullptr;
423 if (CreateDXGIFactory1(IID_PPV_ARGS(&dxgiFactory)) != S_OK)
424 return false;
425
426 BOOL allow_tearing = FALSE;
427 dxgiFactory->CheckFeatureSupport(DXGI_FEATURE_PRESENT_ALLOW_TEARING, &allow_tearing, sizeof(allow_tearing));
428 g_SwapChainTearingSupport = (allow_tearing == TRUE);
429 if (g_SwapChainTearingSupport)
430 sd.Flags |= DXGI_SWAP_CHAIN_FLAG_ALLOW_TEARING;
431
432 if (dxgiFactory->CreateSwapChainForHwnd(g_pd3dCommandQueue, hWnd, &sd, nullptr, nullptr, &swapChain1) != S_OK)
433 return false;
434 if (swapChain1->QueryInterface(IID_PPV_ARGS(&g_pSwapChain)) != S_OK)
435 return false;
436 if (g_SwapChainTearingSupport)
437 dxgiFactory->MakeWindowAssociation(hWnd, DXGI_MWA_NO_ALT_ENTER);
438
439 swapChain1->Release();
440 dxgiFactory->Release();
441 g_pSwapChain->SetMaximumFrameLatency(APP_NUM_BACK_BUFFERS);
442 g_hSwapChainWaitableObject = g_pSwapChain->GetFrameLatencyWaitableObject();
443 }
444
445 CreateRenderTarget();
446 return true;
447}

449void CleanupDeviceD3D()
450{
451 CleanupRenderTarget();
452 if (g_pSwapChain) { g_pSwapChain->SetFullscreenState(false, nullptr); g_pSwapChain->Release(); g_pSwapChain = nullptr; }
453 if (g_hSwapChainWaitableObject != nullptr) { CloseHandle(g_hSwapChainWaitableObject); }
454 for (UINT i = 0; i < APP_NUM_FRAMES_IN_FLIGHT; i++)
455 if (g_frameContext[i].CommandAllocator) { g_frameContext[i].CommandAllocator->Release(); g_frameContext[i].CommandAllocator = nullptr; }
456 if (g_pd3dCommandQueue) { g_pd3dCommandQueue->Release(); g_pd3dCommandQueue = nullptr; }
457 if (g_pd3dCommandList) { g_pd3dCommandList->Release(); g_pd3dCommandList = nullptr; }
458 if (g_pd3dRtvDescHeap) { g_pd3dRtvDescHeap->Release(); g_pd3dRtvDescHeap = nullptr; }
459 if (g_pd3dSrvDescHeap) { g_pd3dSrvDescHeap->Release(); g_pd3dSrvDescHeap = nullptr; }
460 if (g_fence) { g_fence->Release(); g_fence = nullptr; }
461 if (g_fenceEvent) { CloseHandle(g_fenceEvent); g_fenceEvent = nullptr; }
462 if (g_pd3dDevice) { g_pd3dDevice->Release(); g_pd3dDevice = nullptr; }
463
464#ifdef DX12_ENABLE_DEBUG_LAYER
465 IDXGIDebug1* pDebug = nullptr;
466 if (SUCCEEDED(DXGIGetDebugInterface1(0, IID_PPV_ARGS(&pDebug))))
467 {
468 pDebug->ReportLiveObjects(DXGI_DEBUG_ALL, DXGI_DEBUG_RLO_SUMMARY);
469 pDebug->Release();
470 }
471#endif
472}

478 ID3D12Resource* pBackBuffer = nullptr;

481 g_mainRenderTargetResource[i] = pBackBuffer;
482 }
483}

486{
487 WaitForPendingOperations();
488
489 for (UINT i = 0; i < APP_NUM_BACK_BUFFERS; i++)
490 if (g_mainRenderTargetResource[i]) { g_mainRenderTargetResource[i]->Release(); g_mainRenderTargetResource[i] = nullptr; }
491}

493void WaitForPendingOperations()
494{
495 g_pd3dCommandQueue->Signal(g_fence, ++g_fenceLastSignaledValue);
496
497 g_fence->SetEventOnCompletion(g_fenceLastSignaledValue, g_fenceEvent);
498 ::WaitForSingleObject(g_fenceEvent, INFINITE);
499}

501FrameContext* WaitForNextFrameContext()
