191bool CreateDeviceWGL(HWND hWnd, WGL_WindowData* data)
192{
193 HDC hDc = ::GetDC(hWnd);
194 PIXELFORMATDESCRIPTOR pfd = { 0 };
195 pfd.nSize = sizeof(pfd);
196 pfd.nVersion = 1;
197 pfd.dwFlags = PFD_DRAW_TO_WINDOW | PFD_SUPPORT_OPENGL | PFD_DOUBLEBUFFER;
198 pfd.iPixelType = PFD_TYPE_RGBA;
199 pfd.cColorBits = 32;
200
201 const int pf = ::ChoosePixelFormat(hDc, &pfd);
202 if (pf == 0)
203 return false;
204 if (::SetPixelFormat(hDc, pf, &pfd) == FALSE)
205 return false;
206 ::ReleaseDC(hWnd, hDc);
207
208 data->hDC = ::GetDC(hWnd);
209 if (!g_hRC)
210 g_hRC = wglCreateContext(data->hDC);
211 return true;
212}

223// Win32 message handler
224// You can read the io.WantCaptureMouse, io.WantCaptureKeyboard flags to tell if dear imgui wants to use your inputs.
225// - When io.WantCaptureMouse is true, do not dispatch mouse input data to your main application, or clear/overwrite your copy of the mouse data.
226// - When io.WantCaptureKeyboard is true, do not dispatch keyboard input data to your main application, or clear/overwrite your copy of the keyboard data.
227// Generally you may always pass all inputs to dear imgui, and hide them from your application based on those two flags.
228LRESULT WINAPI WndProc(HWND hWnd, UINT msg, WPARAM wParam, LPARAM lParam)
229{
230 if (ImGui_ImplWin32_WndProcHandler(hWnd, msg, wParam, lParam))
231 return true;
232
233 switch (msg)
234 {
235 case WM_SIZE:
236 if (wParam != SIZE_MINIMIZED)
237 {
238 g_Width = LOWORD(lParam);
239 g_Height = HIWORD(lParam);
240 }
241 return 0;
242 case WM_SYSCOMMAND:
243 if ((wParam & 0xfff0) == SC_KEYMENU) // Disable ALT application menu
244 return 0;
245 break;
246 case WM_DESTROY:
247 ::PostQuitMessage(0);
248 return 0;
249 }
250 return ::DefWindowProcW(hWnd, msg, wParam, lParam);
251}
