224// Generate signal based on type
225void GenerateSignal() {
226 sp.ClearVector();
227 signal_data.clear();
228
229 for (int i = 0; i < signal_length; i++) {
230 double t = (double)i / sampling_rate;
231 double value = 0.0;
232
233 switch (current_signal) {
234 case SIGNAL_SINE:
235 value = sin(2 * M_PI * frequency * t);
236 break;
237
238 case SIGNAL_ECG: {
239 // Simplified ECG
240 double phase = fmod(t * frequency * 2 * M_PI, 2 * M_PI);
241 if (phase < M_PI / 4) {
242 value = 0.3 * sin(phase * 4); // P-wave
243 } else if (phase > M_PI / 2 && phase < M_PI * 0.7) {
244 value = 1.5 * sin((phase - M_PI/2) * 10); // QRS
245 } else if (phase > M_PI && phase < M_PI * 1.5) {
246 value = 0.4 * sin((phase - M_PI) * 2); // T-wave
247 }
248 break;
249 }
250
251 case SIGNAL_TURBINE:
252 // Turbine with harmonics
253 value = 1.0 * sin(2 * M_PI * 50 * t); // Fundamental
254 value += 0.3 * sin(2 * M_PI * 50 * t + M_PI/4); // Imbalance
255 value += 0.2 * sin(2 * M_PI * 500 * t); // Bearing
256 break;
257
258 case SIGNAL_NOISE:
259 value = ((double)rand() / RAND_MAX - 0.5) * 2.0;
260 break;
261 }
262
263 // Add noise
264 value += ((double)rand() / RAND_MAX - 0.5) * noise_level;
265
266 signal_data.push_back(value);
267 sp.AddValue(value);
268 }
269}

271// Apply filters
272void ApplyFilters() {
273 filtered_data = signal_data;
274
275 if (apply_kalman) {
276 double* temp = new double[signal_length];
277 sp.KalmanFilter(kalman_process, kalman_measure, temp);
278 for (int i = 0; i < sp.GetIndex(); i++) {
279 filtered_data[i] = temp[i];
280 }
281 delete[] temp;
282 }
283
284 if (apply_median) {
285 SignalProcessing sp_temp;
286 for (size_t i = 0; i < filtered_data.size(); i++) {
287 sp_temp.AddValue(filtered_data[i]);
288 }
289 double* temp = new double[signal_length];
290 sp_temp.MedianFilter(median_window, temp);
291 for (int i = 0; i < sp_temp.GetIndex(); i++) {
292 filtered_data[i] = temp[i];
293 }
294 delete[] temp;
295 }
296
297 if (apply_wavelet) {
298 SignalProcessing sp_temp;
299 for (size_t i = 0; i < filtered_data.size(); i++) {
300 sp_temp.AddValue(filtered_data[i]);
301 }
302 double* temp = new double[signal_length];
303 sp_temp.WaveletDenoise(0.5, temp, 1);
304 for (int i = 0; i < sp_temp.GetIndex(); i++) {
305 filtered_data[i] = temp[i];
306 }
307 delete[] temp;
308 }
309}

311// Detect anomalies
312void DetectAnomalies() {
313 anomaly_indices.clear();
314
315 SignalProcessing sp_check;
316 for (size_t i = 0; i < filtered_data.size(); i++) {
317 sp_check.AddValue(filtered_data[i]);
318 }
319
320 int temp_anomalies[1000];
321 int count = 0;
322
323 switch (anomaly_method) {
324 case 0:
325 count = sp_check.DetectAnomaliesZScore(anomaly_threshold, temp_anomalies, 1000);
326 break;
327 case 1:
328 count = sp_check.DetectAnomaliesIQR(anomaly_threshold, temp_anomalies, 1000);
329 break;
330 case 2:
331 count = sp_check.DetectAnomaliesMAD(20, anomaly_threshold, temp_anomalies, 1000);
332 break;
333 }
334
335 for (int i = 0; i < count; i++) {
336 anomaly_indices.push_back(temp_anomalies[i]);
337 }
338}

340// Compute FFT
341void ComputeFFT() {
342 fft_frequencies.clear();
343 fft_magnitudes.clear();
344
345 SignalProcessing sp_fft;
346 for (size_t i = 0; i < filtered_data.size(); i++) {
347 sp_fft.AddValue(filtered_data[i]);
348 }
349
350 FrequencySpectrum spectrum;
351 if (sp_fft.FFTAnalysis(sampling_rate, &spectrum)) {
352 for (int i = 0; i < spectrum.num_bins && i < 100; i++) {
353 fft_frequencies.push_back(spectrum.bins[i].frequency);
354 fft_magnitudes.push_back(spectrum.bins[i].magnitude);
355 }
356 sp_fft.FreeSpectrum(&spectrum);
357 }
358}

512// Load signal from HDF5
513void LoadSignalFromHDF5() {
514 try {
515 H5File file(load_filename, H5F_ACC_RDONLY);
516
517 // Clear previous data
518 loaded_signal.clear();
519 loaded_filtered.clear();
520 loaded_anomalies.clear();
521 loaded_fft_freq.clear();
522 loaded_fft_mag.clear();
523
524 // Read metadata
525 char info_buffer[512] = "";
526 try {
527 Group session = file.openGroup("/Session");
528 DataSet timestamp_ds = session.openDataSet("timestamp");
529 StrType str_type = timestamp_ds.getStrType();
530 std::string timestamp_str;
531 timestamp_ds.read(timestamp_str, str_type);
532 sprintf(info_buffer, "Session: %s", timestamp_str.c_str());
533 } catch (...) {
534 sprintf(info_buffer, "Loaded from: %s", load_filename);
535 }
536 strcpy(loaded_info, info_buffer);
537
538 // Read original signal
539 try {
540 DataSet dataset = file.openDataSet("/Signals/Raw/original_signal");
541 DataSpace dataspace = dataset.getSpace();
542 hsize_t dims[1];
543 dataspace.getSimpleExtentDims(dims);
544
545 std::vector<float> temp_data(dims[0]);
546 dataset.read(temp_data.data(), PredType::NATIVE_FLOAT);
547
548 loaded_signal.resize(dims[0]);
549 for (size_t i = 0; i < dims[0]; i++) {
550 loaded_signal[i] = temp_data[i];
551 }
552 } catch (...) {
553 // Signal not found
554 }
555
556 // Read filtered signal
557 try {
558 DataSet dataset = file.openDataSet("/Signals/Processed/filtered_signal");
559 DataSpace dataspace = dataset.getSpace();
560 hsize_t dims[1];
561 dataspace.getSimpleExtentDims(dims);
562
563 std::vector<float> temp_data(dims[0]);
564 dataset.read(temp_data.data(), PredType::NATIVE_FLOAT);
565
566 loaded_filtered.resize(dims[0]);
567 for (size_t i = 0; i < dims[0]; i++) {
568 loaded_filtered[i] = temp_data[i];
569 }
570 } catch (...) {
571 // Filtered signal not found
572 }
573
574 // Read anomalies
575 try {
576 DataSet dataset = file.openDataSet("/Anomalies/indices");
577 DataSpace dataspace = dataset.getSpace();
578 hsize_t dims[1];
579 dataspace.getSimpleExtentDims(dims);
580
581 std::vector<float> temp_data(dims[0]);
582 dataset.read(temp_data.data(), PredType::NATIVE_FLOAT);
583
584 loaded_anomalies.resize(dims[0]);
585 for (size_t i = 0; i < dims[0]; i++) {
586 loaded_anomalies[i] = (int)temp_data[i];
587 }
588 } catch (...) {
589 // Anomalies not found
590 }
591
592 // Read FFT data
593 try {
594 DataSet freq_ds = file.openDataSet("/Analysis/FFT/frequencies");
595 DataSet mag_ds = file.openDataSet("/Analysis/FFT/magnitudes");
596
597 DataSpace freq_space = freq_ds.getSpace();
598 hsize_t dims[1];
599 freq_space.getSimpleExtentDims(dims);
600
601 std::vector<float> temp_freq(dims[0]);
602 std::vector<float> temp_mag(dims[0]);
603 freq_ds.read(temp_freq.data(), PredType::NATIVE_FLOAT);
604 mag_ds.read(temp_mag.data(), PredType::NATIVE_FLOAT);
605
606 loaded_fft_freq.resize(dims[0]);
607 loaded_fft_mag.resize(dims[0]);
608 for (size_t i = 0; i < dims[0]; i++) {
609 loaded_fft_freq[i] = temp_freq[i];
610 loaded_fft_mag[i] = temp_mag[i];
611 }
612 } catch (...) {
613 // FFT data not found
614 }
615
616 file.close();
617
618 // Get absolute path
619 char abs_path[PATH_MAX];
620 if (realpath(load_filename, abs_path) != NULL) {
621 load_success = true;
622 load_error = false;
623 sprintf(load_message, "✓ Successfully loaded!\n\nFile:\n%s\n\n"
624 "Original signal: %d samples\n"
625 "Filtered signal: %d samples\n"
626 "Anomalies: %d points\n"
627 "FFT data: %d bins",
628 abs_path,
629 (int)loaded_signal.size(),
630 (int)loaded_filtered.size(),
631 (int)loaded_anomalies.size(),
632 (int)loaded_fft_freq.size());
633 } else {
634 load_success = true;
635 load_error = false;
636 sprintf(load_message, "✓ Successfully loaded %d samples!", (int)loaded_signal.size());
637 }
638
639 show_loaded_data = true;
640
641 } catch (const FileIException& e) {
642 load_success = false;
643 load_error = true;
644 sprintf(load_message, "✗ Error: Cannot open file '%s'\n\nMake sure the file exists.", load_filename);
645 } catch (const Exception& e) {
646 load_success = false;
647 load_error = true;
648 sprintf(load_message, "✗ HDF5 Error: %s", e.getDetailMsg().c_str());
649 } catch (const std::exception& e) {
650 load_success = false;
651 load_error = true;
652 sprintf(load_message, "✗ Error loading file: %s", e.what());
653 }
654}
