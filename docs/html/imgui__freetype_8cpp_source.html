32// About Gamma Correct Blending:
33// - FreeType assumes blending in linear space rather than gamma space.
34// - See https://www.freetype.org/freetype2/docs/reference/ft2-base_interface.html#FT_Render_Glyph
35// - For correct results you need to be using sRGB and convert to linear space in the pixel shader output.
36// - The default dear imgui styles will be impacted by this change (alpha values will need tweaking).
37

175bool ImGui_ImplFreeType_FontSrcData::InitFont(FT_Library ft_library, ImFontConfig* src, ImGuiFreeTypeLoaderFlags extra_font_loader_flags)
176{
177 FT_Error error = FT_New_Memory_Face(ft_library, (const FT_Byte*)src->FontData, (FT_Long)src->FontDataSize, (FT_Long)src->FontNo, &FtFace);
178 if (error != 0)
179 return false;
180 error = FT_Select_Charmap(FtFace, FT_ENCODING_UNICODE);
181 if (error != 0)
182 return false;
183
184 // Convert to FreeType flags (NB: Bold and Oblique are processed separately)
185 UserFlags = (ImGuiFreeTypeLoaderFlags)(src->FontLoaderFlags | extra_font_loader_flags);
186
187 LoadFlags = 0;
188 if ((UserFlags & ImGuiFreeTypeLoaderFlags_Bitmap) == 0)
189 LoadFlags |= FT_LOAD_NO_BITMAP;
190
191 if (UserFlags & ImGuiFreeTypeLoaderFlags_NoHinting)
192 LoadFlags |= FT_LOAD_NO_HINTING;
193 else
194 src->PixelSnapH = true; // FIXME: A bit weird to do this this way.
195
196 if (UserFlags & ImGuiFreeTypeLoaderFlags_NoAutoHint)
197 LoadFlags |= FT_LOAD_NO_AUTOHINT;
198 if (UserFlags & ImGuiFreeTypeLoaderFlags_ForceAutoHint)
199 LoadFlags |= FT_LOAD_FORCE_AUTOHINT;
200 if (UserFlags & ImGuiFreeTypeLoaderFlags_LightHinting)
201 LoadFlags |= FT_LOAD_TARGET_LIGHT;
202 else if (UserFlags & ImGuiFreeTypeLoaderFlags_MonoHinting)
203 LoadFlags |= FT_LOAD_TARGET_MONO;
204 else
205 LoadFlags |= FT_LOAD_TARGET_NORMAL;
206
207 if (UserFlags & ImGuiFreeTypeLoaderFlags_LoadColor)
208 LoadFlags |= FT_LOAD_COLOR;
209
210 return true;
211}

222static const FT_Glyph_Metrics* ImGui_ImplFreeType_LoadGlyph(ImGui_ImplFreeType_FontSrcData* src_data, uint32_t codepoint)
223{
224 uint32_t glyph_index = FT_Get_Char_Index(src_data->FtFace, codepoint);
225 if (glyph_index == 0)
226 return nullptr;
227
228 // If this crash for you: FreeType 2.11.0 has a crash bug on some bitmap/colored fonts.
229 // - https://gitlab.freedesktop.org/freetype/freetype/-/issues/1076
230 // - https://github.com/ocornut/imgui/issues/4567
231 // - https://github.com/ocornut/imgui/issues/4566
232 // You can use FreeType 2.10, or the patched version of 2.11.0 in VcPkg, or probably any upcoming FreeType version.
233 FT_Error error = FT_Load_Glyph(src_data->FtFace, glyph_index, src_data->LoadFlags);
234 if (error)
235 return nullptr;
236
237 // Need an outline for this to work
238 FT_GlyphSlot slot = src_data->FtFace->glyph;
239#if defined(IMGUI_ENABLE_FREETYPE_LUNASVG) || defined(IMGUI_ENABLE_FREETYPE_PLUTOSVG)
240 IM_ASSERT(slot->format == FT_GLYPH_FORMAT_OUTLINE || slot->format == FT_GLYPH_FORMAT_BITMAP || slot->format == FT_GLYPH_FORMAT_SVG);
241#else
242#if ((FREETYPE_MAJOR >= 2) && (FREETYPE_MINOR >= 12))
243 IM_ASSERT(slot->format != FT_GLYPH_FORMAT_SVG && "The font contains SVG glyphs, you'll need to enable IMGUI_ENABLE_FREETYPE_PLUTOSVG or IMGUI_ENABLE_FREETYPE_LUNASVG in imconfig.h and install required libraries in order to use this font");
244#endif
245 IM_ASSERT(slot->format == FT_GLYPH_FORMAT_OUTLINE || slot->format == FT_GLYPH_FORMAT_BITMAP);
246#endif // IMGUI_ENABLE_FREETYPE_LUNASVG
247
248 // Apply convenience transform (this is not picking from real "Bold"/"Italic" fonts! Merely applying FreeType helper transform. Oblique == Slanting)
249 if (src_data->UserFlags & ImGuiFreeTypeLoaderFlags_Bold)
250 FT_GlyphSlot_Embolden(slot);
251 if (src_data->UserFlags & ImGuiFreeTypeLoaderFlags_Oblique)
252 {
253 FT_GlyphSlot_Oblique(slot);
254 //FT_BBox bbox;
255 //FT_Outline_Get_BBox(&slot->outline, &bbox);
256 //slot->metrics.width = bbox.xMax - bbox.xMin;
257 //slot->metrics.height = bbox.yMax - bbox.yMin;
258 }
259
260 return &slot->metrics;
261}

263static void ImGui_ImplFreeType_BlitGlyph(const FT_Bitmap* ft_bitmap, uint32_t* dst, uint32_t dst_pitch)
264{
265 IM_ASSERT(ft_bitmap != nullptr);
266 const uint32_t w = ft_bitmap->width;
267 const uint32_t h = ft_bitmap->rows;
268 const uint8_t* src = ft_bitmap->buffer;
269 const uint32_t src_pitch = ft_bitmap->pitch;
270
271 switch (ft_bitmap->pixel_mode)
272 {
273 case FT_PIXEL_MODE_GRAY: // Grayscale image, 1 byte per pixel.
274 {
275 for (uint32_t y = 0; y < h; y++, src += src_pitch, dst += dst_pitch)
276 for (uint32_t x = 0; x < w; x++)
277 dst[x] = IM_COL32(255, 255, 255, src[x]);
278 break;
279 }
280 case FT_PIXEL_MODE_MONO: // Monochrome image, 1 bit per pixel. The bits in each byte are ordered from MSB to LSB.
281 {
282 for (uint32_t y = 0; y < h; y++, src += src_pitch, dst += dst_pitch)
283 {
284 uint8_t bits = 0;
285 const uint8_t* bits_ptr = src;
286 for (uint32_t x = 0; x < w; x++, bits <<= 1)
287 {
288 if ((x & 7) == 0)
289 bits = *bits_ptr++;
290 dst[x] = IM_COL32(255, 255, 255, (bits & 0x80) ? 255 : 0);
291 }
292 }
293 break;
294 }
295 case FT_PIXEL_MODE_BGRA:
296 {
297 // FIXME: Converting pre-multiplied alpha to straight. Doesn't smell good.
298 #define DE_MULTIPLY(color, alpha) ImMin((ImU32)(255.0f * (float)color / (float)(alpha + FLT_MIN) + 0.5f), 255u)
299 for (uint32_t y = 0; y < h; y++, src += src_pitch, dst += dst_pitch)
300 for (uint32_t x = 0; x < w; x++)
301 {
302 uint8_t r = src[x * 4 + 2], g = src[x * 4 + 1], b = src[x * 4], a = src[x * 4 + 3];
303 dst[x] = IM_COL32(DE_MULTIPLY(r, a), DE_MULTIPLY(g, a), DE_MULTIPLY(b, a), a);
304 }
305 #undef DE_MULTIPLY
306 break;
307 }
308 default:
309 IM_ASSERT(0 && "FreeTypeFont::BlitGlyph(): Unknown bitmap pixel mode!");
310 }
311}

324static void* FreeType_Realloc(FT_Memory /*memory*/, long cur_size, long new_size, void* block)
325{
326 // Implement realloc() as we don't ask user to provide it.
327 if (block == nullptr)
328 return GImGuiFreeTypeAllocFunc((size_t)new_size, GImGuiFreeTypeAllocatorUserData);
329
330 if (new_size == 0)
331 {
332 GImGuiFreeTypeFreeFunc(block, GImGuiFreeTypeAllocatorUserData);
333 return nullptr;
334 }
335
336 if (new_size > cur_size)
337 {
338 void* new_block = GImGuiFreeTypeAllocFunc((size_t)new_size, GImGuiFreeTypeAllocatorUserData);
339 memcpy(new_block, block, (size_t)cur_size);
340 GImGuiFreeTypeFreeFunc(block, GImGuiFreeTypeAllocatorUserData);
341 return new_block;
342 }
343
344 return block;
345}

347static bool ImGui_ImplFreeType_LoaderInit(ImFontAtlas* atlas)
348{
349 IM_ASSERT(atlas->FontLoaderData == nullptr);
350 ImGui_ImplFreeType_Data* bd = IM_NEW(ImGui_ImplFreeType_Data)();
351
352 // FreeType memory management: https://www.freetype.org/freetype2/docs/design/design-4.html
353 bd->MemoryManager.user = nullptr;
354 bd->MemoryManager.alloc = &FreeType_Alloc;
355 bd->MemoryManager.free = &FreeType_Free;
356 bd->MemoryManager.realloc = &FreeType_Realloc;
357
358 // https://www.freetype.org/freetype2/docs/reference/ft2-module_management.html#FT_New_Library
359 FT_Error error = FT_New_Library(&bd->MemoryManager, &bd->Library);
360 if (error != 0)
361 {
362 IM_DELETE(bd);
363 return false;
364 }
365
366 // If you don't call FT_Add_Default_Modules() the rest of code may work, but FreeType won't use our custom allocator.
367 FT_Add_Default_Modules(bd->Library);
368
369#ifdef IMGUI_ENABLE_FREETYPE_LUNASVG
370 // Install svg hooks for FreeType
371 // https://freetype.org/freetype2/docs/reference/ft2-properties.html#svg-hooks
372 // https://freetype.org/freetype2/docs/reference/ft2-svg_fonts.html#svg_fonts
373 SVG_RendererHooks hooks = { ImGuiLunasvgPortInit, ImGuiLunasvgPortFree, ImGuiLunasvgPortRender, ImGuiLunasvgPortPresetSlot };
374 FT_Property_Set(bd->Library, "ot-svg", "svg-hooks", &hooks);
375#endif // IMGUI_ENABLE_FREETYPE_LUNASVG
376#ifdef IMGUI_ENABLE_FREETYPE_PLUTOSVG
377 // With plutosvg, use provided hooks
378 FT_Property_Set(bd->Library, "ot-svg", "svg-hooks", plutosvg_ft_svg_hooks());
379#endif // IMGUI_ENABLE_FREETYPE_PLUTOSVG
380
381 // Store our data
382 atlas->FontLoaderData = (void*)bd;
383
384 return true;
385}

421static bool ImGui_ImplFreeType_FontBakedInit(ImFontAtlas* atlas, ImFontConfig* src, ImFontBaked* baked, void* loader_data_for_baked_src)
422{
423 IM_UNUSED(atlas);
424 float size = baked->Size;
425 if (src->MergeMode && src->SizePixels != 0.0f)
426 size *= (src->SizePixels / baked->OwnerFont->Sources[0]->SizePixels);
427
428 ImGui_ImplFreeType_FontSrcData* bd_font_data = (ImGui_ImplFreeType_FontSrcData*)src->FontLoaderData;
429 bd_font_data->BakedLastActivated = baked;
430
431 // We use one FT_Size per (source + baked) combination.
432 ImGui_ImplFreeType_FontSrcBakedData* bd_baked_data = (ImGui_ImplFreeType_FontSrcBakedData*)loader_data_for_baked_src;
433 IM_ASSERT(bd_baked_data != nullptr);
434 IM_PLACEMENT_NEW(bd_baked_data) ImGui_ImplFreeType_FontSrcBakedData();
435
436 FT_New_Size(bd_font_data->FtFace, &bd_baked_data->FtSize);
437 FT_Activate_Size(bd_baked_data->FtSize);
438
439 // Vuhdo 2017: "I'm not sure how to deal with font sizes properly. As far as I understand, currently ImGui assumes that the 'pixel_height'
440 // is a maximum height of an any given glyph, i.e. it's the sum of font's ascender and descender. Seems strange to me.
441 // FT_Set_Pixel_Sizes() doesn't seem to get us the same result."
442 // (FT_Set_Pixel_Sizes() essentially calls FT_Request_Size() with FT_SIZE_REQUEST_TYPE_NOMINAL)
443 const float rasterizer_density = src->RasterizerDensity * baked->RasterizerDensity;
444 FT_Size_RequestRec req;
445 req.type = (bd_font_data->UserFlags & ImGuiFreeTypeLoaderFlags_Bitmap) ? FT_SIZE_REQUEST_TYPE_NOMINAL : FT_SIZE_REQUEST_TYPE_REAL_DIM;
446 req.width = 0;
447 req.height = (uint32_t)(size * 64 * rasterizer_density);
448 req.horiResolution = 0;
449 req.vertResolution = 0;
450 FT_Request_Size(bd_font_data->FtFace, &req);
451
452 // Output
453 if (src->MergeMode == false)
454 {
455 // Read metrics
456 FT_Size_Metrics metrics = bd_baked_data->FtSize->metrics;
457 const float scale = 1.0f / rasterizer_density;
458 baked->Ascent = (float)FT_CEIL(metrics.ascender) * scale; // The pixel extents above the baseline in pixels (typically positive).
459 baked->Descent = (float)FT_CEIL(metrics.descender) * scale; // The extents below the baseline in pixels (typically negative).
460 //LineSpacing = (float)FT_CEIL(metrics.height) * scale; // The baseline-to-baseline distance. Note that it usually is larger than the sum of the ascender and descender taken as absolute values. There is also no guarantee that no glyphs extend above or below subsequent baselines when using this distance. Think of it as a value the designer of the font finds appropriate.
461 //LineGap = (float)FT_CEIL(metrics.height - metrics.ascender + metrics.descender) * scale; // The spacing in pixels between one row's descent and the next row's ascent.
462 //MaxAdvanceWidth = (float)FT_CEIL(metrics.max_advance) * scale; // This field gives the maximum horizontal cursor advance for all glyphs in the font.
463 }
464 return true;
465}

478static bool ImGui_ImplFreeType_FontBakedLoadGlyph(ImFontAtlas* atlas, ImFontConfig* src, ImFontBaked* baked, void* loader_data_for_baked_src, ImWchar codepoint, ImFontGlyph* out_glyph, float* out_advance_x)
479{
480 ImGui_ImplFreeType_FontSrcData* bd_font_data = (ImGui_ImplFreeType_FontSrcData*)src->FontLoaderData;
481 uint32_t glyph_index = FT_Get_Char_Index(bd_font_data->FtFace, codepoint);
482 if (glyph_index == 0)
483 return false;
484
485 if (bd_font_data->BakedLastActivated != baked) // <-- could use id
486 {
487 // Activate current size
488 ImGui_ImplFreeType_FontSrcBakedData* bd_baked_data = (ImGui_ImplFreeType_FontSrcBakedData*)loader_data_for_baked_src;
489 FT_Activate_Size(bd_baked_data->FtSize);
490 bd_font_data->BakedLastActivated = baked;
491 }
492
493 const FT_Glyph_Metrics* metrics = ImGui_ImplFreeType_LoadGlyph(bd_font_data, codepoint);
494 if (metrics == nullptr)
495 return false;
496
497 FT_Face face = bd_font_data->FtFace;
498 FT_GlyphSlot slot = face->glyph;
499 const float rasterizer_density = src->RasterizerDensity * baked->RasterizerDensity;
500
501 // Load metrics only mode
502 const float advance_x = (slot->advance.x / FT_SCALEFACTOR) / rasterizer_density;
503 if (out_advance_x != NULL)
504 {
505 IM_ASSERT(out_glyph == NULL);
506 *out_advance_x = advance_x;
507 return true;
508 }
509
510 // Render glyph into a bitmap (currently held by FreeType)
511 FT_Render_Mode render_mode = (bd_font_data->UserFlags & ImGuiFreeTypeLoaderFlags_Monochrome) ? FT_RENDER_MODE_MONO : FT_RENDER_MODE_NORMAL;
512 FT_Error error = FT_Render_Glyph(slot, render_mode);
513 const FT_Bitmap* ft_bitmap = &slot->bitmap;
514 if (error != 0 || ft_bitmap == nullptr)
515 return false;
516
517 const int w = (int)ft_bitmap->width;
518 const int h = (int)ft_bitmap->rows;
519 const bool is_visible = (w != 0 && h != 0);
520
521 // Prepare glyph
522 out_glyph->Codepoint = codepoint;
523 out_glyph->AdvanceX = advance_x;
524
525 // Pack and retrieve position inside texture atlas
526 if (is_visible)
527 {
528 ImFontAtlasRectId pack_id = ImFontAtlasPackAddRect(atlas, w, h);
529 if (pack_id == ImFontAtlasRectId_Invalid)
530 {
531 // Pathological out of memory case (TexMaxWidth/TexMaxHeight set too small?)
532 IM_ASSERT(pack_id != ImFontAtlasRectId_Invalid && "Out of texture memory.");
533 return false;
534 }
535 ImTextureRect* r = ImFontAtlasPackGetRect(atlas, pack_id);
536
537 // Render pixels to our temporary buffer
538 atlas->Builder->TempBuffer.resize(w * h * 4);
539 uint32_t* temp_buffer = (uint32_t*)atlas->Builder->TempBuffer.Data;
540 ImGui_ImplFreeType_BlitGlyph(ft_bitmap, temp_buffer, w);
541
542 const float ref_size = baked->OwnerFont->Sources[0]->SizePixels;
543 const float offsets_scale = (ref_size != 0.0f) ? (baked->Size / ref_size) : 1.0f;
544 float font_off_x = (src->GlyphOffset.x * offsets_scale);
545 float font_off_y = (src->GlyphOffset.y * offsets_scale) + baked->Ascent;
546 if (src->PixelSnapH) // Snap scaled offset. This is to mitigate backward compatibility issues for GlyphOffset, but a better design would be welcome.
547 font_off_x = IM_ROUND(font_off_x);
548 if (src->PixelSnapV)
549 font_off_y = IM_ROUND(font_off_y);
550 float recip_h = 1.0f / rasterizer_density;
551 float recip_v = 1.0f / rasterizer_density;
552
553 // Register glyph
554 float glyph_off_x = (float)face->glyph->bitmap_left;
555 float glyph_off_y = (float)-face->glyph->bitmap_top;
556 out_glyph->X0 = glyph_off_x * recip_h + font_off_x;
557 out_glyph->Y0 = glyph_off_y * recip_v + font_off_y;
558 out_glyph->X1 = (glyph_off_x + w) * recip_h + font_off_x;
559 out_glyph->Y1 = (glyph_off_y + h) * recip_v + font_off_y;
560 out_glyph->Visible = true;
561 out_glyph->Colored = (ft_bitmap->pixel_mode == FT_PIXEL_MODE_BGRA);
562 out_glyph->PackId = pack_id;
563 ImFontAtlasBakedSetFontGlyphBitmap(atlas, baked, src, out_glyph, r, (const unsigned char*)temp_buffer, ImTextureFormat_RGBA32, w * 4);
564 }
565
566 return true;
567}

624};

637static FT_Error ImGuiLunasvgPortRender(FT_GlyphSlot slot, FT_Pointer* _state)
638{
639 LunasvgPortState* state = *(LunasvgPortState**)_state;
640
641 // If there was an error while loading the svg in ImGuiLunasvgPortPresetSlot(), the renderer hook still get called, so just returns the error.
642 if (state->err != FT_Err_Ok)
643 return state->err;
644
645 // rows is height, pitch (or stride) equals to width * sizeof(int32)
646 lunasvg::Bitmap bitmap((uint8_t*)slot->bitmap.buffer, slot->bitmap.width, slot->bitmap.rows, slot->bitmap.pitch);
647#if LUNASVG_VERSION_MAJOR >= 3
648 state->svg->render(bitmap, state->matrix); // state->matrix is already scaled and translated
649#else
650 state->svg->setMatrix(state->svg->matrix().identity()); // Reset the svg matrix to the default value
651 state->svg->render(bitmap, state->matrix); // state->matrix is already scaled and translated
652#endif
653 state->err = FT_Err_Ok;
654 return state->err;
655}
656
657static FT_Error ImGuiLunasvgPortPresetSlot(FT_GlyphSlot slot, FT_Bool cache, FT_Pointer* _state)
658{
659 FT_SVG_Document document = (FT_SVG_Document)slot->other;
660 LunasvgPortState* state = *(LunasvgPortState**)_state;
661 FT_Size_Metrics& metrics = document->metrics;
662
663 // This function is called twice, once in the FT_Load_Glyph() and another right before ImGuiLunasvgPortRender().
664 // If it's the latter, don't do anything because it's // already done in the former.
665 if (cache)
666 return state->err;
667
668 state->svg = lunasvg::Document::loadFromData((const char*)document->svg_document, document->svg_document_length);
669 if (state->svg == nullptr)
670 {
671 state->err = FT_Err_Invalid_SVG_Document;
672 return state->err;
673 }
674
675#if LUNASVG_VERSION_MAJOR >= 3
676 lunasvg::Box box = state->svg->boundingBox();
677#else
678 lunasvg::Box box = state->svg->box();
679#endif
680 double scale = std::min(metrics.x_ppem / box.w, metrics.y_ppem / box.h);
681 double xx = (double)document->transform.xx / (1 << 16);
682 double xy = -(double)document->transform.xy / (1 << 16);
683 double yx = -(double)document->transform.yx / (1 << 16);
684 double yy = (double)document->transform.yy / (1 << 16);
685 double x0 = (double)document->delta.x / 64 * box.w / metrics.x_ppem;
686 double y0 = -(double)document->delta.y / 64 * box.h / metrics.y_ppem;
687
688#if LUNASVG_VERSION_MAJOR >= 3
689 // Scale, transform and pre-translate the matrix for the rendering step
690 state->matrix = lunasvg::Matrix::translated(-box.x, -box.y);
691 state->matrix.multiply(lunasvg::Matrix(xx, xy, yx, yy, x0, y0));
692 state->matrix.scale(scale, scale);
693
694 // Apply updated transformation to the bounding box
695 box.transform(state->matrix);
696#else
697 // Scale and transform, we don't translate the svg yet
698 state->matrix.identity();
699 state->matrix.scale(scale, scale);
700 state->matrix.transform(xx, xy, yx, yy, x0, y0);
701 state->svg->setMatrix(state->matrix);
702
703 // Pre-translate the matrix for the rendering step
704 state->matrix.translate(-box.x, -box.y);
705
706 // Get the box again after the transformation
707 box = state->svg->box();
708#endif
709
710 // Calculate the bitmap size
711 slot->bitmap_left = FT_Int(box.x);
712 slot->bitmap_top = FT_Int(-box.y);
713 slot->bitmap.rows = (unsigned int)(ImCeil((float)box.h));
714 slot->bitmap.width = (unsigned int)(ImCeil((float)box.w));
715 slot->bitmap.pitch = slot->bitmap.width * 4;
716 slot->bitmap.pixel_mode = FT_PIXEL_MODE_BGRA;
717
718 // Compute all the bearings and set them correctly. The outline is scaled already, we just need to use the bounding box.
719 double metrics_width = box.w;
720 double metrics_height = box.h;
721 double horiBearingX = box.x;
722 double horiBearingY = -box.y;
723 double vertBearingX = slot->metrics.horiBearingX / 64.0 - slot->metrics.horiAdvance / 64.0 / 2.0;
724 double vertBearingY = (slot->metrics.vertAdvance / 64.0 - slot->metrics.height / 64.0) / 2.0;
725 slot->metrics.width = FT_Pos(IM_ROUND(metrics_width * 64.0)); // Using IM_ROUND() assume width and height are positive
726 slot->metrics.height = FT_Pos(IM_ROUND(metrics_height * 64.0));
727 slot->metrics.horiBearingX = FT_Pos(horiBearingX * 64);
728 slot->metrics.horiBearingY = FT_Pos(horiBearingY * 64);
729 slot->metrics.vertBearingX = FT_Pos(vertBearingX * 64);
730 slot->metrics.vertBearingY = FT_Pos(vertBearingY * 64);
731
732 if (slot->metrics.vertAdvance == 0)
733 slot->metrics.vertAdvance = FT_Pos(metrics_height * 1.2 * 64.0);
734
735 state->err = FT_Err_Ok;
736 return state->err;
737}
