159int32_t ImGui_ImplAndroid_HandleInputEvent(const AInputEvent* input_event)
160{
161 ImGuiIO& io = ImGui::GetIO();
162 int32_t event_type = AInputEvent_getType(input_event);
163 switch (event_type)
164 {
165 case AINPUT_EVENT_TYPE_KEY:
166 {
167 int32_t event_key_code = AKeyEvent_getKeyCode(input_event);
168 int32_t event_scan_code = AKeyEvent_getScanCode(input_event);
169 int32_t event_action = AKeyEvent_getAction(input_event);
170 int32_t event_meta_state = AKeyEvent_getMetaState(input_event);
171
172 io.AddKeyEvent(ImGuiMod_Ctrl, (event_meta_state & AMETA_CTRL_ON) != 0);
173 io.AddKeyEvent(ImGuiMod_Shift, (event_meta_state & AMETA_SHIFT_ON) != 0);
174 io.AddKeyEvent(ImGuiMod_Alt, (event_meta_state & AMETA_ALT_ON) != 0);
175 io.AddKeyEvent(ImGuiMod_Super, (event_meta_state & AMETA_META_ON) != 0);
176
177 switch (event_action)
178 {
179 // FIXME: AKEY_EVENT_ACTION_DOWN and AKEY_EVENT_ACTION_UP occur at once as soon as a touch pointer
180 // goes up from a key. We use a simple key event queue/ and process one event per key per frame in
181 // ImGui_ImplAndroid_NewFrame()...or consider using IO queue, if suitable: https://github.com/ocornut/imgui/issues/2787
182 case AKEY_EVENT_ACTION_DOWN:
183 case AKEY_EVENT_ACTION_UP:
184 {
185 ImGuiKey key = ImGui_ImplAndroid_KeyCodeToImGuiKey(event_key_code);
186 if (key != ImGuiKey_None)
187 {
188 io.AddKeyEvent(key, event_action == AKEY_EVENT_ACTION_DOWN);
189 io.SetKeyEventNativeData(key, event_key_code, event_scan_code);
190 }
191
192 break;
193 }
194 default:
195 break;
196 }
197 break;
198 }
199 case AINPUT_EVENT_TYPE_MOTION:
200 {
201 int32_t event_action = AMotionEvent_getAction(input_event);
202 int32_t event_pointer_index = (event_action & AMOTION_EVENT_ACTION_POINTER_INDEX_MASK) >> AMOTION_EVENT_ACTION_POINTER_INDEX_SHIFT;
203 event_action &= AMOTION_EVENT_ACTION_MASK;
204
205 switch (AMotionEvent_getToolType(input_event, event_pointer_index))
206 {
207 case AMOTION_EVENT_TOOL_TYPE_MOUSE:
208 io.AddMouseSourceEvent(ImGuiMouseSource_Mouse);
209 break;
210 case AMOTION_EVENT_TOOL_TYPE_STYLUS:
211 case AMOTION_EVENT_TOOL_TYPE_ERASER:
212 io.AddMouseSourceEvent(ImGuiMouseSource_Pen);
213 break;
214 case AMOTION_EVENT_TOOL_TYPE_FINGER:
215 default:
216 io.AddMouseSourceEvent(ImGuiMouseSource_TouchScreen);
217 break;
218 }
219
220 switch (event_action)
221 {
222 case AMOTION_EVENT_ACTION_DOWN:
223 case AMOTION_EVENT_ACTION_UP:
224 {
225 // Physical mouse buttons (and probably other physical devices) also invoke the actions AMOTION_EVENT_ACTION_DOWN/_UP,
226 // but we have to process them separately to identify the actual button pressed. This is done below via
227 // AMOTION_EVENT_ACTION_BUTTON_PRESS/_RELEASE. Here, we only process "FINGER" input (and "UNKNOWN", as a fallback).
228 int tool_type = AMotionEvent_getToolType(input_event, event_pointer_index);
229 if (tool_type == AMOTION_EVENT_TOOL_TYPE_FINGER || tool_type == AMOTION_EVENT_TOOL_TYPE_UNKNOWN)
230 {
231 io.AddMousePosEvent(AMotionEvent_getX(input_event, event_pointer_index), AMotionEvent_getY(input_event, event_pointer_index));
232 io.AddMouseButtonEvent(0, event_action == AMOTION_EVENT_ACTION_DOWN);
233 }
234 break;
235 }
236 case AMOTION_EVENT_ACTION_BUTTON_PRESS:
237 case AMOTION_EVENT_ACTION_BUTTON_RELEASE:
238 {
239 int32_t button_state = AMotionEvent_getButtonState(input_event);
240 io.AddMouseButtonEvent(0, (button_state & AMOTION_EVENT_BUTTON_PRIMARY) != 0);
241 io.AddMouseButtonEvent(1, (button_state & AMOTION_EVENT_BUTTON_SECONDARY) != 0);
242 io.AddMouseButtonEvent(2, (button_state & AMOTION_EVENT_BUTTON_TERTIARY) != 0);
243 break;
244 }
245 case AMOTION_EVENT_ACTION_HOVER_MOVE: // Hovering: Tool moves while NOT pressed (such as a physical mouse)
246 case AMOTION_EVENT_ACTION_MOVE: // Touch pointer moves while DOWN
247 io.AddMousePosEvent(AMotionEvent_getX(input_event, event_pointer_index), AMotionEvent_getY(input_event, event_pointer_index));
248 break;
249 case AMOTION_EVENT_ACTION_SCROLL:
250 io.AddMouseWheelEvent(AMotionEvent_getAxisValue(input_event, AMOTION_EVENT_AXIS_HSCROLL, event_pointer_index), AMotionEvent_getAxisValue(input_event, AMOTION_EVENT_AXIS_VSCROLL, event_pointer_index));
251 break;
252 default:
253 break;
254 }
255 }
256 return 1;
257 default:
258 break;
259 }
260
261 return 0;
262}

284void ImGui_ImplAndroid_NewFrame()
285{
286 ImGuiIO& io = ImGui::GetIO();
287
288 // Setup display size (every frame to accommodate for window resizing)
289 int32_t window_width = ANativeWindow_getWidth(g_Window);
290 int32_t window_height = ANativeWindow_getHeight(g_Window);
291 int display_width = window_width;
292 int display_height = window_height;
293
294 io.DisplaySize = ImVec2((float)window_width, (float)window_height);
295 if (window_width > 0 && window_height > 0)
296 io.DisplayFramebufferScale = ImVec2((float)display_width / window_width, (float)display_height / window_height);
297
298 // Setup time step
299 struct timespec current_timespec;
300 clock_gettime(CLOCK_MONOTONIC, &current_timespec);
301 double current_time = (double)(current_timespec.tv_sec) + (current_timespec.tv_nsec / 1000000000.0);
302 io.DeltaTime = g_Time > 0.0 ? (float)(current_time - g_Time) : (float)(1.0f / 60.0f);
303 g_Time = current_time;
304}
