1// dear imgui: Renderer Backend for DirectX10
2// This needs to be used along with a Platform Backend (e.g. Win32)
3
4// Implemented features:
5// [X] Renderer: User texture binding. Use 'ID3D10ShaderResourceView*' as texture identifier. Read the FAQ about ImTextureID/ImTextureRef!
6// [X] Renderer: Large meshes support (64k+ vertices) even with 16-bit indices (ImGuiBackendFlags_RendererHasVtxOffset).
7// [X] Renderer: Texture updates support for dynamic font atlas (ImGuiBackendFlags_RendererHasTextures).
8
9// You can use unmodified imgui_impl_* files in your project. See examples/ folder for examples of using this.

100static void ImGui_ImplDX10_SetupRenderState(ImDrawData* draw_data, ID3D10Device* device)
101{
102 ImGui_ImplDX10_Data* bd = ImGui_ImplDX10_GetBackendData();
103
104 // Setup viewport
105 D3D10_VIEWPORT vp = {};
106 vp.Width = (UINT)(draw_data->DisplaySize.x * draw_data->FramebufferScale.x);
107 vp.Height = (UINT)(draw_data->DisplaySize.y * draw_data->FramebufferScale.y);
108 vp.MinDepth = 0.0f;
109 vp.MaxDepth = 1.0f;
110 vp.TopLeftX = vp.TopLeftY = 0;
111 device->RSSetViewports(1, &vp);
112
113 // Setup orthographic projection matrix into our constant buffer
114 // Our visible imgui space lies from draw_data->DisplayPos (top left) to draw_data->DisplayPos+data_data->DisplaySize (bottom right). DisplayPos is (0,0) for single viewport apps.
115 void* mapped_resource;
116 if (bd->pVertexConstantBuffer->Map(D3D10_MAP_WRITE_DISCARD, 0, &mapped_resource) == S_OK)
117 {
118 VERTEX_CONSTANT_BUFFER_DX10* constant_buffer = (VERTEX_CONSTANT_BUFFER_DX10*)mapped_resource;
119 float L = draw_data->DisplayPos.x;
120 float R = draw_data->DisplayPos.x + draw_data->DisplaySize.x;
121 float T = draw_data->DisplayPos.y;
122 float B = draw_data->DisplayPos.y + draw_data->DisplaySize.y;
123 float mvp[4][4] =
124 {
125 { 2.0f/(R-L), 0.0f, 0.0f, 0.0f },
126 { 0.0f, 2.0f/(T-B), 0.0f, 0.0f },
127 { 0.0f, 0.0f, 0.5f, 0.0f },
128 { (R+L)/(L-R), (T+B)/(B-T), 0.5f, 1.0f },
129 };
130 memcpy(&constant_buffer->mvp, mvp, sizeof(mvp));
131 bd->pVertexConstantBuffer->Unmap();
132 }
133
134 // Setup shader and vertex buffers
135 unsigned int stride = sizeof(ImDrawVert);
136 unsigned int offset = 0;
137 device->IASetInputLayout(bd->pInputLayout);
138 device->IASetVertexBuffers(0, 1, &bd->pVB, &stride, &offset);
139 device->IASetIndexBuffer(bd->pIB, sizeof(ImDrawIdx) == 2 ? DXGI_FORMAT_R16_UINT : DXGI_FORMAT_R32_UINT, 0);
140 device->IASetPrimitiveTopology(D3D10_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
141 device->VSSetShader(bd->pVertexShader);
142 device->VSSetConstantBuffers(0, 1, &bd->pVertexConstantBuffer);
143 device->PSSetShader(bd->pPixelShader);
144 device->PSSetSamplers(0, 1, &bd->pTexSamplerLinear);
145 device->GSSetShader(nullptr);
146
147 // Setup render state
148 const float blend_factor[4] = { 0.f, 0.f, 0.f, 0.f };
149 device->OMSetBlendState(bd->pBlendState, blend_factor, 0xffffffff);
150 device->OMSetDepthStencilState(bd->pDepthStencilState, 0);
151 device->RSSetState(bd->pRasterizerState);
152}

154// Render function
155void ImGui_ImplDX10_RenderDrawData(ImDrawData* draw_data)
156{
157 // Avoid rendering when minimized
158 if (draw_data->DisplaySize.x <= 0.0f || draw_data->DisplaySize.y <= 0.0f)
159 return;
160
161 ImGui_ImplDX10_Data* bd = ImGui_ImplDX10_GetBackendData();
162 ID3D10Device* device = bd->pd3dDevice;
163
164 // Catch up with texture updates. Most of the times, the list will have 1 element with an OK status, aka nothing to do.
165 // (This almost always points to ImGui::GetPlatformIO().Textures[] but is part of ImDrawData to allow overriding or disabling texture updates).
166 if (draw_data->Textures != nullptr)
167 for (ImTextureData* tex : *draw_data->Textures)
168 if (tex->Status != ImTextureStatus_OK)
169 ImGui_ImplDX10_UpdateTexture(tex);
170
171 // Create and grow vertex/index buffers if needed
172 if (!bd->pVB || bd->VertexBufferSize < draw_data->TotalVtxCount)
173 {
174 if (bd->pVB) { bd->pVB->Release(); bd->pVB = nullptr; }
175 bd->VertexBufferSize = draw_data->TotalVtxCount + 5000;
176 D3D10_BUFFER_DESC desc = {};
177 desc.Usage = D3D10_USAGE_DYNAMIC;
178 desc.ByteWidth = bd->VertexBufferSize * sizeof(ImDrawVert);
179 desc.BindFlags = D3D10_BIND_VERTEX_BUFFER;
180 desc.CPUAccessFlags = D3D10_CPU_ACCESS_WRITE;
181 desc.MiscFlags = 0;
182 if (device->CreateBuffer(&desc, nullptr, &bd->pVB) < 0)
183 return;
184 }
185
186 if (!bd->pIB || bd->IndexBufferSize < draw_data->TotalIdxCount)
187 {
188 if (bd->pIB) { bd->pIB->Release(); bd->pIB = nullptr; }
189 bd->IndexBufferSize = draw_data->TotalIdxCount + 10000;
190 D3D10_BUFFER_DESC desc = {};
191 desc.Usage = D3D10_USAGE_DYNAMIC;
192 desc.ByteWidth = bd->IndexBufferSize * sizeof(ImDrawIdx);
193 desc.BindFlags = D3D10_BIND_INDEX_BUFFER;
194 desc.CPUAccessFlags = D3D10_CPU_ACCESS_WRITE;
195 if (device->CreateBuffer(&desc, nullptr, &bd->pIB) < 0)
196 return;
197 }
198
199 // Copy and convert all vertices into a single contiguous buffer
200 ImDrawVert* vtx_dst = nullptr;
201 ImDrawIdx* idx_dst = nullptr;
202 bd->pVB->Map(D3D10_MAP_WRITE_DISCARD, 0, (void**)&vtx_dst);
203 bd->pIB->Map(D3D10_MAP_WRITE_DISCARD, 0, (void**)&idx_dst);
204 for (const ImDrawList* draw_list : draw_data->CmdLists)
205 {
206 memcpy(vtx_dst, draw_list->VtxBuffer.Data, draw_list->VtxBuffer.Size * sizeof(ImDrawVert));
207 memcpy(idx_dst, draw_list->IdxBuffer.Data, draw_list->IdxBuffer.Size * sizeof(ImDrawIdx));
208 vtx_dst += draw_list->VtxBuffer.Size;
209 idx_dst += draw_list->IdxBuffer.Size;
210 }
211 bd->pVB->Unmap();
212 bd->pIB->Unmap();
213
214 // Backup DX state that will be modified to restore it afterwards (unfortunately this is very ugly looking and verbose. Close your eyes!)
215 struct BACKUP_DX10_STATE
216 {
217 UINT ScissorRectsCount, ViewportsCount;
218 D3D10_RECT ScissorRects[D3D10_VIEWPORT_AND_SCISSORRECT_OBJECT_COUNT_PER_PIPELINE];
219 D3D10_VIEWPORT Viewports[D3D10_VIEWPORT_AND_SCISSORRECT_OBJECT_COUNT_PER_PIPELINE];
220 ID3D10RasterizerState* RS;
221 ID3D10BlendState* BlendState;
222 FLOAT BlendFactor[4];
223 UINT SampleMask;
224 UINT StencilRef;
225 ID3D10DepthStencilState* DepthStencilState;
226 ID3D10ShaderResourceView* PSShaderResource;
227 ID3D10SamplerState* PSSampler;
228 ID3D10PixelShader* PS;
229 ID3D10VertexShader* VS;
230 ID3D10GeometryShader* GS;
231 D3D10_PRIMITIVE_TOPOLOGY PrimitiveTopology;
232 ID3D10Buffer* IndexBuffer, *VertexBuffer, *VSConstantBuffer;
233 UINT IndexBufferOffset, VertexBufferStride, VertexBufferOffset;
234 DXGI_FORMAT IndexBufferFormat;
235 ID3D10InputLayout* InputLayout;
236 };
237 BACKUP_DX10_STATE old = {};
238 old.ScissorRectsCount = old.ViewportsCount = D3D10_VIEWPORT_AND_SCISSORRECT_OBJECT_COUNT_PER_PIPELINE;
239 device->RSGetScissorRects(&old.ScissorRectsCount, old.ScissorRects);
240 device->RSGetViewports(&old.ViewportsCount, old.Viewports);
241 device->RSGetState(&old.RS);
242 device->OMGetBlendState(&old.BlendState, old.BlendFactor, &old.SampleMask);
243 device->OMGetDepthStencilState(&old.DepthStencilState, &old.StencilRef);
244 device->PSGetShaderResources(0, 1, &old.PSShaderResource);
245 device->PSGetSamplers(0, 1, &old.PSSampler);
246 device->PSGetShader(&old.PS);
247 device->VSGetShader(&old.VS);
248 device->VSGetConstantBuffers(0, 1, &old.VSConstantBuffer);
249 device->GSGetShader(&old.GS);
250 device->IAGetPrimitiveTopology(&old.PrimitiveTopology);
251 device->IAGetIndexBuffer(&old.IndexBuffer, &old.IndexBufferFormat, &old.IndexBufferOffset);
252 device->IAGetVertexBuffers(0, 1, &old.VertexBuffer, &old.VertexBufferStride, &old.VertexBufferOffset);
253 device->IAGetInputLayout(&old.InputLayout);
254
255 // Setup desired DX state
256 ImGui_ImplDX10_SetupRenderState(draw_data, device);
257 // Setup render state structure (for callbacks and custom texture bindings)
258 ImGuiPlatformIO& platform_io = ImGui::GetPlatformIO();
259 ImGui_ImplDX10_RenderState render_state;
260 render_state.Device = bd->pd3dDevice;
261 render_state.SamplerDefault = bd->pTexSamplerLinear;
262 render_state.VertexConstantBuffer = bd->pVertexConstantBuffer;
263 platform_io.Renderer_RenderState = &render_state;
264
265 // Render command lists
266 // (Because we merged all buffers into a single one, we maintain our own offset into them)
267 int global_vtx_offset = 0;
268 int global_idx_offset = 0;
269 ImVec2 clip_off = draw_data->DisplayPos;
270 ImVec2 clip_scale = draw_data->FramebufferScale;
271 for (const ImDrawList* draw_list : draw_data->CmdLists)
272 {
273 for (int cmd_i = 0; cmd_i < draw_list->CmdBuffer.Size; cmd_i++)
274 {
275 const ImDrawCmd* pcmd = &draw_list->CmdBuffer[cmd_i];
276 if (pcmd->UserCallback != nullptr)
277 {
278 // User callback, registered via ImDrawList::AddCallback()
279 // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
280 if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
281 ImGui_ImplDX10_SetupRenderState(draw_data, device);
282 else
283 pcmd->UserCallback(draw_list, pcmd);
284 }
285 else
286 {
287 // Project scissor/clipping rectangles into framebuffer space
288 ImVec2 clip_min((pcmd->ClipRect.x - clip_off.x) * clip_scale.x, (pcmd->ClipRect.y - clip_off.y) * clip_scale.y);
289 ImVec2 clip_max((pcmd->ClipRect.z - clip_off.x) * clip_scale.x, (pcmd->ClipRect.w - clip_off.y) * clip_scale.y);
290 if (clip_max.x <= clip_min.x || clip_max.y <= clip_min.y)
291 continue;
292
293 // Apply scissor/clipping rectangle
294 const D3D10_RECT r = { (LONG)clip_min.x, (LONG)clip_min.y, (LONG)clip_max.x, (LONG)clip_max.y };
295 device->RSSetScissorRects(1, &r);
296
297 // Bind texture, Draw
298 ID3D10ShaderResourceView* texture_srv = (ID3D10ShaderResourceView*)pcmd->GetTexID();
299 device->PSSetShaderResources(0, 1, &texture_srv);
300 device->DrawIndexed(pcmd->ElemCount, pcmd->IdxOffset + global_idx_offset, pcmd->VtxOffset + global_vtx_offset);
301 }
302 }
303 global_idx_offset += draw_list->IdxBuffer.Size;
304 global_vtx_offset += draw_list->VtxBuffer.Size;
305 }
306 platform_io.Renderer_RenderState = nullptr;
307
308 // Restore modified DX state
309 device->RSSetScissorRects(old.ScissorRectsCount, old.ScissorRects);
310 device->RSSetViewports(old.ViewportsCount, old.Viewports);
311 device->RSSetState(old.RS); if (old.RS) old.RS->Release();
312 device->OMSetBlendState(old.BlendState, old.BlendFactor, old.SampleMask); if (old.BlendState) old.BlendState->Release();
313 device->OMSetDepthStencilState(old.DepthStencilState, old.StencilRef); if (old.DepthStencilState) old.DepthStencilState->Release();
314 device->PSSetShaderResources(0, 1, &old.PSShaderResource); if (old.PSShaderResource) old.PSShaderResource->Release();
315 device->PSSetSamplers(0, 1, &old.PSSampler); if (old.PSSampler) old.PSSampler->Release();
316 device->PSSetShader(old.PS); if (old.PS) old.PS->Release();
317 device->VSSetShader(old.VS); if (old.VS) old.VS->Release();
318 device->GSSetShader(old.GS); if (old.GS) old.GS->Release();
319 device->VSSetConstantBuffers(0, 1, &old.VSConstantBuffer); if (old.VSConstantBuffer) old.VSConstantBuffer->Release();
320 device->IASetPrimitiveTopology(old.PrimitiveTopology);
321 device->IASetIndexBuffer(old.IndexBuffer, old.IndexBufferFormat, old.IndexBufferOffset); if (old.IndexBuffer) old.IndexBuffer->Release();
322 device->IASetVertexBuffers(0, 1, &old.VertexBuffer, &old.VertexBufferStride, &old.VertexBufferOffset); if (old.VertexBuffer) old.VertexBuffer->Release();
323 device->IASetInputLayout(old.InputLayout); if (old.InputLayout) old.InputLayout->Release();
324}

326static void ImGui_ImplDX10_DestroyTexture(ImTextureData* tex)
327{
328 if (ImGui_ImplDX10_Texture* backend_tex = (ImGui_ImplDX10_Texture*)tex->BackendUserData)
329 {
330 IM_ASSERT(backend_tex->pTextureView == (ID3D10ShaderResourceView*)(intptr_t)tex->TexID);
331 backend_tex->pTextureView->Release();
332 backend_tex->pTexture->Release();
333 IM_DELETE(backend_tex);
334
335 // Clear identifiers and mark as destroyed (in order to allow e.g. calling InvalidateDeviceObjects while running)
336 tex->SetTexID(ImTextureID_Invalid);
337 tex->BackendUserData = nullptr;
338 }
339 tex->SetStatus(ImTextureStatus_Destroyed);
340}

342void ImGui_ImplDX10_UpdateTexture(ImTextureData* tex)
343{
344 ImGui_ImplDX10_Data* bd = ImGui_ImplDX10_GetBackendData();
345 if (tex->Status == ImTextureStatus_WantCreate)
346 {
347 // Create and upload new texture to graphics system
348 //IMGUI_DEBUG_LOG("UpdateTexture #%03d: WantCreate %dx%d\n", tex->UniqueID, tex->Width, tex->Height);
349 IM_ASSERT(tex->TexID == ImTextureID_Invalid && tex->BackendUserData == nullptr);
350 IM_ASSERT(tex->Format == ImTextureFormat_RGBA32);
351 unsigned int* pixels = (unsigned int*)tex->GetPixels();
352 ImGui_ImplDX10_Texture* backend_tex = IM_NEW(ImGui_ImplDX10_Texture)();
353
354 // Create texture
355 D3D10_TEXTURE2D_DESC desc;
356 ZeroMemory(&desc, sizeof(desc));
357 desc.Width = (UINT)tex->Width;
358 desc.Height = (UINT)tex->Height;
359 desc.MipLevels = 1;
360 desc.ArraySize = 1;
361 desc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
362 desc.SampleDesc.Count = 1;
363 desc.Usage = D3D10_USAGE_DEFAULT;
364 desc.BindFlags = D3D10_BIND_SHADER_RESOURCE;
365 desc.CPUAccessFlags = 0;
366
367 D3D10_SUBRESOURCE_DATA subResource;
368 subResource.pSysMem = pixels;
369 subResource.SysMemPitch = desc.Width * 4;
370 subResource.SysMemSlicePitch = 0;
371 bd->pd3dDevice->CreateTexture2D(&desc, &subResource, &backend_tex->pTexture);
372 IM_ASSERT(backend_tex->pTexture != nullptr && "Backend failed to create texture!");
373
374 // Create texture view
375 D3D10_SHADER_RESOURCE_VIEW_DESC srv_desc;
376 ZeroMemory(&srv_desc, sizeof(srv_desc));
377 srv_desc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
378 srv_desc.ViewDimension = D3D10_SRV_DIMENSION_TEXTURE2D;
379 srv_desc.Texture2D.MipLevels = desc.MipLevels;
380 srv_desc.Texture2D.MostDetailedMip = 0;
381 bd->pd3dDevice->CreateShaderResourceView(backend_tex->pTexture, &srv_desc, &backend_tex->pTextureView);
382 IM_ASSERT(backend_tex->pTextureView != nullptr && "Backend failed to create texture!");
383
384 // Store identifiers
385 tex->SetTexID((ImTextureID)(intptr_t)backend_tex->pTextureView);
386 tex->SetStatus(ImTextureStatus_OK);
387 tex->BackendUserData = backend_tex;
388 }
389 else if (tex->Status == ImTextureStatus_WantUpdates)
390 {
391 // Update selected blocks. We only ever write to textures regions which have never been used before!
392 // This backend choose to use tex->Updates[] but you can use tex->UpdateRect to upload a single region.
393 ImGui_ImplDX10_Texture* backend_tex = (ImGui_ImplDX10_Texture*)tex->BackendUserData;
394 IM_ASSERT(backend_tex->pTextureView == (ID3D10ShaderResourceView*)(intptr_t)tex->TexID);
395 for (ImTextureRect& r : tex->Updates)
396 {
397 D3D10_BOX box = { (UINT)r.x, (UINT)r.y, (UINT)0, (UINT)(r.x + r.w), (UINT)(r.y + r.h), (UINT)1 };
398 bd->pd3dDevice->UpdateSubresource(backend_tex->pTexture, 0, &box, tex->GetPixelsAt(r.x, r.y), (UINT)tex->GetPitch(), 0);
399 }
400 tex->SetStatus(ImTextureStatus_OK);
401 }
402 if (tex->Status == ImTextureStatus_WantDestroy && tex->UnusedFrames > 0)
403 ImGui_ImplDX10_DestroyTexture(tex);
404}

406bool ImGui_ImplDX10_CreateDeviceObjects()
407{
408 ImGui_ImplDX10_Data* bd = ImGui_ImplDX10_GetBackendData();
409 if (!bd->pd3dDevice)
410 return false;
411 ImGui_ImplDX10_InvalidateDeviceObjects();
412
413 // By using D3DCompile() from <d3dcompiler.h> / d3dcompiler.lib, we introduce a dependency to a given version of d3dcompiler_XX.dll (see D3DCOMPILER_DLL_A)
414 // If you would like to use this DX10 sample code but remove this dependency you can:
415 // 1) compile once, save the compiled shader blobs into a file or source code and pass them to CreateVertexShader()/CreatePixelShader() [preferred solution]
416 // 2) use code to detect any version of the DLL and grab a pointer to D3DCompile from the DLL.
417 // See https://github.com/ocornut/imgui/pull/638 for sources and details.
418
419 // Create the vertex shader
420 {
421 static const char* vertexShader =
422 "cbuffer vertexBuffer : register(b0) \
423 {\
424 float4x4 ProjectionMatrix; \
425 };\
426 struct VS_INPUT\
427 {\
428 float2 pos : POSITION;\
429 float4 col : COLOR0;\
430 float2 uv : TEXCOORD0;\
431 };\
432 \
433 struct PS_INPUT\
434 {\
435 float4 pos : SV_POSITION;\
436 float4 col : COLOR0;\
437 float2 uv : TEXCOORD0;\
438 };\
439 \
440 PS_INPUT main(VS_INPUT input)\
441 {\
442 PS_INPUT output;\
443 output.pos = mul( ProjectionMatrix, float4(input.pos.xy, 0.f, 1.f));\
444 output.col = input.col;\
445 output.uv = input.uv;\
446 return output;\
447 }";
448
449 ID3DBlob* vertexShaderBlob;
450 if (FAILED(D3DCompile(vertexShader, strlen(vertexShader), nullptr, nullptr, nullptr, "main", "vs_4_0", 0, 0, &vertexShaderBlob, nullptr)))
451 return false; // NB: Pass ID3DBlob* pErrorBlob to D3DCompile() to get error showing in (const char*)pErrorBlob->GetBufferPointer(). Make sure to Release() the blob!
452 if (bd->pd3dDevice->CreateVertexShader(vertexShaderBlob->GetBufferPointer(), vertexShaderBlob->GetBufferSize(), &bd->pVertexShader) != S_OK)
453 {
454 vertexShaderBlob->Release();
455 return false;
456 }
457
458 // Create the input layout
459 D3D10_INPUT_ELEMENT_DESC local_layout[] =
460 {
461 { "POSITION", 0, DXGI_FORMAT_R32G32_FLOAT, 0, (UINT)offsetof(ImDrawVert, pos), D3D10_INPUT_PER_VERTEX_DATA, 0 },
462 { "TEXCOORD", 0, DXGI_FORMAT_R32G32_FLOAT, 0, (UINT)offsetof(ImDrawVert, uv), D3D10_INPUT_PER_VERTEX_DATA, 0 },
463 { "COLOR", 0, DXGI_FORMAT_R8G8B8A8_UNORM, 0, (UINT)offsetof(ImDrawVert, col), D3D10_INPUT_PER_VERTEX_DATA, 0 },
464 };
465 if (bd->pd3dDevice->CreateInputLayout(local_layout, 3, vertexShaderBlob->GetBufferPointer(), vertexShaderBlob->GetBufferSize(), &bd->pInputLayout) != S_OK)
466 {
467 vertexShaderBlob->Release();
468 return false;
469 }
470 vertexShaderBlob->Release();
471
472 // Create the constant buffer
473 {
474 D3D10_BUFFER_DESC desc = {};
475 desc.ByteWidth = sizeof(VERTEX_CONSTANT_BUFFER_DX10);
476 desc.Usage = D3D10_USAGE_DYNAMIC;
477 desc.BindFlags = D3D10_BIND_CONSTANT_BUFFER;
478 desc.CPUAccessFlags = D3D10_CPU_ACCESS_WRITE;
479 desc.MiscFlags = 0;
480 bd->pd3dDevice->CreateBuffer(&desc, nullptr, &bd->pVertexConstantBuffer);
481 }
482 }
483
484 // Create the pixel shader
485 {
486 static const char* pixelShader =
487 "struct PS_INPUT\
488 {\
489 float4 pos : SV_POSITION;\
490 float4 col : COLOR0;\
491 float2 uv : TEXCOORD0;\
492 };\
493 sampler sampler0;\
494 Texture2D texture0;\
495 \
496 float4 main(PS_INPUT input) : SV_Target\
497 {\
498 float4 out_col = input.col * texture0.Sample(sampler0, input.uv); \
499 return out_col; \
500 }";
501
502 ID3DBlob* pixelShaderBlob;
503 if (FAILED(D3DCompile(pixelShader, strlen(pixelShader), nullptr, nullptr, nullptr, "main", "ps_4_0", 0, 0, &pixelShaderBlob, nullptr)))
504 return false; // NB: Pass ID3DBlob* pErrorBlob to D3DCompile() to get error showing in (const char*)pErrorBlob->GetBufferPointer(). Make sure to Release() the blob!
505 if (bd->pd3dDevice->CreatePixelShader(pixelShaderBlob->GetBufferPointer(), pixelShaderBlob->GetBufferSize(), &bd->pPixelShader) != S_OK)
506 {
507 pixelShaderBlob->Release();
508 return false;
509 }
510 pixelShaderBlob->Release();
511 }
512
513 // Create the blending setup
514 {
515 D3D10_BLEND_DESC desc;
516 ZeroMemory(&desc, sizeof(desc));
517 desc.AlphaToCoverageEnable = false;
518 desc.BlendEnable[0] = true;
519 desc.SrcBlend = D3D10_BLEND_SRC_ALPHA;
520 desc.DestBlend = D3D10_BLEND_INV_SRC_ALPHA;
521 desc.BlendOp = D3D10_BLEND_OP_ADD;
522 desc.SrcBlendAlpha = D3D10_BLEND_ONE;
523 desc.DestBlendAlpha = D3D10_BLEND_INV_SRC_ALPHA;
524 desc.BlendOpAlpha = D3D10_BLEND_OP_ADD;
525 desc.RenderTargetWriteMask[0] = D3D10_COLOR_WRITE_ENABLE_ALL;
526 bd->pd3dDevice->CreateBlendState(&desc, &bd->pBlendState);
527 }
528
529 // Create the rasterizer state
530 {
531 D3D10_RASTERIZER_DESC desc;
532 ZeroMemory(&desc, sizeof(desc));
533 desc.FillMode = D3D10_FILL_SOLID;
534 desc.CullMode = D3D10_CULL_NONE;
535 desc.ScissorEnable = true;
536 desc.DepthClipEnable = true;
537 bd->pd3dDevice->CreateRasterizerState(&desc, &bd->pRasterizerState);
538 }
539
540 // Create depth-stencil State
541 {
542 D3D10_DEPTH_STENCIL_DESC desc;
543 ZeroMemory(&desc, sizeof(desc));
544 desc.DepthEnable = false;
545 desc.DepthWriteMask = D3D10_DEPTH_WRITE_MASK_ALL;
546 desc.DepthFunc = D3D10_COMPARISON_ALWAYS;
547 desc.StencilEnable = false;
548 desc.FrontFace.StencilFailOp = desc.FrontFace.StencilDepthFailOp = desc.FrontFace.StencilPassOp = D3D10_STENCIL_OP_KEEP;
549 desc.FrontFace.StencilFunc = D3D10_COMPARISON_ALWAYS;
550 desc.BackFace = desc.FrontFace;
551 bd->pd3dDevice->CreateDepthStencilState(&desc, &bd->pDepthStencilState);
552 }
553
554 // Create texture sampler
555 // (Bilinear sampling is required by default. Set 'io.Fonts->Flags |= ImFontAtlasFlags_NoBakedLines' or 'style.AntiAliasedLinesUseTex = false' to allow point/nearest sampling)
556 {
557 D3D10_SAMPLER_DESC desc;
558 ZeroMemory(&desc, sizeof(desc));
559 desc.Filter = D3D10_FILTER_MIN_MAG_MIP_LINEAR;
560 desc.AddressU = D3D10_TEXTURE_ADDRESS_CLAMP;
561 desc.AddressV = D3D10_TEXTURE_ADDRESS_CLAMP;
562 desc.AddressW = D3D10_TEXTURE_ADDRESS_CLAMP;
563 desc.MipLODBias = 0.f;
564 desc.ComparisonFunc = D3D10_COMPARISON_ALWAYS;
565 desc.MinLOD = 0.f;
566 desc.MaxLOD = 0.f;
567 bd->pd3dDevice->CreateSamplerState(&desc, &bd->pTexSamplerLinear);
568 }
569
570 return true;
571}

573void ImGui_ImplDX10_InvalidateDeviceObjects()
574{
575 ImGui_ImplDX10_Data* bd = ImGui_ImplDX10_GetBackendData();
576 if (!bd->pd3dDevice)
577 return;
578
579 // Destroy all textures
580 for (ImTextureData* tex : ImGui::GetPlatformIO().Textures)
581 if (tex->RefCount == 1)
582 ImGui_ImplDX10_DestroyTexture(tex);
583 if (bd->pTexSamplerLinear) { bd->pTexSamplerLinear->Release(); bd->pTexSamplerLinear = nullptr; }
584 if (bd->pIB) { bd->pIB->Release(); bd->pIB = nullptr; }
585 if (bd->pVB) { bd->pVB->Release(); bd->pVB = nullptr; }
586 if (bd->pBlendState) { bd->pBlendState->Release(); bd->pBlendState = nullptr; }
587 if (bd->pDepthStencilState) { bd->pDepthStencilState->Release(); bd->pDepthStencilState = nullptr; }
588 if (bd->pRasterizerState) { bd->pRasterizerState->Release(); bd->pRasterizerState = nullptr; }
589 if (bd->pPixelShader) { bd->pPixelShader->Release(); bd->pPixelShader = nullptr; }
590 if (bd->pVertexConstantBuffer) { bd->pVertexConstantBuffer->Release(); bd->pVertexConstantBuffer = nullptr; }
591 if (bd->pInputLayout) { bd->pInputLayout->Release(); bd->pInputLayout = nullptr; }
592 if (bd->pVertexShader) { bd->pVertexShader->Release(); bd->pVertexShader = nullptr; }
593}

595bool ImGui_ImplDX10_Init(ID3D10Device* device)
596{
597 ImGuiIO& io = ImGui::GetIO();
598 IMGUI_CHECKVERSION();
599 IM_ASSERT(io.BackendRendererUserData == nullptr && "Already initialized a renderer backend!");
600
601 // Setup backend capabilities flags
602 ImGui_ImplDX10_Data* bd = IM_NEW(ImGui_ImplDX10_Data)();
603 io.BackendRendererUserData = (void*)bd;
604 io.BackendRendererName = "imgui_impl_dx10";
605 io.BackendFlags |= ImGuiBackendFlags_RendererHasVtxOffset; // We can honor the ImDrawCmd::VtxOffset field, allowing for large meshes.
606 io.BackendFlags |= ImGuiBackendFlags_RendererHasTextures; // We can honor ImGuiPlatformIO::Textures[] requests during render.
607
608 ImGuiPlatformIO& platform_io = ImGui::GetPlatformIO();
609 platform_io.Renderer_TextureMaxWidth = platform_io.Renderer_TextureMaxHeight = D3D10_REQ_TEXTURE2D_U_OR_V_DIMENSION;
610
611 // Get factory from device
612 IDXGIDevice* pDXGIDevice = nullptr;
613 IDXGIAdapter* pDXGIAdapter = nullptr;
614 IDXGIFactory* pFactory = nullptr;
615 if (device->QueryInterface(IID_PPV_ARGS(&pDXGIDevice)) == S_OK)
616 if (pDXGIDevice->GetParent(IID_PPV_ARGS(&pDXGIAdapter)) == S_OK)
617 if (pDXGIAdapter->GetParent(IID_PPV_ARGS(&pFactory)) == S_OK)
618 {
619 bd->pd3dDevice = device;
620 bd->pFactory = pFactory;
621 }
622 if (pDXGIDevice) pDXGIDevice->Release();
623 if (pDXGIAdapter) pDXGIAdapter->Release();
624 bd->pd3dDevice->AddRef();
625
626 return true;
627}

630{
631 ImGui_ImplDX10_Data* bd = ImGui_ImplDX10_GetBackendData();
632 IM_ASSERT(bd != nullptr && "No renderer backend to shutdown, or already shutdown?");
633 ImGuiIO& io = ImGui::GetIO();
634 ImGuiPlatformIO& platform_io = ImGui::GetPlatformIO();
635
636 ImGui_ImplDX10_InvalidateDeviceObjects();

639
640 io.BackendRendererName = nullptr;
641 io.BackendRendererUserData = nullptr;
642 io.BackendFlags &= ~(ImGuiBackendFlags_RendererHasVtxOffset | ImGuiBackendFlags_RendererHasTextures);
643 platform_io.ClearRendererHandlers();
644 IM_DELETE(bd);
645}
