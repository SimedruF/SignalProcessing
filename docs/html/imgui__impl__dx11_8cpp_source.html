1// dear imgui: Renderer Backend for DirectX11
2// This needs to be used along with a Platform Backend (e.g. Win32)
3
4// Implemented features:
5// [X] Renderer: User texture binding. Use 'ID3D11ShaderResourceView*' as texture identifier. Read the FAQ about ImTextureID/ImTextureRef!
6// [X] Renderer: Large meshes support (64k+ vertices) even with 16-bit indices (ImGuiBackendFlags_RendererHasVtxOffset).
7// [X] Renderer: Texture updates support for dynamic font atlas (ImGuiBackendFlags_RendererHasTextures).
8// [X] Renderer: Expose selected render state for draw callbacks to use. Access in '(ImGui_ImplXXXX_RenderState*)GetPlatformIO().Renderer_RenderState'.
9
10// You can use unmodified imgui_impl_* files in your project. See examples/ folder for examples of using this.
11// Prefer including the entire imgui/ repository into your project (either as a copy or as a submodule), and only build the backends you need.
12// Learn about Dear ImGui:
13// - FAQ https://dearimgui.com/faq
14// - Getting Started https://dearimgui.com/getting-started
15// - Documentation https://dearimgui.com/docs (same as your local docs/ folder).
16// - Introduction, links and more at the top of imgui.cpp

103static void ImGui_ImplDX11_SetupRenderState(const ImDrawData* draw_data, ID3D11DeviceContext* device_ctx)
104{
105 ImGui_ImplDX11_Data* bd = ImGui_ImplDX11_GetBackendData();
106
107 // Setup viewport
108 D3D11_VIEWPORT vp = {};
109 vp.Width = draw_data->DisplaySize.x * draw_data->FramebufferScale.x;
110 vp.Height = draw_data->DisplaySize.y * draw_data->FramebufferScale.y;
111 vp.MinDepth = 0.0f;
112 vp.MaxDepth = 1.0f;
113 vp.TopLeftX = vp.TopLeftY = 0;
114 device_ctx->RSSetViewports(1, &vp);
115
116 // Setup orthographic projection matrix into our constant buffer
117 // Our visible imgui space lies from draw_data->DisplayPos (top left) to draw_data->DisplayPos+data_data->DisplaySize (bottom right). DisplayPos is (0,0) for single viewport apps.
118 D3D11_MAPPED_SUBRESOURCE mapped_resource;
119 if (device_ctx->Map(bd->pVertexConstantBuffer, 0, D3D11_MAP_WRITE_DISCARD, 0, &mapped_resource) == S_OK)
120 {
121 VERTEX_CONSTANT_BUFFER_DX11* constant_buffer = (VERTEX_CONSTANT_BUFFER_DX11*)mapped_resource.pData;
122 float L = draw_data->DisplayPos.x;
123 float R = draw_data->DisplayPos.x + draw_data->DisplaySize.x;
124 float T = draw_data->DisplayPos.y;
125 float B = draw_data->DisplayPos.y + draw_data->DisplaySize.y;
126 float mvp[4][4] =
127 {
128 { 2.0f/(R-L), 0.0f, 0.0f, 0.0f },
129 { 0.0f, 2.0f/(T-B), 0.0f, 0.0f },
130 { 0.0f, 0.0f, 0.5f, 0.0f },
131 { (R+L)/(L-R), (T+B)/(B-T), 0.5f, 1.0f },
132 };
133 memcpy(&constant_buffer->mvp, mvp, sizeof(mvp));
134 device_ctx->Unmap(bd->pVertexConstantBuffer, 0);
135 }
136
137 // Setup shader and vertex buffers
138 unsigned int stride = sizeof(ImDrawVert);
139 unsigned int offset = 0;
140 device_ctx->IASetInputLayout(bd->pInputLayout);
141 device_ctx->IASetVertexBuffers(0, 1, &bd->pVB, &stride, &offset);
142 device_ctx->IASetIndexBuffer(bd->pIB, sizeof(ImDrawIdx) == 2 ? DXGI_FORMAT_R16_UINT : DXGI_FORMAT_R32_UINT, 0);
143 device_ctx->IASetPrimitiveTopology(D3D11_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
144 device_ctx->VSSetShader(bd->pVertexShader, nullptr, 0);
145 device_ctx->VSSetConstantBuffers(0, 1, &bd->pVertexConstantBuffer);
146 device_ctx->PSSetShader(bd->pPixelShader, nullptr, 0);
147 device_ctx->PSSetSamplers(0, 1, &bd->pTexSamplerLinear);
148 device_ctx->GSSetShader(nullptr, nullptr, 0);
149 device_ctx->HSSetShader(nullptr, nullptr, 0); // In theory we should backup and restore this as well.. very infrequently used..
150 device_ctx->DSSetShader(nullptr, nullptr, 0); // In theory we should backup and restore this as well.. very infrequently used..
151 device_ctx->CSSetShader(nullptr, nullptr, 0); // In theory we should backup and restore this as well.. very infrequently used..
152
153 // Setup render state
154 const float blend_factor[4] = { 0.f, 0.f, 0.f, 0.f };
155 device_ctx->OMSetBlendState(bd->pBlendState, blend_factor, 0xffffffff);
156 device_ctx->OMSetDepthStencilState(bd->pDepthStencilState, 0);
157 device_ctx->RSSetState(bd->pRasterizerState);
158}

160// Render function
161void ImGui_ImplDX11_RenderDrawData(ImDrawData* draw_data)
162{
163 // Avoid rendering when minimized
164 if (draw_data->DisplaySize.x <= 0.0f || draw_data->DisplaySize.y <= 0.0f)
165 return;
166
167 ImGui_ImplDX11_Data* bd = ImGui_ImplDX11_GetBackendData();
168 ID3D11DeviceContext* device = bd->pd3dDeviceContext;
169
170 // Catch up with texture updates. Most of the times, the list will have 1 element with an OK status, aka nothing to do.
171 // (This almost always points to ImGui::GetPlatformIO().Textures[] but is part of ImDrawData to allow overriding or disabling texture updates).
172 if (draw_data->Textures != nullptr)
173 for (ImTextureData* tex : *draw_data->Textures)
174 if (tex->Status != ImTextureStatus_OK)
175 ImGui_ImplDX11_UpdateTexture(tex);
176
177 // Create and grow vertex/index buffers if needed
178 if (!bd->pVB || bd->VertexBufferSize < draw_data->TotalVtxCount)
179 {
180 if (bd->pVB) { bd->pVB->Release(); bd->pVB = nullptr; }
181 bd->VertexBufferSize = draw_data->TotalVtxCount + 5000;
182 D3D11_BUFFER_DESC desc = {};
183 desc.Usage = D3D11_USAGE_DYNAMIC;
184 desc.ByteWidth = bd->VertexBufferSize * sizeof(ImDrawVert);
185 desc.BindFlags = D3D11_BIND_VERTEX_BUFFER;
186 desc.CPUAccessFlags = D3D11_CPU_ACCESS_WRITE;
187 desc.MiscFlags = 0;
188 if (bd->pd3dDevice->CreateBuffer(&desc, nullptr, &bd->pVB) < 0)
189 return;
190 }
191 if (!bd->pIB || bd->IndexBufferSize < draw_data->TotalIdxCount)
192 {
193 if (bd->pIB) { bd->pIB->Release(); bd->pIB = nullptr; }
194 bd->IndexBufferSize = draw_data->TotalIdxCount + 10000;
195 D3D11_BUFFER_DESC desc = {};
196 desc.Usage = D3D11_USAGE_DYNAMIC;
197 desc.ByteWidth = bd->IndexBufferSize * sizeof(ImDrawIdx);
198 desc.BindFlags = D3D11_BIND_INDEX_BUFFER;
199 desc.CPUAccessFlags = D3D11_CPU_ACCESS_WRITE;
200 if (bd->pd3dDevice->CreateBuffer(&desc, nullptr, &bd->pIB) < 0)
201 return;
202 }
203
204 // Upload vertex/index data into a single contiguous GPU buffer
205 D3D11_MAPPED_SUBRESOURCE vtx_resource, idx_resource;
206 if (device->Map(bd->pVB, 0, D3D11_MAP_WRITE_DISCARD, 0, &vtx_resource) != S_OK)
207 return;
208 if (device->Map(bd->pIB, 0, D3D11_MAP_WRITE_DISCARD, 0, &idx_resource) != S_OK)
209 return;
210 ImDrawVert* vtx_dst = (ImDrawVert*)vtx_resource.pData;
211 ImDrawIdx* idx_dst = (ImDrawIdx*)idx_resource.pData;
212 for (const ImDrawList* draw_list : draw_data->CmdLists)
213 {
214 memcpy(vtx_dst, draw_list->VtxBuffer.Data, draw_list->VtxBuffer.Size * sizeof(ImDrawVert));
215 memcpy(idx_dst, draw_list->IdxBuffer.Data, draw_list->IdxBuffer.Size * sizeof(ImDrawIdx));
216 vtx_dst += draw_list->VtxBuffer.Size;
217 idx_dst += draw_list->IdxBuffer.Size;
218 }
219 device->Unmap(bd->pVB, 0);
220 device->Unmap(bd->pIB, 0);
221
222 // Backup DX state that will be modified to restore it afterwards (unfortunately this is very ugly looking and verbose. Close your eyes!)
223 struct BACKUP_DX11_STATE
224 {
225 UINT ScissorRectsCount, ViewportsCount;
226 D3D11_RECT ScissorRects[D3D11_VIEWPORT_AND_SCISSORRECT_OBJECT_COUNT_PER_PIPELINE];
227 D3D11_VIEWPORT Viewports[D3D11_VIEWPORT_AND_SCISSORRECT_OBJECT_COUNT_PER_PIPELINE];
228 ID3D11RasterizerState* RS;
229 ID3D11BlendState* BlendState;
230 FLOAT BlendFactor[4];
231 UINT SampleMask;
232 UINT StencilRef;
233 ID3D11DepthStencilState* DepthStencilState;
234 ID3D11ShaderResourceView* PSShaderResource;
235 ID3D11SamplerState* PSSampler;
236 ID3D11PixelShader* PS;
237 ID3D11VertexShader* VS;
238 ID3D11GeometryShader* GS;
239 UINT PSInstancesCount, VSInstancesCount, GSInstancesCount;
240 ID3D11ClassInstance *PSInstances[256], *VSInstances[256], *GSInstances[256]; // 256 is max according to PSSetShader documentation
241 D3D11_PRIMITIVE_TOPOLOGY PrimitiveTopology;
242 ID3D11Buffer* IndexBuffer, *VertexBuffer, *VSConstantBuffer;
243 UINT IndexBufferOffset, VertexBufferStride, VertexBufferOffset;
244 DXGI_FORMAT IndexBufferFormat;
245 ID3D11InputLayout* InputLayout;
246 };
247 BACKUP_DX11_STATE old = {};
248 old.ScissorRectsCount = old.ViewportsCount = D3D11_VIEWPORT_AND_SCISSORRECT_OBJECT_COUNT_PER_PIPELINE;
249 device->RSGetScissorRects(&old.ScissorRectsCount, old.ScissorRects);
250 device->RSGetViewports(&old.ViewportsCount, old.Viewports);
251 device->RSGetState(&old.RS);
252 device->OMGetBlendState(&old.BlendState, old.BlendFactor, &old.SampleMask);
253 device->OMGetDepthStencilState(&old.DepthStencilState, &old.StencilRef);
254 device->PSGetShaderResources(0, 1, &old.PSShaderResource);
255 device->PSGetSamplers(0, 1, &old.PSSampler);
256 old.PSInstancesCount = old.VSInstancesCount = old.GSInstancesCount = 256;
257 device->PSGetShader(&old.PS, old.PSInstances, &old.PSInstancesCount);
258 device->VSGetShader(&old.VS, old.VSInstances, &old.VSInstancesCount);
259 device->VSGetConstantBuffers(0, 1, &old.VSConstantBuffer);
260 device->GSGetShader(&old.GS, old.GSInstances, &old.GSInstancesCount);
261
262 device->IAGetPrimitiveTopology(&old.PrimitiveTopology);
263 device->IAGetIndexBuffer(&old.IndexBuffer, &old.IndexBufferFormat, &old.IndexBufferOffset);
264 device->IAGetVertexBuffers(0, 1, &old.VertexBuffer, &old.VertexBufferStride, &old.VertexBufferOffset);
265 device->IAGetInputLayout(&old.InputLayout);
266
267 // Setup desired DX state
268 ImGui_ImplDX11_SetupRenderState(draw_data, device);
269
270 // Setup render state structure (for callbacks and custom texture bindings)
271 ImGuiPlatformIO& platform_io = ImGui::GetPlatformIO();
272 ImGui_ImplDX11_RenderState render_state;
273 render_state.Device = bd->pd3dDevice;
274 render_state.DeviceContext = bd->pd3dDeviceContext;
275 render_state.SamplerDefault = bd->pTexSamplerLinear;
276 render_state.VertexConstantBuffer = bd->pVertexConstantBuffer;
277 platform_io.Renderer_RenderState = &render_state;
278
279 // Render command lists
280 // (Because we merged all buffers into a single one, we maintain our own offset into them)
281 int global_idx_offset = 0;
282 int global_vtx_offset = 0;
283 ImVec2 clip_off = draw_data->DisplayPos;
284 ImVec2 clip_scale = draw_data->FramebufferScale;
285 for (const ImDrawList* draw_list : draw_data->CmdLists)
286 {
287 for (int cmd_i = 0; cmd_i < draw_list->CmdBuffer.Size; cmd_i++)
288 {
289 const ImDrawCmd* pcmd = &draw_list->CmdBuffer[cmd_i];
290 if (pcmd->UserCallback != nullptr)
291 {
292 // User callback, registered via ImDrawList::AddCallback()
293 // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
294 if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
295 ImGui_ImplDX11_SetupRenderState(draw_data, device);
296 else
297 pcmd->UserCallback(draw_list, pcmd);
298 }
299 else
300 {
301 // Project scissor/clipping rectangles into framebuffer space
302 ImVec2 clip_min((pcmd->ClipRect.x - clip_off.x) * clip_scale.x, (pcmd->ClipRect.y - clip_off.y) * clip_scale.y);
303 ImVec2 clip_max((pcmd->ClipRect.z - clip_off.x) * clip_scale.x, (pcmd->ClipRect.w - clip_off.y) * clip_scale.y);
304 if (clip_max.x <= clip_min.x || clip_max.y <= clip_min.y)
305 continue;
306
307 // Apply scissor/clipping rectangle
308 const D3D11_RECT r = { (LONG)clip_min.x, (LONG)clip_min.y, (LONG)clip_max.x, (LONG)clip_max.y };
309 device->RSSetScissorRects(1, &r);
310
311 // Bind texture, Draw
312 ID3D11ShaderResourceView* texture_srv = (ID3D11ShaderResourceView*)pcmd->GetTexID();
313 device->PSSetShaderResources(0, 1, &texture_srv);
314 device->DrawIndexed(pcmd->ElemCount, pcmd->IdxOffset + global_idx_offset, pcmd->VtxOffset + global_vtx_offset);
315 }
316 }
317 global_idx_offset += draw_list->IdxBuffer.Size;
318 global_vtx_offset += draw_list->VtxBuffer.Size;
319 }
320 platform_io.Renderer_RenderState = nullptr;
321
322 // Restore modified DX state
323 device->RSSetScissorRects(old.ScissorRectsCount, old.ScissorRects);
324 device->RSSetViewports(old.ViewportsCount, old.Viewports);
325 device->RSSetState(old.RS); if (old.RS) old.RS->Release();
326 device->OMSetBlendState(old.BlendState, old.BlendFactor, old.SampleMask); if (old.BlendState) old.BlendState->Release();
327 device->OMSetDepthStencilState(old.DepthStencilState, old.StencilRef); if (old.DepthStencilState) old.DepthStencilState->Release();
328 device->PSSetShaderResources(0, 1, &old.PSShaderResource); if (old.PSShaderResource) old.PSShaderResource->Release();
329 device->PSSetSamplers(0, 1, &old.PSSampler); if (old.PSSampler) old.PSSampler->Release();
330 device->PSSetShader(old.PS, old.PSInstances, old.PSInstancesCount); if (old.PS) old.PS->Release();
331 for (UINT i = 0; i < old.PSInstancesCount; i++) if (old.PSInstances[i]) old.PSInstances[i]->Release();
332 device->VSSetShader(old.VS, old.VSInstances, old.VSInstancesCount); if (old.VS) old.VS->Release();
333 device->VSSetConstantBuffers(0, 1, &old.VSConstantBuffer); if (old.VSConstantBuffer) old.VSConstantBuffer->Release();
334 device->GSSetShader(old.GS, old.GSInstances, old.GSInstancesCount); if (old.GS) old.GS->Release();
335 for (UINT i = 0; i < old.VSInstancesCount; i++) if (old.VSInstances[i]) old.VSInstances[i]->Release();
336 device->IASetPrimitiveTopology(old.PrimitiveTopology);
337 device->IASetIndexBuffer(old.IndexBuffer, old.IndexBufferFormat, old.IndexBufferOffset); if (old.IndexBuffer) old.IndexBuffer->Release();
338 device->IASetVertexBuffers(0, 1, &old.VertexBuffer, &old.VertexBufferStride, &old.VertexBufferOffset); if (old.VertexBuffer) old.VertexBuffer->Release();
339 device->IASetInputLayout(old.InputLayout); if (old.InputLayout) old.InputLayout->Release();
340}

342static void ImGui_ImplDX11_DestroyTexture(ImTextureData* tex)
343{
344 if (ImGui_ImplDX11_Texture* backend_tex = (ImGui_ImplDX11_Texture*)tex->BackendUserData)
345 {
346 IM_ASSERT(backend_tex->pTextureView == (ID3D11ShaderResourceView*)(intptr_t)tex->TexID);
347 backend_tex->pTextureView->Release();
348 backend_tex->pTexture->Release();
349 IM_DELETE(backend_tex);
350
351 // Clear identifiers and mark as destroyed (in order to allow e.g. calling InvalidateDeviceObjects while running)
352 tex->SetTexID(ImTextureID_Invalid);
353 tex->BackendUserData = nullptr;
354 }
355 tex->SetStatus(ImTextureStatus_Destroyed);
356}

358void ImGui_ImplDX11_UpdateTexture(ImTextureData* tex)
359{
360 ImGui_ImplDX11_Data* bd = ImGui_ImplDX11_GetBackendData();
361 if (tex->Status == ImTextureStatus_WantCreate)
362 {
363 // Create and upload new texture to graphics system
364 //IMGUI_DEBUG_LOG("UpdateTexture #%03d: WantCreate %dx%d\n", tex->UniqueID, tex->Width, tex->Height);
365 IM_ASSERT(tex->TexID == ImTextureID_Invalid && tex->BackendUserData == nullptr);
366 IM_ASSERT(tex->Format == ImTextureFormat_RGBA32);
367 unsigned int* pixels = (unsigned int*)tex->GetPixels();
368 ImGui_ImplDX11_Texture* backend_tex = IM_NEW(ImGui_ImplDX11_Texture)();
369
370 // Create texture
371 D3D11_TEXTURE2D_DESC desc;
372 ZeroMemory(&desc, sizeof(desc));
373 desc.Width = (UINT)tex->Width;
374 desc.Height = (UINT)tex->Height;
375 desc.MipLevels = 1;
376 desc.ArraySize = 1;
377 desc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
378 desc.SampleDesc.Count = 1;
379 desc.Usage = D3D11_USAGE_DEFAULT;
380 desc.BindFlags = D3D11_BIND_SHADER_RESOURCE;
381 desc.CPUAccessFlags = 0;
382 D3D11_SUBRESOURCE_DATA subResource;
383 subResource.pSysMem = pixels;
384 subResource.SysMemPitch = desc.Width * 4;
385 subResource.SysMemSlicePitch = 0;
386 bd->pd3dDevice->CreateTexture2D(&desc, &subResource, &backend_tex->pTexture);
387 IM_ASSERT(backend_tex->pTexture != nullptr && "Backend failed to create texture!");
388
389 // Create texture view
390 D3D11_SHADER_RESOURCE_VIEW_DESC srvDesc;
391 ZeroMemory(&srvDesc, sizeof(srvDesc));
392 srvDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
393 srvDesc.ViewDimension = D3D11_SRV_DIMENSION_TEXTURE2D;
394 srvDesc.Texture2D.MipLevels = desc.MipLevels;
395 srvDesc.Texture2D.MostDetailedMip = 0;
396 bd->pd3dDevice->CreateShaderResourceView(backend_tex->pTexture, &srvDesc, &backend_tex->pTextureView);
397 IM_ASSERT(backend_tex->pTextureView != nullptr && "Backend failed to create texture!");
398
399 // Store identifiers
400 tex->SetTexID((ImTextureID)(intptr_t)backend_tex->pTextureView);
401 tex->SetStatus(ImTextureStatus_OK);
402 tex->BackendUserData = backend_tex;
403 }
404 else if (tex->Status == ImTextureStatus_WantUpdates)
405 {
406 // Update selected blocks. We only ever write to textures regions which have never been used before!
407 // This backend choose to use tex->Updates[] but you can use tex->UpdateRect to upload a single region.
408 ImGui_ImplDX11_Texture* backend_tex = (ImGui_ImplDX11_Texture*)tex->BackendUserData;
409 IM_ASSERT(backend_tex->pTextureView == (ID3D11ShaderResourceView*)(intptr_t)tex->TexID);
410 for (ImTextureRect& r : tex->Updates)
411 {
412 D3D11_BOX box = { (UINT)r.x, (UINT)r.y, (UINT)0, (UINT)(r.x + r.w), (UINT)(r.y + r .h), (UINT)1 };
413 bd->pd3dDeviceContext->UpdateSubresource(backend_tex->pTexture, 0, &box, tex->GetPixelsAt(r.x, r.y), (UINT)tex->GetPitch(), 0);
414 }
415 tex->SetStatus(ImTextureStatus_OK);
416 }
417 if (tex->Status == ImTextureStatus_WantDestroy && tex->UnusedFrames > 0)
418 ImGui_ImplDX11_DestroyTexture(tex);
419}

421bool ImGui_ImplDX11_CreateDeviceObjects()
422{
423 ImGui_ImplDX11_Data* bd = ImGui_ImplDX11_GetBackendData();
424 if (!bd->pd3dDevice)
425 return false;
426 ImGui_ImplDX11_InvalidateDeviceObjects();
427
428 // By using D3DCompile() from <d3dcompiler.h> / d3dcompiler.lib, we introduce a dependency to a given version of d3dcompiler_XX.dll (see D3DCOMPILER_DLL_A)
429 // If you would like to use this DX11 sample code but remove this dependency you can:

455 PS_INPUT main(VS_INPUT input)\

457 PS_INPUT output;\
458 output.pos = mul( ProjectionMatrix, float4(input.pos.xy, 0.f, 1.f));\
459 output.col = input.col;\
460 output.uv = input.uv;\

462 }";

588void ImGui_ImplDX11_InvalidateDeviceObjects()
589{
590 ImGui_ImplDX11_Data* bd = ImGui_ImplDX11_GetBackendData();
591 if (!bd->pd3dDevice)
592 return;
593
594 // Destroy all textures
595 for (ImTextureData* tex : ImGui::GetPlatformIO().Textures)
596 if (tex->RefCount == 1)
597 ImGui_ImplDX11_DestroyTexture(tex);
598
599 if (bd->pTexSamplerLinear) { bd->pTexSamplerLinear->Release(); bd->pTexSamplerLinear = nullptr; }
600 if (bd->pIB) { bd->pIB->Release(); bd->pIB = nullptr; }
601 if (bd->pVB) { bd->pVB->Release(); bd->pVB = nullptr; }
602 if (bd->pBlendState) { bd->pBlendState->Release(); bd->pBlendState = nullptr; }
603 if (bd->pDepthStencilState) { bd->pDepthStencilState->Release(); bd->pDepthStencilState = nullptr; }
604 if (bd->pRasterizerState) { bd->pRasterizerState->Release(); bd->pRasterizerState = nullptr; }
605 if (bd->pPixelShader) { bd->pPixelShader->Release(); bd->pPixelShader = nullptr; }
606 if (bd->pVertexConstantBuffer) { bd->pVertexConstantBuffer->Release(); bd->pVertexConstantBuffer = nullptr; }
607 if (bd->pInputLayout) { bd->pInputLayout->Release(); bd->pInputLayout = nullptr; }
608 if (bd->pVertexShader) { bd->pVertexShader->Release(); bd->pVertexShader = nullptr; }
609}

611bool ImGui_ImplDX11_Init(ID3D11Device* device, ID3D11DeviceContext* device_context)
612{
613 ImGuiIO& io = ImGui::GetIO();
614 IMGUI_CHECKVERSION();
615 IM_ASSERT(io.BackendRendererUserData == nullptr && "Already initialized a renderer backend!");
616
617 // Setup backend capabilities flags
618 ImGui_ImplDX11_Data* bd = IM_NEW(ImGui_ImplDX11_Data)();
619 io.BackendRendererUserData = (void*)bd;
620 io.BackendRendererName = "imgui_impl_dx11";
621 io.BackendFlags |= ImGuiBackendFlags_RendererHasVtxOffset; // We can honor the ImDrawCmd::VtxOffset field, allowing for large meshes.
622 io.BackendFlags |= ImGuiBackendFlags_RendererHasTextures; // We can honor ImGuiPlatformIO::Textures[] requests during render.
623
624 ImGuiPlatformIO& platform_io = ImGui::GetPlatformIO();
625 platform_io.Renderer_TextureMaxWidth = platform_io.Renderer_TextureMaxHeight = D3D11_REQ_TEXTURE2D_U_OR_V_DIMENSION;
626
627 // Get factory from device
628 IDXGIDevice* pDXGIDevice = nullptr;
629 IDXGIAdapter* pDXGIAdapter = nullptr;
630 IDXGIFactory* pFactory = nullptr;
631
632 if (device->QueryInterface(IID_PPV_ARGS(&pDXGIDevice)) == S_OK)
633 if (pDXGIDevice->GetParent(IID_PPV_ARGS(&pDXGIAdapter)) == S_OK)
634 if (pDXGIAdapter->GetParent(IID_PPV_ARGS(&pFactory)) == S_OK)
635 {
636 bd->pd3dDevice = device;
637 bd->pd3dDeviceContext = device_context;
638 bd->pFactory = pFactory;
639 }
640 if (pDXGIDevice) pDXGIDevice->Release();
641 if (pDXGIAdapter) pDXGIAdapter->Release();
642 bd->pd3dDevice->AddRef();
643 bd->pd3dDeviceContext->AddRef();
644
645 return true;
646}

652 ImGuiIO& io = ImGui::GetIO();

664 IM_DELETE(bd);
