144static void ImGui_ImplDX12_SetupRenderState(ImDrawData* draw_data, ID3D12GraphicsCommandList* command_list, ImGui_ImplDX12_RenderBuffers* fr)
145{
146 ImGui_ImplDX12_Data* bd = ImGui_ImplDX12_GetBackendData();
147
148 // Setup orthographic projection matrix into our constant buffer
149 // Our visible imgui space lies from draw_data->DisplayPos (top left) to draw_data->DisplayPos+data_data->DisplaySize (bottom right).
150 VERTEX_CONSTANT_BUFFER_DX12 vertex_constant_buffer;
151 {
152 float L = draw_data->DisplayPos.x;
153 float R = draw_data->DisplayPos.x + draw_data->DisplaySize.x;
154 float T = draw_data->DisplayPos.y;
155 float B = draw_data->DisplayPos.y + draw_data->DisplaySize.y;
156 float mvp[4][4] =
157 {
158 { 2.0f/(R-L), 0.0f, 0.0f, 0.0f },
159 { 0.0f, 2.0f/(T-B), 0.0f, 0.0f },
160 { 0.0f, 0.0f, 0.5f, 0.0f },
161 { (R+L)/(L-R), (T+B)/(B-T), 0.5f, 1.0f },
162 };
163 memcpy(&vertex_constant_buffer.mvp, mvp, sizeof(mvp));
164 }
165
166 // Setup viewport
167 D3D12_VIEWPORT vp = {};
168 vp.Width = draw_data->DisplaySize.x * draw_data->FramebufferScale.x;
169 vp.Height = draw_data->DisplaySize.y * draw_data->FramebufferScale.y;
170 vp.MinDepth = 0.0f;
171 vp.MaxDepth = 1.0f;
172 vp.TopLeftX = vp.TopLeftY = 0.0f;
173 command_list->RSSetViewports(1, &vp);
174
175 // Bind shader and vertex buffers
176 unsigned int stride = sizeof(ImDrawVert);
177 unsigned int offset = 0;
178 D3D12_VERTEX_BUFFER_VIEW vbv = {};
179 vbv.BufferLocation = fr->VertexBuffer->GetGPUVirtualAddress() + offset;
180 vbv.SizeInBytes = fr->VertexBufferSize * stride;
181 vbv.StrideInBytes = stride;
182 command_list->IASetVertexBuffers(0, 1, &vbv);
183 D3D12_INDEX_BUFFER_VIEW ibv = {};
184 ibv.BufferLocation = fr->IndexBuffer->GetGPUVirtualAddress();
185 ibv.SizeInBytes = fr->IndexBufferSize * sizeof(ImDrawIdx);
186 ibv.Format = sizeof(ImDrawIdx) == 2 ? DXGI_FORMAT_R16_UINT : DXGI_FORMAT_R32_UINT;
187 command_list->IASetIndexBuffer(&ibv);
188 command_list->IASetPrimitiveTopology(D3D_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
189 command_list->SetPipelineState(bd->pPipelineState);
190 command_list->SetGraphicsRootSignature(bd->pRootSignature);
191 command_list->SetGraphicsRoot32BitConstants(0, 16, &vertex_constant_buffer, 0);
192
193 // Setup blend factor
194 const float blend_factor[4] = { 0.f, 0.f, 0.f, 0.f };
195 command_list->OMSetBlendFactor(blend_factor);
196}

206// Render function
207void ImGui_ImplDX12_RenderDrawData(ImDrawData* draw_data, ID3D12GraphicsCommandList* command_list)
208{
209 // Avoid rendering when minimized
210 if (draw_data->DisplaySize.x <= 0.0f || draw_data->DisplaySize.y <= 0.0f)
211 return;
212
213 // Catch up with texture updates. Most of the times, the list will have 1 element with an OK status, aka nothing to do.
214 // (This almost always points to ImGui::GetPlatformIO().Textures[] but is part of ImDrawData to allow overriding or disabling texture updates).
215 if (draw_data->Textures != nullptr)
216 for (ImTextureData* tex : *draw_data->Textures)
217 if (tex->Status != ImTextureStatus_OK)
218 ImGui_ImplDX12_UpdateTexture(tex);
219
220 // FIXME: We are assuming that this only gets called once per frame!
221 ImGui_ImplDX12_Data* bd = ImGui_ImplDX12_GetBackendData();
222 bd->frameIndex = bd->frameIndex + 1;
223 ImGui_ImplDX12_RenderBuffers* fr = &bd->pFrameResources[bd->frameIndex % bd->numFramesInFlight];
224
225 // Create and grow vertex/index buffers if needed
226 if (fr->VertexBuffer == nullptr || fr->VertexBufferSize < draw_data->TotalVtxCount)
227 {
228 SafeRelease(fr->VertexBuffer);
229 fr->VertexBufferSize = draw_data->TotalVtxCount + 5000;
230 D3D12_HEAP_PROPERTIES props = {};
231 props.Type = D3D12_HEAP_TYPE_UPLOAD;
232 props.CPUPageProperty = D3D12_CPU_PAGE_PROPERTY_UNKNOWN;
233 props.MemoryPoolPreference = D3D12_MEMORY_POOL_UNKNOWN;
234 D3D12_RESOURCE_DESC desc = {};
235 desc.Dimension = D3D12_RESOURCE_DIMENSION_BUFFER;
236 desc.Width = fr->VertexBufferSize * sizeof(ImDrawVert);
237 desc.Height = 1;
238 desc.DepthOrArraySize = 1;
239 desc.MipLevels = 1;
240 desc.Format = DXGI_FORMAT_UNKNOWN;
241 desc.SampleDesc.Count = 1;
242 desc.Layout = D3D12_TEXTURE_LAYOUT_ROW_MAJOR;
243 desc.Flags = D3D12_RESOURCE_FLAG_NONE;
244 if (bd->pd3dDevice->CreateCommittedResource(&props, D3D12_HEAP_FLAG_NONE, &desc, D3D12_RESOURCE_STATE_GENERIC_READ, nullptr, IID_PPV_ARGS(&fr->VertexBuffer)) < 0)
245 return;
246 }
247 if (fr->IndexBuffer == nullptr || fr->IndexBufferSize < draw_data->TotalIdxCount)
248 {
249 SafeRelease(fr->IndexBuffer);
250 fr->IndexBufferSize = draw_data->TotalIdxCount + 10000;
251 D3D12_HEAP_PROPERTIES props = {};
252 props.Type = D3D12_HEAP_TYPE_UPLOAD;
253 props.CPUPageProperty = D3D12_CPU_PAGE_PROPERTY_UNKNOWN;
254 props.MemoryPoolPreference = D3D12_MEMORY_POOL_UNKNOWN;
255 D3D12_RESOURCE_DESC desc = {};
256 desc.Dimension = D3D12_RESOURCE_DIMENSION_BUFFER;
257 desc.Width = fr->IndexBufferSize * sizeof(ImDrawIdx);
258 desc.Height = 1;
259 desc.DepthOrArraySize = 1;
260 desc.MipLevels = 1;
261 desc.Format = DXGI_FORMAT_UNKNOWN;
262 desc.SampleDesc.Count = 1;
263 desc.Layout = D3D12_TEXTURE_LAYOUT_ROW_MAJOR;
264 desc.Flags = D3D12_RESOURCE_FLAG_NONE;
265 if (bd->pd3dDevice->CreateCommittedResource(&props, D3D12_HEAP_FLAG_NONE, &desc, D3D12_RESOURCE_STATE_GENERIC_READ, nullptr, IID_PPV_ARGS(&fr->IndexBuffer)) < 0)
266 return;
267 }
268
269 // Upload vertex/index data into a single contiguous GPU buffer
270 // During Map() we specify a null read range (as per DX12 API, this is informational and for tooling only)
271 void* vtx_resource, *idx_resource;
272 D3D12_RANGE range = { 0, 0 };
273 if (fr->VertexBuffer->Map(0, &range, &vtx_resource) != S_OK)
274 return;
275 if (fr->IndexBuffer->Map(0, &range, &idx_resource) != S_OK)
276 return;
277 ImDrawVert* vtx_dst = (ImDrawVert*)vtx_resource;
278 ImDrawIdx* idx_dst = (ImDrawIdx*)idx_resource;
279 for (const ImDrawList* draw_list : draw_data->CmdLists)
280 {
281 memcpy(vtx_dst, draw_list->VtxBuffer.Data, draw_list->VtxBuffer.Size * sizeof(ImDrawVert));
282 memcpy(idx_dst, draw_list->IdxBuffer.Data, draw_list->IdxBuffer.Size * sizeof(ImDrawIdx));
283 vtx_dst += draw_list->VtxBuffer.Size;
284 idx_dst += draw_list->IdxBuffer.Size;
285 }
286
287 // During Unmap() we specify the written range (as per DX12 API, this is informational and for tooling only)
288 range.End = (SIZE_T)((intptr_t)vtx_dst - (intptr_t)vtx_resource);
289 IM_ASSERT(range.End == draw_data->TotalVtxCount * sizeof(ImDrawVert));
290 fr->VertexBuffer->Unmap(0, &range);
291 range.End = (SIZE_T)((intptr_t)idx_dst - (intptr_t)idx_resource);
292 IM_ASSERT(range.End == draw_data->TotalIdxCount * sizeof(ImDrawIdx));
293 fr->IndexBuffer->Unmap(0, &range);
294
295 // Setup desired DX state
296 ImGui_ImplDX12_SetupRenderState(draw_data, command_list, fr);
297
298 // Setup render state structure (for callbacks and custom texture bindings)
299 ImGuiPlatformIO& platform_io = ImGui::GetPlatformIO();
300 ImGui_ImplDX12_RenderState render_state;
301 render_state.Device = bd->pd3dDevice;
302 render_state.CommandList = command_list;
303 platform_io.Renderer_RenderState = &render_state;
304
305 // Render command lists
306 // (Because we merged all buffers into a single one, we maintain our own offset into them)
307 int global_vtx_offset = 0;
308 int global_idx_offset = 0;
309 ImVec2 clip_off = draw_data->DisplayPos;
310 ImVec2 clip_scale = draw_data->FramebufferScale;
311 for (const ImDrawList* draw_list : draw_data->CmdLists)
312 {
313 for (int cmd_i = 0; cmd_i < draw_list->CmdBuffer.Size; cmd_i++)
314 {
315 const ImDrawCmd* pcmd = &draw_list->CmdBuffer[cmd_i];
316 if (pcmd->UserCallback != nullptr)
317 {
318 // User callback, registered via ImDrawList::AddCallback()
319 // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
320 if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
321 ImGui_ImplDX12_SetupRenderState(draw_data, command_list, fr);
322 else
323 pcmd->UserCallback(draw_list, pcmd);
324 }
325 else
326 {
327 // Project scissor/clipping rectangles into framebuffer space
328 ImVec2 clip_min((pcmd->ClipRect.x - clip_off.x) * clip_scale.x, (pcmd->ClipRect.y - clip_off.y) * clip_scale.y);
329 ImVec2 clip_max((pcmd->ClipRect.z - clip_off.x) * clip_scale.x, (pcmd->ClipRect.w - clip_off.y) * clip_scale.y);
330 if (clip_max.x <= clip_min.x || clip_max.y <= clip_min.y)
331 continue;
332
333 // Apply scissor/clipping rectangle
334 const D3D12_RECT r = { (LONG)clip_min.x, (LONG)clip_min.y, (LONG)clip_max.x, (LONG)clip_max.y };
335 command_list->RSSetScissorRects(1, &r);
336
337 // Bind texture, Draw
338 D3D12_GPU_DESCRIPTOR_HANDLE texture_handle = {};
339 texture_handle.ptr = (UINT64)pcmd->GetTexID();
340 command_list->SetGraphicsRootDescriptorTable(1, texture_handle);
341 command_list->DrawIndexedInstanced(pcmd->ElemCount, 1, pcmd->IdxOffset + global_idx_offset, pcmd->VtxOffset + global_vtx_offset, 0);
342 }
343 }
344 global_idx_offset += draw_list->IdxBuffer.Size;
345 global_vtx_offset += draw_list->VtxBuffer.Size;
346 }
347 platform_io.Renderer_RenderState = nullptr;
348}

350static void ImGui_ImplDX12_DestroyTexture(ImTextureData* tex)
351{
352 if (ImGui_ImplDX12_Texture* backend_tex = (ImGui_ImplDX12_Texture*)tex->BackendUserData)
353 {
354 IM_ASSERT(backend_tex->hFontSrvGpuDescHandle.ptr == (UINT64)tex->TexID);

356 bd->InitInfo.SrvDescriptorFreeFn(&bd->InitInfo, backend_tex->hFontSrvCpuDescHandle, backend_tex->hFontSrvGpuDescHandle);
357 SafeRelease(backend_tex->pTextureResource);
358 backend_tex->hFontSrvCpuDescHandle.ptr = 0;
359 backend_tex->hFontSrvGpuDescHandle.ptr = 0;

361

363 tex->SetTexID(ImTextureID_Invalid);
364 tex->BackendUserData = nullptr;
365 }
366 tex->SetStatus(ImTextureStatus_Destroyed);
367}

369void ImGui_ImplDX12_UpdateTexture(ImTextureData* tex)
370{
371 ImGui_ImplDX12_Data* bd = ImGui_ImplDX12_GetBackendData();
372 bool need_barrier_before_copy = true; // Do we need a resource barrier before we copy new data in?
373
374 if (tex->Status == ImTextureStatus_WantCreate)
375 {
376 // Create and upload new texture to graphics system
377 //IMGUI_DEBUG_LOG("UpdateTexture #%03d: WantCreate %dx%d\n", tex->UniqueID, tex->Width, tex->Height);
378 IM_ASSERT(tex->TexID == ImTextureID_Invalid && tex->BackendUserData == nullptr);
379 IM_ASSERT(tex->Format == ImTextureFormat_RGBA32);
380 ImGui_ImplDX12_Texture* backend_tex = IM_NEW(ImGui_ImplDX12_Texture)();
381 bd->InitInfo.SrvDescriptorAllocFn(&bd->InitInfo, &backend_tex->hFontSrvCpuDescHandle, &backend_tex->hFontSrvGpuDescHandle); // Allocate a desctriptor handle
382
383 D3D12_HEAP_PROPERTIES props = {};
384 props.Type = D3D12_HEAP_TYPE_DEFAULT;
385 props.CPUPageProperty = D3D12_CPU_PAGE_PROPERTY_UNKNOWN;
386 props.MemoryPoolPreference = D3D12_MEMORY_POOL_UNKNOWN;
387
388 D3D12_RESOURCE_DESC desc;
389 ZeroMemory(&desc, sizeof(desc));
390 desc.Dimension = D3D12_RESOURCE_DIMENSION_TEXTURE2D;
391 desc.Alignment = 0;
392 desc.Width = tex->Width;
393 desc.Height = tex->Height;
394 desc.DepthOrArraySize = 1;
395 desc.MipLevels = 1;
396 desc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
397 desc.SampleDesc.Count = 1;
398 desc.SampleDesc.Quality = 0;
399 desc.Layout = D3D12_TEXTURE_LAYOUT_UNKNOWN;
400 desc.Flags = D3D12_RESOURCE_FLAG_NONE;
401
402 ID3D12Resource* pTexture = nullptr;
403 bd->pd3dDevice->CreateCommittedResource(&props, D3D12_HEAP_FLAG_NONE, &desc,
404 D3D12_RESOURCE_STATE_COPY_DEST, nullptr, IID_PPV_ARGS(&pTexture));
405
406 // Create SRV
407 D3D12_SHADER_RESOURCE_VIEW_DESC srvDesc;
408 ZeroMemory(&srvDesc, sizeof(srvDesc));
409 srvDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
410 srvDesc.ViewDimension = D3D12_SRV_DIMENSION_TEXTURE2D;
411 srvDesc.Texture2D.MipLevels = desc.MipLevels;
412 srvDesc.Texture2D.MostDetailedMip = 0;
413 srvDesc.Shader4ComponentMapping = D3D12_DEFAULT_SHADER_4_COMPONENT_MAPPING;
414 bd->pd3dDevice->CreateShaderResourceView(pTexture, &srvDesc, backend_tex->hFontSrvCpuDescHandle);
415 SafeRelease(backend_tex->pTextureResource);
416 backend_tex->pTextureResource = pTexture;
417
418 // Store identifiers
419 tex->SetTexID((ImTextureID)backend_tex->hFontSrvGpuDescHandle.ptr);
420 tex->BackendUserData = backend_tex;
421 need_barrier_before_copy = false; // Because this is a newly-created texture it will be in D3D12_RESOURCE_STATE_COMMON and thus we don't need a barrier
422 // We don't set tex->Status to ImTextureStatus_OK to let the code fallthrough below.
423 }
424
425 if (tex->Status == ImTextureStatus_WantCreate || tex->Status == ImTextureStatus_WantUpdates)
426 {
427 ImGui_ImplDX12_Texture* backend_tex = (ImGui_ImplDX12_Texture*)tex->BackendUserData;
428 IM_ASSERT(tex->Format == ImTextureFormat_RGBA32);
429
430 // We could use the smaller rect on _WantCreate but using the full rect allows us to clear the texture.
431 // FIXME-OPT: Uploading single box even when using ImTextureStatus_WantUpdates. Could use tex->Updates[]
432 // - Copy all blocks contiguously in upload buffer.
433 // - Barrier before copy, submit all CopyTextureRegion(), barrier after copy.
434 const int upload_x = (tex->Status == ImTextureStatus_WantCreate) ? 0 : tex->UpdateRect.x;
435 const int upload_y = (tex->Status == ImTextureStatus_WantCreate) ? 0 : tex->UpdateRect.y;
436 const int upload_w = (tex->Status == ImTextureStatus_WantCreate) ? tex->Width : tex->UpdateRect.w;
437 const int upload_h = (tex->Status == ImTextureStatus_WantCreate) ? tex->Height : tex->UpdateRect.h;
438
439 // Update full texture or selected blocks. We only ever write to textures regions which have never been used before!
440 // This backend choose to use tex->UpdateRect but you can use tex->Updates[] to upload individual regions.
441 UINT upload_pitch_src = upload_w * tex->BytesPerPixel;
442 UINT upload_pitch_dst = (upload_pitch_src + D3D12_TEXTURE_DATA_PITCH_ALIGNMENT - 1u) & ~(D3D12_TEXTURE_DATA_PITCH_ALIGNMENT - 1u);
443 UINT upload_size = upload_pitch_dst * upload_h;
444
445 if (bd->pTexUploadBuffer == nullptr || upload_size > bd->pTexUploadBufferSize)
446 {
447 if (bd->pTexUploadBufferMapped)
448 {
449 D3D12_RANGE range = { 0, bd->pTexUploadBufferSize };
450 bd->pTexUploadBuffer->Unmap(0, &range);
451 bd->pTexUploadBufferMapped = nullptr;
452 }
453 SafeRelease(bd->pTexUploadBuffer);
454
455 D3D12_RESOURCE_DESC desc;
456 ZeroMemory(&desc, sizeof(desc));
457 desc.Dimension = D3D12_RESOURCE_DIMENSION_BUFFER;
458 desc.Alignment = 0;
459 desc.Width = upload_size;
460 desc.Height = 1;
461 desc.DepthOrArraySize = 1;
462 desc.MipLevels = 1;
463 desc.Format = DXGI_FORMAT_UNKNOWN;
464 desc.SampleDesc.Count = 1;
465 desc.SampleDesc.Quality = 0;
466 desc.Layout = D3D12_TEXTURE_LAYOUT_ROW_MAJOR;
467 desc.Flags = D3D12_RESOURCE_FLAG_NONE;
468
469 D3D12_HEAP_PROPERTIES props;
470 memset(&props, 0, sizeof(D3D12_HEAP_PROPERTIES));
471 props.Type = D3D12_HEAP_TYPE_UPLOAD;
472 props.CPUPageProperty = D3D12_CPU_PAGE_PROPERTY_UNKNOWN;
473 props.MemoryPoolPreference = D3D12_MEMORY_POOL_UNKNOWN;
474
475 HRESULT hr = bd->pd3dDevice->CreateCommittedResource(&props, D3D12_HEAP_FLAG_NONE, &desc,
476 D3D12_RESOURCE_STATE_GENERIC_READ, nullptr, IID_PPV_ARGS(&bd->pTexUploadBuffer));
477 IM_ASSERT(SUCCEEDED(hr));
478
479 D3D12_RANGE range = {0, upload_size};
480 hr = bd->pTexUploadBuffer->Map(0, &range, &bd->pTexUploadBufferMapped);
481 IM_ASSERT(SUCCEEDED(hr));
482 bd->pTexUploadBufferSize = upload_size;
483 }
484
485 bd->pTexCmdAllocator->Reset();
486 bd->pTexCmdList->Reset(bd->pTexCmdAllocator, nullptr);
487 ID3D12GraphicsCommandList* cmdList = bd->pTexCmdList;
488
489 // Copy to upload buffer
490 for (int y = 0; y < upload_h; y++)
491 memcpy((void*)((uintptr_t)bd->pTexUploadBufferMapped + y * upload_pitch_dst), tex->GetPixelsAt(upload_x, upload_y + y), upload_pitch_src);
492
493 if (need_barrier_before_copy)
494 {
495 D3D12_RESOURCE_BARRIER barrier = {};
496 barrier.Type = D3D12_RESOURCE_BARRIER_TYPE_TRANSITION;
497 barrier.Flags = D3D12_RESOURCE_BARRIER_FLAG_NONE;
498 barrier.Transition.pResource = backend_tex->pTextureResource;
499 barrier.Transition.Subresource = D3D12_RESOURCE_BARRIER_ALL_SUBRESOURCES;
500 barrier.Transition.StateBefore = D3D12_RESOURCE_STATE_PIXEL_SHADER_RESOURCE;
501 barrier.Transition.StateAfter = D3D12_RESOURCE_STATE_COPY_DEST;
502 cmdList->ResourceBarrier(1, &barrier);
503 }
504
505 D3D12_TEXTURE_COPY_LOCATION srcLocation = {};
506 D3D12_TEXTURE_COPY_LOCATION dstLocation = {};
507 {
508 srcLocation.pResource = bd->pTexUploadBuffer;
509 srcLocation.Type = D3D12_TEXTURE_COPY_TYPE_PLACED_FOOTPRINT;
510 srcLocation.PlacedFootprint.Footprint.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
511 srcLocation.PlacedFootprint.Footprint.Width = upload_w;
512 srcLocation.PlacedFootprint.Footprint.Height = upload_h;
513 srcLocation.PlacedFootprint.Footprint.Depth = 1;
514 srcLocation.PlacedFootprint.Footprint.RowPitch = upload_pitch_dst;
515 dstLocation.pResource = backend_tex->pTextureResource;
516 dstLocation.Type = D3D12_TEXTURE_COPY_TYPE_SUBRESOURCE_INDEX;
517 dstLocation.SubresourceIndex = 0;
518 }
519 cmdList->CopyTextureRegion(&dstLocation, upload_x, upload_y, 0, &srcLocation, nullptr);
520
521 {
522 D3D12_RESOURCE_BARRIER barrier = {};
523 barrier.Type = D3D12_RESOURCE_BARRIER_TYPE_TRANSITION;
524 barrier.Flags = D3D12_RESOURCE_BARRIER_FLAG_NONE;
525 barrier.Transition.pResource = backend_tex->pTextureResource;
526 barrier.Transition.Subresource = D3D12_RESOURCE_BARRIER_ALL_SUBRESOURCES;
527 barrier.Transition.StateBefore = D3D12_RESOURCE_STATE_COPY_DEST;
528 barrier.Transition.StateAfter = D3D12_RESOURCE_STATE_PIXEL_SHADER_RESOURCE;
529 cmdList->ResourceBarrier(1, &barrier);
530 }
531
532 HRESULT hr = cmdList->Close();
533 IM_ASSERT(SUCCEEDED(hr));
534 ID3D12CommandQueue* cmdQueue = bd->pCommandQueue;
535 cmdQueue->ExecuteCommandLists(1, (ID3D12CommandList* const*)&cmdList);
536 hr = cmdQueue->Signal(bd->Fence, ++bd->FenceLastSignaledValue);
537 IM_ASSERT(SUCCEEDED(hr));
538
539 // FIXME-OPT: Suboptimal?
540 // - To remove this may need to create NumFramesInFlight x ImGui_ImplDX12_FrameContext in backend data (mimick docking version)
541 // - Store per-frame in flight: upload buffer?
542 // - Where do cmdList and cmdAlloc fit?
543 bd->Fence->SetEventOnCompletion(bd->FenceLastSignaledValue, bd->FenceEvent);
544 ::WaitForSingleObject(bd->FenceEvent, INFINITE);
545
546 tex->SetStatus(ImTextureStatus_OK);
547 }
548
549 if (tex->Status == ImTextureStatus_WantDestroy && tex->UnusedFrames >= (int)bd->numFramesInFlight)
550 ImGui_ImplDX12_DestroyTexture(tex);
551}
