203void ImGui_ImplGLUT_NewFrame()
204{
205 // Setup time step
206 ImGuiIO& io = ImGui::GetIO();
207 int current_time = glutGet(GLUT_ELAPSED_TIME);
208 int delta_time_ms = (current_time - g_Time);
209 if (delta_time_ms <= 0)
210 delta_time_ms = 1;
211 io.DeltaTime = delta_time_ms / 1000.0f;
212 g_Time = current_time;
213}

216{
217 ImGuiIO& io = ImGui::GetIO();
218 int glut_key_mods = glutGetModifiers();

231void ImGui_ImplGLUT_KeyboardFunc(unsigned char c, int x, int y)
232{
233 // Send character to imgui
234 //printf("char_down_func %d '%c'\n", c, c);

272void ImGui_ImplGLUT_MouseFunc(int glut_button, int state, int x, int y)
273{
274 ImGuiIO& io = ImGui::GetIO();
275 io.AddMousePosEvent((float)x, (float)y);
276 int button = -1;
277 if (glut_button == GLUT_LEFT_BUTTON) button = 0;
278 if (glut_button == GLUT_RIGHT_BUTTON) button = 1;
279 if (glut_button == GLUT_MIDDLE_BUTTON) button = 2;
280 if (button != -1 && (state == GLUT_DOWN || state == GLUT_UP))
281 io.AddMouseButtonEvent(button, state == GLUT_DOWN);
282}
