1// dear imgui: Renderer Backend for OpenGL2 (legacy OpenGL, fixed pipeline)
2// This needs to be used along with a Platform Backend (e.g. GLFW, SDL, Win32, custom..)
3
4// Implemented features:
5// [X] Renderer: User texture binding. Use 'GLuint' OpenGL texture as texture identifier. Read the FAQ about ImTextureID/ImTextureRef!
6// [X] Renderer: Texture updates support for dynamic font atlas (ImGuiBackendFlags_RendererHasTextures).
7// Missing features or Issues:
8// [ ] Renderer: Large meshes support (64k+ vertices) even with 16-bit indices (ImGuiBackendFlags_RendererHasVtxOffset).
9
10// You can use unmodified imgui_impl_* files in your project. See examples/ folder for examples of using this.
11// Prefer including the entire imgui/ repository into your project (either as a copy or as a submodule), and only build the backends you need.
12// Learn about Dear ImGui:
13// - FAQ https://dearimgui.com/faq
14// - Getting Started https://dearimgui.com/getting-started
15// - Documentation https://dearimgui.com/docs (same as your local docs/ folder).
16// - Introduction, links and more at the top of imgui.cpp
17
18// **DO NOT USE THIS CODE IF YOUR CODE/ENGINE IS USING MODERN OPENGL (SHADERS, VBO, VAO, etc.)**
19// **Prefer using the code in imgui_impl_opengl3.cpp**
20// This code is mostly provided as a reference to learn how ImGui integration works, because it is shorter to read.
21// If your code is using GL3+ context or any semi modern OpenGL calls, using this is likely to make everything more
22// complicated, will require your code to reset every single OpenGL attributes to their initial state, and might
23// confuse your GPU driver.
24// The GL2 code is unable to reset attributes or even call e.g. "glUseProgram(0)" because they don't exist in that API.

137static void ImGui_ImplOpenGL2_SetupRenderState(ImDrawData* draw_data, int fb_width, int fb_height)
138{
139 // Setup render state: alpha-blending enabled, no face culling, no depth testing, scissor enabled, vertex/texcoord/color pointers, polygon fill.
140 glEnable(GL_BLEND);
141 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
142 //glBlendFuncSeparate(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA, GL_ONE, GL_ONE_MINUS_SRC_ALPHA); // In order to composite our output buffer we need to preserve alpha
143 glDisable(GL_CULL_FACE);
144 glDisable(GL_DEPTH_TEST);
145 glDisable(GL_STENCIL_TEST);
146 glDisable(GL_LIGHTING);
147 glDisable(GL_COLOR_MATERIAL);
148 glEnable(GL_SCISSOR_TEST);
149 glEnableClientState(GL_VERTEX_ARRAY);
150 glEnableClientState(GL_TEXTURE_COORD_ARRAY);
151 glEnableClientState(GL_COLOR_ARRAY);
152 glDisableClientState(GL_NORMAL_ARRAY);
153 glEnable(GL_TEXTURE_2D);
154 glPolygonMode(GL_FRONT_AND_BACK, GL_FILL);
155 glShadeModel(GL_SMOOTH);
156 glTexEnvi(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, GL_MODULATE);
157
158 // If you are using this code with non-legacy OpenGL header/contexts (which you should not, prefer using imgui_impl_opengl3.cpp!!),
159 // you may need to backup/reset/restore other state, e.g. for current shader using the commented lines below.
160 // (DO NOT MODIFY THIS FILE! Add the code in your calling function)
161 // GLint last_program;
162 // glGetIntegerv(GL_CURRENT_PROGRAM, &last_program);
163 // glUseProgram(0);
164 // ImGui_ImplOpenGL2_RenderDrawData(...);
165 // glUseProgram(last_program)
166 // There are potentially many more states you could need to clear/setup that we can't access from default headers.
167 // e.g. glBindBuffer(GL_ARRAY_BUFFER, 0), glDisable(GL_TEXTURE_CUBE_MAP).
168
169 // Setup viewport, orthographic projection matrix
170 // Our visible imgui space lies from draw_data->DisplayPos (top left) to draw_data->DisplayPos+data_data->DisplaySize (bottom right). DisplayPos is (0,0) for single viewport apps.
171 GL_CALL(glViewport(0, 0, (GLsizei)fb_width, (GLsizei)fb_height));
172 glMatrixMode(GL_PROJECTION);
173 glPushMatrix();
174 glLoadIdentity();
175 glOrtho(draw_data->DisplayPos.x, draw_data->DisplayPos.x + draw_data->DisplaySize.x, draw_data->DisplayPos.y + draw_data->DisplaySize.y, draw_data->DisplayPos.y, -1.0f, +1.0f);
176 glMatrixMode(GL_MODELVIEW);
177 glPushMatrix();
178 glLoadIdentity();
179}

181// OpenGL2 Render function.
182// Note that this implementation is little overcomplicated because we are saving/setting up/restoring every OpenGL state explicitly.
183// This is in order to be able to run within an OpenGL engine that doesn't do so.
184void ImGui_ImplOpenGL2_RenderDrawData(ImDrawData* draw_data)
185{
186 // Avoid rendering when minimized, scale coordinates for retina displays (screen coordinates != framebuffer coordinates)
187 int fb_width = (int)(draw_data->DisplaySize.x * draw_data->FramebufferScale.x);
188 int fb_height = (int)(draw_data->DisplaySize.y * draw_data->FramebufferScale.y);
189 if (fb_width == 0 || fb_height == 0)
190 return;
191
192 // Catch up with texture updates. Most of the times, the list will have 1 element with an OK status, aka nothing to do.
193 // (This almost always points to ImGui::GetPlatformIO().Textures[] but is part of ImDrawData to allow overriding or disabling texture updates).
194 if (draw_data->Textures != nullptr)
195 for (ImTextureData* tex : *draw_data->Textures)
196 if (tex->Status != ImTextureStatus_OK)
197 ImGui_ImplOpenGL2_UpdateTexture(tex);
198
199 // Backup GL state
200 GLint last_texture; glGetIntegerv(GL_TEXTURE_BINDING_2D, &last_texture);
201 GLint last_polygon_mode[2]; glGetIntegerv(GL_POLYGON_MODE, last_polygon_mode);
202 GLint last_viewport[4]; glGetIntegerv(GL_VIEWPORT, last_viewport);
203 GLint last_scissor_box[4]; glGetIntegerv(GL_SCISSOR_BOX, last_scissor_box);
204 GLint last_shade_model; glGetIntegerv(GL_SHADE_MODEL, &last_shade_model);
205 GLint last_tex_env_mode; glGetTexEnviv(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, &last_tex_env_mode);
206 glPushAttrib(GL_ENABLE_BIT | GL_COLOR_BUFFER_BIT | GL_TRANSFORM_BIT);
207
208 // Setup desired GL state
209 ImGui_ImplOpenGL2_SetupRenderState(draw_data, fb_width, fb_height);
210
211 // Will project scissor/clipping rectangles into framebuffer space
212 ImVec2 clip_off = draw_data->DisplayPos; // (0,0) unless using multi-viewports
213 ImVec2 clip_scale = draw_data->FramebufferScale; // (1,1) unless using retina display which are often (2,2)
214
215 // Render command lists
216 for (const ImDrawList* draw_list : draw_data->CmdLists)
217 {
218 const ImDrawVert* vtx_buffer = draw_list->VtxBuffer.Data;
219 const ImDrawIdx* idx_buffer = draw_list->IdxBuffer.Data;
220 glVertexPointer(2, GL_FLOAT, sizeof(ImDrawVert), (const GLvoid*)((const char*)vtx_buffer + offsetof(ImDrawVert, pos)));
221 glTexCoordPointer(2, GL_FLOAT, sizeof(ImDrawVert), (const GLvoid*)((const char*)vtx_buffer + offsetof(ImDrawVert, uv)));
222 glColorPointer(4, GL_UNSIGNED_BYTE, sizeof(ImDrawVert), (const GLvoid*)((const char*)vtx_buffer + offsetof(ImDrawVert, col)));
223
224 for (int cmd_i = 0; cmd_i < draw_list->CmdBuffer.Size; cmd_i++)
225 {
226 const ImDrawCmd* pcmd = &draw_list->CmdBuffer[cmd_i];
227 if (pcmd->UserCallback)
228 {
229 // User callback, registered via ImDrawList::AddCallback()
230 // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
231 if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
232 ImGui_ImplOpenGL2_SetupRenderState(draw_data, fb_width, fb_height);
233 else
234 pcmd->UserCallback(draw_list, pcmd);
235 }
236 else
237 {
238 // Project scissor/clipping rectangles into framebuffer space
239 ImVec2 clip_min((pcmd->ClipRect.x - clip_off.x) * clip_scale.x, (pcmd->ClipRect.y - clip_off.y) * clip_scale.y);
240 ImVec2 clip_max((pcmd->ClipRect.z - clip_off.x) * clip_scale.x, (pcmd->ClipRect.w - clip_off.y) * clip_scale.y);
241 if (clip_max.x <= clip_min.x || clip_max.y <= clip_min.y)
242 continue;
243
244 // Apply scissor/clipping rectangle (Y is inverted in OpenGL)
245 glScissor((int)clip_min.x, (int)((float)fb_height - clip_max.y), (int)(clip_max.x - clip_min.x), (int)(clip_max.y - clip_min.y));
246
247 // Bind texture, Draw
248 glBindTexture(GL_TEXTURE_2D, (GLuint)(intptr_t)pcmd->GetTexID());
249 glDrawElements(GL_TRIANGLES, (GLsizei)pcmd->ElemCount, sizeof(ImDrawIdx) == 2 ? GL_UNSIGNED_SHORT : GL_UNSIGNED_INT, idx_buffer + pcmd->IdxOffset);
250 }
251 }
252 }
253
254 // Restore modified GL state
255 glDisableClientState(GL_COLOR_ARRAY);
256 glDisableClientState(GL_TEXTURE_COORD_ARRAY);
257 glDisableClientState(GL_VERTEX_ARRAY);
258 glBindTexture(GL_TEXTURE_2D, (GLuint)last_texture);
259 glMatrixMode(GL_MODELVIEW);
260 glPopMatrix();
261 glMatrixMode(GL_PROJECTION);
262 glPopMatrix();
263 glPopAttrib();
264 glPolygonMode(GL_FRONT, (GLenum)last_polygon_mode[0]); glPolygonMode(GL_BACK, (GLenum)last_polygon_mode[1]);
265 glViewport(last_viewport[0], last_viewport[1], (GLsizei)last_viewport[2], (GLsizei)last_viewport[3]);
266 glScissor(last_scissor_box[0], last_scissor_box[1], (GLsizei)last_scissor_box[2], (GLsizei)last_scissor_box[3]);
267 glShadeModel(last_shade_model);
268 glTexEnvi(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, last_tex_env_mode);
269}

271void ImGui_ImplOpenGL2_UpdateTexture(ImTextureData* tex)
272{
273 if (tex->Status == ImTextureStatus_WantCreate)
274 {
275 // Create and upload new texture to graphics system
276 //IMGUI_DEBUG_LOG("UpdateTexture #%03d: WantCreate %dx%d\n", tex->UniqueID, tex->Width, tex->Height);
277 IM_ASSERT(tex->TexID == 0 && tex->BackendUserData == nullptr);
278 IM_ASSERT(tex->Format == ImTextureFormat_RGBA32);
279 const void* pixels = tex->GetPixels();
280 GLuint gl_texture_id = 0;
281
282 // Upload texture to graphics system
283 // (Bilinear sampling is required by default. Set 'io.Fonts->Flags |= ImFontAtlasFlags_NoBakedLines' or 'style.AntiAliasedLinesUseTex = false' to allow point/nearest sampling)
284 GLint last_texture;
285 GL_CALL(glGetIntegerv(GL_TEXTURE_BINDING_2D, &last_texture));
286 GL_CALL(glGenTextures(1, &gl_texture_id));
287 GL_CALL(glBindTexture(GL_TEXTURE_2D, gl_texture_id));
288 GL_CALL(glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR));
289 GL_CALL(glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR));
290 GL_CALL(glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP));
291 GL_CALL(glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP));
292 GL_CALL(glPixelStorei(GL_UNPACK_ROW_LENGTH, 0));
293 GL_CALL(glPixelStorei(GL_UNPACK_ALIGNMENT, 1));
294 GL_CALL(glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, tex->Width, tex->Height, 0, GL_RGBA, GL_UNSIGNED_BYTE, pixels));
295
296 // Store identifiers
297 tex->SetTexID((ImTextureID)(intptr_t)gl_texture_id);
298 tex->SetStatus(ImTextureStatus_OK);
299
300 // Restore state
301 GL_CALL(glBindTexture(GL_TEXTURE_2D, last_texture));
302 }
303 else if (tex->Status == ImTextureStatus_WantUpdates)
304 {
305 // Update selected blocks. We only ever write to textures regions which have never been used before!
306 // This backend choose to use tex->Updates[] but you can use tex->UpdateRect to upload a single region.
307 GLint last_texture;
308 GL_CALL(glGetIntegerv(GL_TEXTURE_BINDING_2D, &last_texture));
309
310 GLuint gl_tex_id = (GLuint)(intptr_t)tex->TexID;
311 GL_CALL(glBindTexture(GL_TEXTURE_2D, gl_tex_id));
312 GL_CALL(glPixelStorei(GL_UNPACK_ROW_LENGTH, tex->Width));
313 GL_CALL(glPixelStorei(GL_UNPACK_ALIGNMENT, 1));
314 for (ImTextureRect& r : tex->Updates)
315 GL_CALL(glTexSubImage2D(GL_TEXTURE_2D, 0, r.x, r.y, r.w, r.h, GL_RGBA, GL_UNSIGNED_BYTE, tex->GetPixelsAt(r.x, r.y)));
316 GL_CALL(glPixelStorei(GL_UNPACK_ROW_LENGTH, 0));
317 GL_CALL(glBindTexture(GL_TEXTURE_2D, last_texture)); // Restore state
318 tex->SetStatus(ImTextureStatus_OK);
319 }
320 else if (tex->Status == ImTextureStatus_WantDestroy)
321 {
322 GLuint gl_tex_id = (GLuint)(intptr_t)tex->TexID;
323 glDeleteTextures(1, &gl_tex_id);
324
325 // Clear identifiers and mark as destroyed (in order to allow e.g. calling InvalidateDeviceObjects while running)
326 tex->SetTexID(ImTextureID_Invalid);
327 tex->SetStatus(ImTextureStatus_Destroyed);
328 }
329}
