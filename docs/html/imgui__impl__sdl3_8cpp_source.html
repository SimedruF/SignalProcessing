177// We discard viewport passed via ImGuiPlatformImeData and always call SDL_StartTextInput() on SDL_GetKeyboardFocus().
178static void ImGui_ImplSDL3_UpdateIme()
179{
180 ImGui_ImplSDL3_Data* bd = ImGui_ImplSDL3_GetBackendData();
181 ImGuiPlatformImeData* data = &bd->ImeData;
182 SDL_Window* window = SDL_GetKeyboardFocus();
183
184 // Stop previous input
185 if ((!(data->WantVisible || data->WantTextInput) || bd->ImeWindow != window) && bd->ImeWindow != nullptr)
186 {
187 SDL_StopTextInput(bd->ImeWindow);
188 bd->ImeWindow = nullptr;
189 }
190 if ((!bd->ImeDirty && bd->ImeWindow == window) || (window == NULL))
191 return;
192
193 // Start/update current input
194 bd->ImeDirty = false;
195 if (data->WantVisible)
196 {
197 SDL_Rect r;
198 r.x = (int)data->InputPos.x;
199 r.y = (int)data->InputPos.y;
200 r.w = 1;
201 r.h = (int)data->InputLineHeight;
202 SDL_SetTextInputArea(window, &r, 0);
203 bd->ImeWindow = window;
204 }
205 if (!SDL_TextInputActive(window) && (data->WantVisible || data->WantTextInput))
206 SDL_StartTextInput(window);
207}

371// You can read the io.WantCaptureMouse, io.WantCaptureKeyboard flags to tell if dear imgui wants to use your inputs.
372// - When io.WantCaptureMouse is true, do not dispatch mouse input data to your main application, or clear/overwrite your copy of the mouse data.
373// - When io.WantCaptureKeyboard is true, do not dispatch keyboard input data to your main application, or clear/overwrite your copy of the keyboard data.
374// Generally you may always pass all inputs to dear imgui, and hide them from your application based on those two flags.
375// If you have multiple SDL events and some of them are not meant to be used by dear imgui, you may need to filter events based on their windowID field.
376bool ImGui_ImplSDL3_ProcessEvent(const SDL_Event* event)
377{
378 ImGui_ImplSDL3_Data* bd = ImGui_ImplSDL3_GetBackendData();
379 IM_ASSERT(bd != nullptr && "Context or backend not initialized! Did you call ImGui_ImplSDL3_Init()?");
380 ImGuiIO& io = ImGui::GetIO();
381
382 switch (event->type)
383 {
384 case SDL_EVENT_MOUSE_MOTION:
385 {
386 if (ImGui_ImplSDL3_GetViewportForWindowID(event->motion.windowID) == nullptr)
387 return false;
388 ImVec2 mouse_pos((float)event->motion.x, (float)event->motion.y);
389 io.AddMouseSourceEvent(event->motion.which == SDL_TOUCH_MOUSEID ? ImGuiMouseSource_TouchScreen : ImGuiMouseSource_Mouse);
390 io.AddMousePosEvent(mouse_pos.x, mouse_pos.y);
391 return true;
392 }
393 case SDL_EVENT_MOUSE_WHEEL:
394 {
395 if (ImGui_ImplSDL3_GetViewportForWindowID(event->wheel.windowID) == nullptr)
396 return false;
397 //IMGUI_DEBUG_LOG("wheel %.2f %.2f, precise %.2f %.2f\n", (float)event->wheel.x, (float)event->wheel.y, event->wheel.preciseX, event->wheel.preciseY);
398 float wheel_x = -event->wheel.x;
399 float wheel_y = event->wheel.y;
400 io.AddMouseSourceEvent(event->wheel.which == SDL_TOUCH_MOUSEID ? ImGuiMouseSource_TouchScreen : ImGuiMouseSource_Mouse);
401 io.AddMouseWheelEvent(wheel_x, wheel_y);
402 return true;
403 }
404 case SDL_EVENT_MOUSE_BUTTON_DOWN:
405 case SDL_EVENT_MOUSE_BUTTON_UP:
406 {
407 if (ImGui_ImplSDL3_GetViewportForWindowID(event->button.windowID) == nullptr)
408 return false;
409 int mouse_button = -1;
410 if (event->button.button == SDL_BUTTON_LEFT) { mouse_button = 0; }
411 if (event->button.button == SDL_BUTTON_RIGHT) { mouse_button = 1; }
412 if (event->button.button == SDL_BUTTON_MIDDLE) { mouse_button = 2; }
413 if (event->button.button == SDL_BUTTON_X1) { mouse_button = 3; }
414 if (event->button.button == SDL_BUTTON_X2) { mouse_button = 4; }
415 if (mouse_button == -1)
416 break;
417 io.AddMouseSourceEvent(event->button.which == SDL_TOUCH_MOUSEID ? ImGuiMouseSource_TouchScreen : ImGuiMouseSource_Mouse);
418 io.AddMouseButtonEvent(mouse_button, (event->type == SDL_EVENT_MOUSE_BUTTON_DOWN));
419 bd->MouseButtonsDown = (event->type == SDL_EVENT_MOUSE_BUTTON_DOWN) ? (bd->MouseButtonsDown | (1 << mouse_button)) : (bd->MouseButtonsDown & ~(1 << mouse_button));
420 return true;
421 }
422 case SDL_EVENT_TEXT_INPUT:
423 {
424 if (ImGui_ImplSDL3_GetViewportForWindowID(event->text.windowID) == nullptr)
425 return false;
426 io.AddInputCharactersUTF8(event->text.text);
427 return true;
428 }
429 case SDL_EVENT_KEY_DOWN:
430 case SDL_EVENT_KEY_UP:
431 {
432 if (ImGui_ImplSDL3_GetViewportForWindowID(event->key.windowID) == nullptr)
433 return false;
434 ImGui_ImplSDL3_UpdateKeyModifiers((SDL_Keymod)event->key.mod);
435 //IMGUI_DEBUG_LOG("SDL_EVENT_KEY_%s : key=%d ('%s'), scancode=%d ('%s'), mod=%X\n",
436 // (event->type == SDL_EVENT_KEY_DOWN) ? "DOWN" : "UP ", event->key.key, SDL_GetKeyName(event->key.key), event->key.scancode, SDL_GetScancodeName(event->key.scancode), event->key.mod);
437 ImGuiKey key = ImGui_ImplSDL3_KeyEventToImGuiKey(event->key.key, event->key.scancode);
438 io.AddKeyEvent(key, (event->type == SDL_EVENT_KEY_DOWN));
439 io.SetKeyEventNativeData(key, (int)event->key.key, (int)event->key.scancode, (int)event->key.scancode); // To support legacy indexing (<1.87 user code). Legacy backend uses SDLK_*** as indices to IsKeyXXX() functions.
440 return true;
441 }
442 case SDL_EVENT_WINDOW_MOUSE_ENTER:
443 {
444 if (ImGui_ImplSDL3_GetViewportForWindowID(event->window.windowID) == nullptr)
445 return false;
446 bd->MouseWindowID = event->window.windowID;
447 bd->MousePendingLeaveFrame = 0;
448 return true;
449 }
450 // - In some cases, when detaching a window from main viewport SDL may send SDL_WINDOWEVENT_ENTER one frame too late,
451 // causing SDL_WINDOWEVENT_LEAVE on previous frame to interrupt drag operation by clear mouse position. This is why
452 // we delay process the SDL_WINDOWEVENT_LEAVE events by one frame. See issue #5012 for details.
453 // FIXME: Unconfirmed whether this is still needed with SDL3.
454 case SDL_EVENT_WINDOW_MOUSE_LEAVE:
455 {
456 if (ImGui_ImplSDL3_GetViewportForWindowID(event->window.windowID) == nullptr)
457 return false;
458 bd->MousePendingLeaveFrame = ImGui::GetFrameCount() + 1;
459 return true;
460 }
461 case SDL_EVENT_WINDOW_FOCUS_GAINED:
462 case SDL_EVENT_WINDOW_FOCUS_LOST:
463 {
464 if (ImGui_ImplSDL3_GetViewportForWindowID(event->window.windowID) == nullptr)
465 return false;
466 io.AddFocusEvent(event->type == SDL_EVENT_WINDOW_FOCUS_GAINED);
467 return true;
468 }
469 case SDL_EVENT_GAMEPAD_ADDED:
470 case SDL_EVENT_GAMEPAD_REMOVED:
471 {
472 bd->WantUpdateGamepadsList = true;
473 return true;
474 }
475 default:
476 break;
477 }
478 return false;
479}

632static void ImGui_ImplSDL3_UpdateMouseData()
633{
634 ImGui_ImplSDL3_Data* bd = ImGui_ImplSDL3_GetBackendData();
635 ImGuiIO& io = ImGui::GetIO();
636
637 // We forward mouse input when hovered or captured (via SDL_EVENT_MOUSE_MOTION) or when focused (below)
638#if SDL_HAS_CAPTURE_AND_GLOBAL_MOUSE
639 // - SDL_CaptureMouse() let the OS know e.g. that our drags can extend outside of parent boundaries (we want updated position) and shouldn't trigger other operations outside.
640 // - Debuggers under Linux tends to leave captured mouse on break, which may be very inconvenient, so to mitigate the issue we wait until mouse has moved to begin capture.
641 if (bd->MouseCanUseCapture)
642 {
643 bool want_capture = false;
644 for (int button_n = 0; button_n < ImGuiMouseButton_COUNT && !want_capture; button_n++)
645 if (ImGui::IsMouseDragging(button_n, 1.0f))
646 want_capture = true;
647 SDL_CaptureMouse(want_capture);
648 }
649
650 SDL_Window* focused_window = SDL_GetKeyboardFocus();
651 const bool is_app_focused = (bd->Window == focused_window);
652#else
653 SDL_Window* focused_window = bd->Window;
654 const bool is_app_focused = (SDL_GetWindowFlags(bd->Window) & SDL_WINDOW_INPUT_FOCUS) != 0; // SDL 2.0.3 and non-windowed systems: single-viewport only
655#endif
656 if (is_app_focused)
657 {
658 // (Optional) Set OS mouse position from Dear ImGui if requested (rarely used, only when io.ConfigNavMoveSetMousePos is enabled by user)
659 if (io.WantSetMousePos)
660 SDL_WarpMouseInWindow(bd->Window, io.MousePos.x, io.MousePos.y);
661
662 // (Optional) Fallback to provide unclamped mouse position when focused but not hovered (SDL_EVENT_MOUSE_MOTION already provides this when hovered or captured)
663 // Note that SDL_GetGlobalMouseState() is in theory slow on X11, but this only runs on rather specific cases. If a problem we may provide a way to opt-out this feature.
664 SDL_Window* hovered_window = SDL_GetMouseFocus();
665 const bool is_relative_mouse_mode = SDL_GetWindowRelativeMouseMode(bd->Window);
666 if (hovered_window == NULL && bd->MouseCanUseGlobalState && bd->MouseButtonsDown == 0 && !is_relative_mouse_mode)
667 {
668 // Single-viewport mode: mouse position in client window coordinates (io.MousePos is (0,0) when the mouse is on the upper-left corner of the app window)
669 float mouse_x, mouse_y;
670 int window_x, window_y;
671 SDL_GetGlobalMouseState(&mouse_x, &mouse_y);
672 SDL_GetWindowPosition(focused_window, &window_x, &window_y);
673 mouse_x -= window_x;
674 mouse_y -= window_y;
675 io.AddMousePosEvent(mouse_x, mouse_y);
676 }
677 }

832void ImGui_ImplSDL3_NewFrame()
833{
834 ImGui_ImplSDL3_Data* bd = ImGui_ImplSDL3_GetBackendData();
835 IM_ASSERT(bd != nullptr && "Context or backend not initialized! Did you call ImGui_ImplSDL3_Init()?");
836 ImGuiIO& io = ImGui::GetIO();
837
838 // Setup main viewport size (every frame to accommodate for window resizing)
839 ImGui_ImplSDL3_GetWindowSizeAndFramebufferScale(bd->Window, &io.DisplaySize, &io.DisplayFramebufferScale);
840
841 // Setup time step (we could also use SDL_GetTicksNS() available since SDL3)
842 // (Accept SDL_GetPerformanceCounter() not returning a monotonically increasing value. Happens in VMs and Emscripten, see #6189, #6114, #3644)
843 static Uint64 frequency = SDL_GetPerformanceFrequency();
844 Uint64 current_time = SDL_GetPerformanceCounter();
845 if (current_time <= bd->Time)
846 current_time = bd->Time + 1;
847 io.DeltaTime = bd->Time > 0 ? (float)((double)(current_time - bd->Time) / frequency) : (float)(1.0f / 60.0f);
848 bd->Time = current_time;
849
850 if (bd->MousePendingLeaveFrame && bd->MousePendingLeaveFrame >= ImGui::GetFrameCount() && bd->MouseButtonsDown == 0)
851 {
852 bd->MouseWindowID = 0;
853 bd->MousePendingLeaveFrame = 0;
854 io.AddMousePosEvent(-FLT_MAX, -FLT_MAX);
855 }
856
857 ImGui_ImplSDL3_UpdateMouseData();
858 ImGui_ImplSDL3_UpdateMouseCursor();
859 ImGui_ImplSDL3_UpdateIme();
860
861 // Update game controllers (if enabled and available)
862 ImGui_ImplSDL3_UpdateGamepads();
863}
