1// dear imgui: Renderer Backend for SDL_Renderer for SDL2
2// (Requires: SDL 2.0.17+)
3
4// Note that SDL_Renderer is an _optional_ component of SDL2, which IMHO is now largely obsolete.
5// For a multi-platform app consider using other technologies:
6// - SDL3+SDL_GPU: SDL_GPU is SDL3 new graphics abstraction API. You will need to update to SDL3.
7// - SDL2+DirectX, SDL2+OpenGL, SDL2+Vulkan: combine SDL with dedicated renderers.
8// If your application wants to render any non trivial amount of graphics other than UI,
9// please be aware that SDL_Renderer currently offers a limited graphic API to the end-user
10// and it might be difficult to step out of those boundaries.
11
12// Implemented features:
13// [X] Renderer: User texture binding. Use 'SDL_Texture*' as texture identifier. Read the FAQ about ImTextureID/ImTextureRef!
14// [X] Renderer: Large meshes support (64k+ vertices) even with 16-bit indices (ImGuiBackendFlags_RendererHasVtxOffset).
15// [X] Renderer: Texture updates support for dynamic font atlas (ImGuiBackendFlags_RendererHasTextures).
16// [X] Renderer: Expose selected render state for draw callbacks to use. Access in '(ImGui_ImplXXXX_RenderState*)GetPlatformIO().Renderer_RenderState'.
17
18// You can copy and use unmodified imgui_impl_* files in your project. See examples/ folder for examples of using this.
19// Prefer including the entire imgui/ repository into your project (either as a copy or as a submodule), and only build the backends you need.
20// Learn about Dear ImGui:
21// - FAQ https://dearimgui.com/faq
22// - Getting Started https://dearimgui.com/getting-started
23// - Documentation https://dearimgui.com/docs (same as your local docs/ folder).
24// - Introduction, links and more at the top of imgui.cpp

74bool ImGui_ImplSDLRenderer2_Init(SDL_Renderer* renderer)
75{
76 ImGuiIO& io = ImGui::GetIO();
77 IMGUI_CHECKVERSION();
78 IM_ASSERT(io.BackendRendererUserData == nullptr && "Already initialized a renderer backend!");
79 IM_ASSERT(renderer != nullptr && "SDL_Renderer not initialized!");
80
81 // Setup backend capabilities flags
82 ImGui_ImplSDLRenderer2_Data* bd = IM_NEW(ImGui_ImplSDLRenderer2_Data)();
83 io.BackendRendererUserData = (void*)bd;
84 io.BackendRendererName = "imgui_impl_sdlrenderer2";
85 io.BackendFlags |= ImGuiBackendFlags_RendererHasVtxOffset; // We can honor the ImDrawCmd::VtxOffset field, allowing for large meshes.
86 io.BackendFlags |= ImGuiBackendFlags_RendererHasTextures; // We can honor ImGuiPlatformIO::Textures[] requests during render.
87
88 bd->Renderer = renderer;
89
90 return true;
91}

124void ImGui_ImplSDLRenderer2_RenderDrawData(ImDrawData* draw_data, SDL_Renderer* renderer)
125{
126 // If there's a scale factor set by the user, use that instead
127 // If the user has specified a scale factor to SDL_Renderer already via SDL_RenderSetScale(), SDL will scale whatever we pass
128 // to SDL_RenderGeometryRaw() by that scale factor. In that case we don't want to be also scaling it ourselves here.
129 float rsx = 1.0f;
130 float rsy = 1.0f;
131 SDL_RenderGetScale(renderer, &rsx, &rsy);
132 ImVec2 render_scale;
133 render_scale.x = (rsx == 1.0f) ? draw_data->FramebufferScale.x : 1.0f;
134 render_scale.y = (rsy == 1.0f) ? draw_data->FramebufferScale.y : 1.0f;
135
136 // Avoid rendering when minimized, scale coordinates for retina displays (screen coordinates != framebuffer coordinates)
137 int fb_width = (int)(draw_data->DisplaySize.x * render_scale.x);
138 int fb_height = (int)(draw_data->DisplaySize.y * render_scale.y);
139 if (fb_width == 0 || fb_height == 0)
140 return;
141
142 // Catch up with texture updates. Most of the times, the list will have 1 element with an OK status, aka nothing to do.
143 // (This almost always points to ImGui::GetPlatformIO().Textures[] but is part of ImDrawData to allow overriding or disabling texture updates).
144 if (draw_data->Textures != nullptr)
145 for (ImTextureData* tex : *draw_data->Textures)
146 if (tex->Status != ImTextureStatus_OK)
147 ImGui_ImplSDLRenderer2_UpdateTexture(tex);
148
149 // Backup SDL_Renderer state that will be modified to restore it afterwards
150 struct BackupSDLRendererState
151 {
152 SDL_Rect Viewport;
153 bool ClipEnabled;
154 SDL_Rect ClipRect;
155 };
156 BackupSDLRendererState old = {};
157 old.ClipEnabled = SDL_RenderIsClipEnabled(renderer) == SDL_TRUE;
158 SDL_RenderGetViewport(renderer, &old.Viewport);
159 SDL_RenderGetClipRect(renderer, &old.ClipRect);
160
161 // Setup desired state
162 ImGui_ImplSDLRenderer2_SetupRenderState(renderer);
163
164 // Setup render state structure (for callbacks and custom texture bindings)
165 ImGuiPlatformIO& platform_io = ImGui::GetPlatformIO();
166 ImGui_ImplSDLRenderer2_RenderState render_state;
167 render_state.Renderer = renderer;
168 platform_io.Renderer_RenderState = &render_state;
169
170 // Will project scissor/clipping rectangles into framebuffer space
171 ImVec2 clip_off = draw_data->DisplayPos; // (0,0) unless using multi-viewports
172 ImVec2 clip_scale = render_scale;
173
174 // Render command lists
175 for (const ImDrawList* draw_list : draw_data->CmdLists)
176 {
177 const ImDrawVert* vtx_buffer = draw_list->VtxBuffer.Data;
178 const ImDrawIdx* idx_buffer = draw_list->IdxBuffer.Data;
179
180 for (int cmd_i = 0; cmd_i < draw_list->CmdBuffer.Size; cmd_i++)
181 {
182 const ImDrawCmd* pcmd = &draw_list->CmdBuffer[cmd_i];
183 if (pcmd->UserCallback)
184 {
185 // User callback, registered via ImDrawList::AddCallback()
186 // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
187 if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
188 ImGui_ImplSDLRenderer2_SetupRenderState(renderer);
189 else
190 pcmd->UserCallback(draw_list, pcmd);
191 }
192 else
193 {
194 // Project scissor/clipping rectangles into framebuffer space
195 ImVec2 clip_min((pcmd->ClipRect.x - clip_off.x) * clip_scale.x, (pcmd->ClipRect.y - clip_off.y) * clip_scale.y);
196 ImVec2 clip_max((pcmd->ClipRect.z - clip_off.x) * clip_scale.x, (pcmd->ClipRect.w - clip_off.y) * clip_scale.y);
197 if (clip_min.x < 0.0f) { clip_min.x = 0.0f; }
198 if (clip_min.y < 0.0f) { clip_min.y = 0.0f; }
199 if (clip_max.x > (float)fb_width) { clip_max.x = (float)fb_width; }
200 if (clip_max.y > (float)fb_height) { clip_max.y = (float)fb_height; }
201 if (clip_max.x <= clip_min.x || clip_max.y <= clip_min.y)
202 continue;
203
204 SDL_Rect r = { (int)(clip_min.x), (int)(clip_min.y), (int)(clip_max.x - clip_min.x), (int)(clip_max.y - clip_min.y) };
205 SDL_RenderSetClipRect(renderer, &r);
206
207 const float* xy = (const float*)(const void*)((const char*)(vtx_buffer + pcmd->VtxOffset) + offsetof(ImDrawVert, pos));
208 const float* uv = (const float*)(const void*)((const char*)(vtx_buffer + pcmd->VtxOffset) + offsetof(ImDrawVert, uv));
209#if SDL_VERSION_ATLEAST(2,0,19)
210 const SDL_Color* color = (const SDL_Color*)(const void*)((const char*)(vtx_buffer + pcmd->VtxOffset) + offsetof(ImDrawVert, col)); // SDL 2.0.19+
211#else
212 const int* color = (const int*)(const void*)((const char*)(vtx_buffer + pcmd->VtxOffset) + offsetof(ImDrawVert, col)); // SDL 2.0.17 and 2.0.18
213#endif
214
215 // Bind texture, Draw
216 SDL_Texture* tex = (SDL_Texture*)pcmd->GetTexID();
217 SDL_RenderGeometryRaw(renderer, tex,
218 xy, (int)sizeof(ImDrawVert),
219 color, (int)sizeof(ImDrawVert),
220 uv, (int)sizeof(ImDrawVert),
221 draw_list->VtxBuffer.Size - pcmd->VtxOffset,
222 idx_buffer + pcmd->IdxOffset, pcmd->ElemCount, sizeof(ImDrawIdx));
223 }
224 }
225 }
226 platform_io.Renderer_RenderState = nullptr;
227
228 // Restore modified SDL_Renderer state
229 SDL_RenderSetViewport(renderer, &old.Viewport);
230 SDL_RenderSetClipRect(renderer, old.ClipEnabled ? &old.ClipRect : nullptr);
231}

233void ImGui_ImplSDLRenderer2_UpdateTexture(ImTextureData* tex)
234{
235 ImGui_ImplSDLRenderer2_Data* bd = ImGui_ImplSDLRenderer2_GetBackendData();
236
237 if (tex->Status == ImTextureStatus_WantCreate)
238 {
239 // Create and upload new texture to graphics system
240 //IMGUI_DEBUG_LOG("UpdateTexture #%03d: WantCreate %dx%d\n", tex->UniqueID, tex->Width, tex->Height);
241 IM_ASSERT(tex->TexID == ImTextureID_Invalid && tex->BackendUserData == nullptr);
242 IM_ASSERT(tex->Format == ImTextureFormat_RGBA32);
243
244 // Create texture
245 // (Bilinear sampling is required by default. Set 'io.Fonts->Flags |= ImFontAtlasFlags_NoBakedLines' or 'style.AntiAliasedLinesUseTex = false' to allow point/nearest sampling)
246 SDL_Texture* sdl_texture = SDL_CreateTexture(bd->Renderer, SDL_PIXELFORMAT_RGBA32, SDL_TEXTUREACCESS_STATIC, tex->Width, tex->Height);
247 IM_ASSERT(sdl_texture != nullptr && "Backend failed to create texture!");
248 SDL_UpdateTexture(sdl_texture, nullptr, tex->GetPixels(), tex->GetPitch());
249 SDL_SetTextureBlendMode(sdl_texture, SDL_BLENDMODE_BLEND);
250 SDL_SetTextureScaleMode(sdl_texture, SDL_ScaleModeLinear);
251
252 // Store identifiers
253 tex->SetTexID((ImTextureID)(intptr_t)sdl_texture);
254 tex->SetStatus(ImTextureStatus_OK);
255 }
256 else if (tex->Status == ImTextureStatus_WantUpdates)
257 {
258 // Update selected blocks. We only ever write to textures regions which have never been used before!
259 // This backend choose to use tex->Updates[] but you can use tex->UpdateRect to upload a single region.
260 SDL_Texture* sdl_texture = (SDL_Texture*)(intptr_t)tex->TexID;
261 for (ImTextureRect& r : tex->Updates)
262 {
263 SDL_Rect sdl_r = { r.x, r.y, r.w, r.h };
264 SDL_UpdateTexture(sdl_texture, &sdl_r, tex->GetPixelsAt(r.x, r.y), tex->GetPitch());
265 }
266 tex->SetStatus(ImTextureStatus_OK);
267 }
268 else if (tex->Status == ImTextureStatus_WantDestroy)
269 {
270 if (SDL_Texture* sdl_texture = (SDL_Texture*)(intptr_t)tex->TexID)
271 SDL_DestroyTexture(sdl_texture);
272
273 // Clear identifiers and mark as destroyed (in order to allow e.g. calling InvalidateDeviceObjects while running)
274 tex->SetTexID(ImTextureID_Invalid);
275 tex->SetStatus(ImTextureStatus_Destroyed);
276 }
277}
