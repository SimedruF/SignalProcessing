1// dear imgui: Renderer Backend for SDL_Renderer for SDL3
2// (Requires: SDL 3.1.8+)
3
4// Note that SDL_Renderer is an _optional_ component of SDL3, which IMHO is now largely obsolete.
5// For a multi-platform app consider using other technologies:
6// - SDL3+SDL_GPU: SDL_GPU is SDL3 new graphics abstraction API.
7// - SDL3+DirectX, SDL3+OpenGL, SDL3+Vulkan: combine SDL with dedicated renderers.
8// If your application wants to render any non trivial amount of graphics other than UI,
9// please be aware that SDL_Renderer currently offers a limited graphic API to the end-user
10// and it might be difficult to step out of those boundaries.
11
12// Implemented features:
13// [X] Renderer: User texture binding. Use 'SDL_Texture*' as texture identifier. Read the FAQ about ImTextureID/ImTextureRef!
14// [X] Renderer: Large meshes support (64k+ vertices) even with 16-bit indices (ImGuiBackendFlags_RendererHasVtxOffset).
15// [X] Renderer: Texture updates support for dynamic font atlas (ImGuiBackendFlags_RendererHasTextures).
16// [X] Renderer: Expose selected render state for draw callbacks to use. Access in '(ImGui_ImplXXXX_RenderState*)GetPlatformIO().Renderer_RenderState'.
17
18// You can copy and use unmodified imgui_impl_* files in your project. See examples/ folder for examples of using this.
19// Prefer including the entire imgui/ repository into your project (either as a copy or as a submodule), and only build the backends you need.
20// Learn about Dear ImGui:
21// - FAQ https://dearimgui.com/faq
22// - Getting Started https://dearimgui.com/getting-started
23// - Documentation https://dearimgui.com/docs (same as your local docs/ folder).
24// - Introduction, links and more at the top of imgui.cpp
25

72bool ImGui_ImplSDLRenderer3_Init(SDL_Renderer* renderer)
73{
74 ImGuiIO& io = ImGui::GetIO();
75 IMGUI_CHECKVERSION();
76 IM_ASSERT(io.BackendRendererUserData == nullptr && "Already initialized a renderer backend!");
77 IM_ASSERT(renderer != nullptr && "SDL_Renderer not initialized!");
78
79 // Setup backend capabilities flags
80 ImGui_ImplSDLRenderer3_Data* bd = IM_NEW(ImGui_ImplSDLRenderer3_Data)();
81 io.BackendRendererUserData = (void*)bd;
82 io.BackendRendererName = "imgui_impl_sdlrenderer3";
83 io.BackendFlags |= ImGuiBackendFlags_RendererHasVtxOffset; // We can honor the ImDrawCmd::VtxOffset field, allowing for large meshes.
84 io.BackendFlags |= ImGuiBackendFlags_RendererHasTextures; // We can honor ImGuiPlatformIO::Textures[] requests during render.
85
86 bd->Renderer = renderer;
87
88 return true;
89}

123static int SDL_RenderGeometryRaw8BitColor(SDL_Renderer* renderer, ImVector<SDL_FColor>& colors_out, SDL_Texture* texture, const float* xy, int xy_stride, const SDL_Color* color, int color_stride, const float* uv, int uv_stride, int num_vertices, const void* indices, int num_indices, int size_indices)
124{
125 const Uint8* color2 = (const Uint8*)color;
126 colors_out.resize(num_vertices);
127 SDL_FColor* color3 = colors_out.Data;
128 for (int i = 0; i < num_vertices; i++)
129 {
130 color3[i].r = color->r / 255.0f;
131 color3[i].g = color->g / 255.0f;
132 color3[i].b = color->b / 255.0f;
133 color3[i].a = color->a / 255.0f;
134 color2 += color_stride;
135 color = (const SDL_Color*)color2;
136 }
137 return SDL_RenderGeometryRaw(renderer, texture, xy, xy_stride, color3, sizeof(*color3), uv, uv_stride, num_vertices, indices, num_indices, size_indices);
138}

140void ImGui_ImplSDLRenderer3_RenderDrawData(ImDrawData* draw_data, SDL_Renderer* renderer)
141{
142 ImGui_ImplSDLRenderer3_Data* bd = ImGui_ImplSDLRenderer3_GetBackendData();
143
144 // If there's a scale factor set by the user, use that instead
145 // If the user has specified a scale factor to SDL_Renderer already via SDL_RenderSetScale(), SDL will scale whatever we pass
146 // to SDL_RenderGeometryRaw() by that scale factor. In that case we don't want to be also scaling it ourselves here.
147 float rsx = 1.0f;
148 float rsy = 1.0f;
149 SDL_GetRenderScale(renderer, &rsx, &rsy);
150 ImVec2 render_scale;
151 render_scale.x = (rsx == 1.0f) ? draw_data->FramebufferScale.x : 1.0f;
152 render_scale.y = (rsy == 1.0f) ? draw_data->FramebufferScale.y : 1.0f;
153
154 // Avoid rendering when minimized, scale coordinates for retina displays (screen coordinates != framebuffer coordinates)
155 int fb_width = (int)(draw_data->DisplaySize.x * render_scale.x);
156 int fb_height = (int)(draw_data->DisplaySize.y * render_scale.y);
157 if (fb_width == 0 || fb_height == 0)
158 return;
159
160 // Catch up with texture updates. Most of the times, the list will have 1 element with an OK status, aka nothing to do.
161 // (This almost always points to ImGui::GetPlatformIO().Textures[] but is part of ImDrawData to allow overriding or disabling texture updates).
162 if (draw_data->Textures != nullptr)
163 for (ImTextureData* tex : *draw_data->Textures)
164 if (tex->Status != ImTextureStatus_OK)
165 ImGui_ImplSDLRenderer3_UpdateTexture(tex);
166
167 // Backup SDL_Renderer state that will be modified to restore it afterwards
168 struct BackupSDLRendererState
169 {
170 SDL_Rect Viewport;
171 bool ViewportEnabled;
172 bool ClipEnabled;
173 SDL_Rect ClipRect;
174 };
175 BackupSDLRendererState old = {};
176 old.ViewportEnabled = SDL_RenderViewportSet(renderer);
177 old.ClipEnabled = SDL_RenderClipEnabled(renderer);
178 SDL_GetRenderViewport(renderer, &old.Viewport);
179 SDL_GetRenderClipRect(renderer, &old.ClipRect);
180
181 // Setup desired state
182 ImGui_ImplSDLRenderer3_SetupRenderState(renderer);
183
184 // Setup render state structure (for callbacks and custom texture bindings)
185 ImGuiPlatformIO& platform_io = ImGui::GetPlatformIO();
186 ImGui_ImplSDLRenderer3_RenderState render_state;
187 render_state.Renderer = renderer;
188 platform_io.Renderer_RenderState = &render_state;
189
190 // Will project scissor/clipping rectangles into framebuffer space
191 ImVec2 clip_off = draw_data->DisplayPos; // (0,0) unless using multi-viewports
192 ImVec2 clip_scale = render_scale;
193
194 // Render command lists
195 for (const ImDrawList* draw_list : draw_data->CmdLists)
196 {
197 const ImDrawVert* vtx_buffer = draw_list->VtxBuffer.Data;
198 const ImDrawIdx* idx_buffer = draw_list->IdxBuffer.Data;
199
200 for (int cmd_i = 0; cmd_i < draw_list->CmdBuffer.Size; cmd_i++)
201 {
202 const ImDrawCmd* pcmd = &draw_list->CmdBuffer[cmd_i];
203 if (pcmd->UserCallback)
204 {
205 // User callback, registered via ImDrawList::AddCallback()
206 // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
207 if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
208 ImGui_ImplSDLRenderer3_SetupRenderState(renderer);
209 else
210 pcmd->UserCallback(draw_list, pcmd);
211 }
212 else
213 {
214 // Project scissor/clipping rectangles into framebuffer space
215 ImVec2 clip_min((pcmd->ClipRect.x - clip_off.x) * clip_scale.x, (pcmd->ClipRect.y - clip_off.y) * clip_scale.y);
216 ImVec2 clip_max((pcmd->ClipRect.z - clip_off.x) * clip_scale.x, (pcmd->ClipRect.w - clip_off.y) * clip_scale.y);
217 if (clip_min.x < 0.0f) { clip_min.x = 0.0f; }
218 if (clip_min.y < 0.0f) { clip_min.y = 0.0f; }
219 if (clip_max.x > (float)fb_width) { clip_max.x = (float)fb_width; }
220 if (clip_max.y > (float)fb_height) { clip_max.y = (float)fb_height; }
221 if (clip_max.x <= clip_min.x || clip_max.y <= clip_min.y)
222 continue;
223
224 SDL_Rect r = { (int)(clip_min.x), (int)(clip_min.y), (int)(clip_max.x - clip_min.x), (int)(clip_max.y - clip_min.y) };
225 SDL_SetRenderClipRect(renderer, &r);
226
227 const float* xy = (const float*)(const void*)((const char*)(vtx_buffer + pcmd->VtxOffset) + offsetof(ImDrawVert, pos));
228 const float* uv = (const float*)(const void*)((const char*)(vtx_buffer + pcmd->VtxOffset) + offsetof(ImDrawVert, uv));
229 const SDL_Color* color = (const SDL_Color*)(const void*)((const char*)(vtx_buffer + pcmd->VtxOffset) + offsetof(ImDrawVert, col)); // SDL 2.0.19+
230
231 // Bind texture, Draw
232 SDL_Texture* tex = (SDL_Texture*)pcmd->GetTexID();
233 SDL_RenderGeometryRaw8BitColor(renderer, bd->ColorBuffer, tex,
234 xy, (int)sizeof(ImDrawVert),
235 color, (int)sizeof(ImDrawVert),
236 uv, (int)sizeof(ImDrawVert),
237 draw_list->VtxBuffer.Size - pcmd->VtxOffset,
238 idx_buffer + pcmd->IdxOffset, pcmd->ElemCount, sizeof(ImDrawIdx));
239 }
240 }
241 }
242 platform_io.Renderer_RenderState = nullptr;
243
244 // Restore modified SDL_Renderer state
245 SDL_SetRenderViewport(renderer, old.ViewportEnabled ? &old.Viewport : nullptr);
246 SDL_SetRenderClipRect(renderer, old.ClipEnabled ? &old.ClipRect : nullptr);
247}

249void ImGui_ImplSDLRenderer3_UpdateTexture(ImTextureData* tex)
250{
251 ImGui_ImplSDLRenderer3_Data* bd = ImGui_ImplSDLRenderer3_GetBackendData();
252
253 if (tex->Status == ImTextureStatus_WantCreate)
254 {
255 // Create and upload new texture to graphics system
256 //IMGUI_DEBUG_LOG("UpdateTexture #%03d: WantCreate %dx%d\n", tex->UniqueID, tex->Width, tex->Height);
257 IM_ASSERT(tex->TexID == 0 && tex->BackendUserData == nullptr);
258 IM_ASSERT(tex->Format == ImTextureFormat_RGBA32);
259
260 // Create texture
261 // (Bilinear sampling is required by default. Set 'io.Fonts->Flags |= ImFontAtlasFlags_NoBakedLines' or 'style.AntiAliasedLinesUseTex = false' to allow point/nearest sampling)
262 SDL_Texture* sdl_texture = SDL_CreateTexture(bd->Renderer, SDL_PIXELFORMAT_RGBA32, SDL_TEXTUREACCESS_STATIC, tex->Width, tex->Height);
263 IM_ASSERT(sdl_texture != nullptr && "Backend failed to create texture!");
264 SDL_UpdateTexture(sdl_texture, nullptr, tex->GetPixels(), tex->GetPitch());
265 SDL_SetTextureBlendMode(sdl_texture, SDL_BLENDMODE_BLEND);
266 SDL_SetTextureScaleMode(sdl_texture, SDL_SCALEMODE_LINEAR);
267
268 // Store identifiers
269 tex->SetTexID((ImTextureID)(intptr_t)sdl_texture);
270 tex->SetStatus(ImTextureStatus_OK);
271 }
272 else if (tex->Status == ImTextureStatus_WantUpdates)
273 {
274 // Update selected blocks. We only ever write to textures regions which have never been used before!
275 // This backend choose to use tex->Updates[] but you can use tex->UpdateRect to upload a single region.
276 SDL_Texture* sdl_texture = (SDL_Texture*)(intptr_t)tex->TexID;
277 for (ImTextureRect& r : tex->Updates)
278 {
279 SDL_Rect sdl_r = { r.x, r.y, r.w, r.h };
280 SDL_UpdateTexture(sdl_texture, &sdl_r, tex->GetPixelsAt(r.x, r.y), tex->GetPitch());
281 }
282 tex->SetStatus(ImTextureStatus_OK);
283 }
284 else if (tex->Status == ImTextureStatus_WantDestroy)
285 {
286 if (SDL_Texture* sdl_texture = (SDL_Texture*)(intptr_t)tex->TexID)
287 SDL_DestroyTexture(sdl_texture);
288
289 // Clear identifiers and mark as destroyed (in order to allow e.g. calling InvalidateDeviceObjects while running)
290 tex->SetTexID(ImTextureID_Invalid);
291 tex->SetStatus(ImTextureStatus_Destroyed);
292 }
293}
