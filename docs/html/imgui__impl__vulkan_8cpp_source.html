1// dear imgui: Renderer Backend for Vulkan
2// This needs to be used along with a Platform Backend (e.g. GLFW, SDL, Win32, custom..)
3
4// Implemented features:
5// [!] Renderer: User texture binding. Use 'VkDescriptorSet' as texture identifier. Call ImGui_ImplVulkan_AddTexture() to register one. Read the FAQ about ImTextureID/ImTextureRef + https://github.com/ocornut/imgui/pull/914 for discussions.
6// [X] Renderer: Large meshes support (64k+ vertices) even with 16-bit indices (ImGuiBackendFlags_RendererHasVtxOffset).
7// [X] Renderer: Texture updates support for dynamic font atlas (ImGuiBackendFlags_RendererHasTextures).
8// [X] Renderer: Expose selected render state for draw callbacks to use. Access in '(ImGui_ImplXXXX_RenderState*)GetPlatformIO().Renderer_RenderState'.
9
10// The aim of imgui_impl_vulkan.h/.cpp is to be usable in your engine without any modification.
11// IF YOU FEEL YOU NEED TO MAKE ANY CHANGE TO THIS CODE, please share them and your feedback at https://github.com/ocornut/imgui/
12
13// You can use unmodified imgui_impl_* files in your project. See examples/ folder for examples of using this.
14// Prefer including the entire imgui/ repository into your project (either as a copy or as a submodule), and only build the backends you need.

21// Important note to the reader who wish to integrate imgui_impl_vulkan.cpp/.h in their own engine/app.

434static void CreateOrResizeBuffer(VkBuffer& buffer, VkDeviceMemory& buffer_memory, VkDeviceSize& buffer_size, VkDeviceSize new_size, VkBufferUsageFlagBits usage)
435{
436 ImGui_ImplVulkan_Data* bd = ImGui_ImplVulkan_GetBackendData();
437 ImGui_ImplVulkan_InitInfo* v = &bd->VulkanInitInfo;
438 VkResult err;
439 if (buffer != VK_NULL_HANDLE)
440 vkDestroyBuffer(v->Device, buffer, v->Allocator);
441 if (buffer_memory != VK_NULL_HANDLE)
442 vkFreeMemory(v->Device, buffer_memory, v->Allocator);
443
444 VkDeviceSize buffer_size_aligned = AlignBufferSize(IM_MAX(v->MinAllocationSize, new_size), bd->BufferMemoryAlignment);
445 VkBufferCreateInfo buffer_info = {};
446 buffer_info.sType = VK_STRUCTURE_TYPE_BUFFER_CREATE_INFO;
447 buffer_info.size = buffer_size_aligned;
448 buffer_info.usage = usage;
449 buffer_info.sharingMode = VK_SHARING_MODE_EXCLUSIVE;
450 err = vkCreateBuffer(v->Device, &buffer_info, v->Allocator, &buffer);
451 check_vk_result(err);
452
453 VkMemoryRequirements req;
454 vkGetBufferMemoryRequirements(v->Device, buffer, &req);
455 bd->BufferMemoryAlignment = (bd->BufferMemoryAlignment > req.alignment) ? bd->BufferMemoryAlignment : req.alignment;
456 VkMemoryAllocateInfo alloc_info = {};
457 alloc_info.sType = VK_STRUCTURE_TYPE_MEMORY_ALLOCATE_INFO;
458 alloc_info.allocationSize = req.size;
459 alloc_info.memoryTypeIndex = ImGui_ImplVulkan_MemoryType(VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT, req.memoryTypeBits);
460 err = vkAllocateMemory(v->Device, &alloc_info, v->Allocator, &buffer_memory);
461 check_vk_result(err);
462
463 err = vkBindBufferMemory(v->Device, buffer, buffer_memory, 0);
464 check_vk_result(err);
465 buffer_size = buffer_size_aligned;
466}

468static void ImGui_ImplVulkan_SetupRenderState(ImDrawData* draw_data, VkPipeline pipeline, VkCommandBuffer command_buffer, ImGui_ImplVulkan_FrameRenderBuffers* rb, int fb_width, int fb_height)
469{
470 ImGui_ImplVulkan_Data* bd = ImGui_ImplVulkan_GetBackendData();
471
472 // Bind pipeline:
473 {
474 vkCmdBindPipeline(command_buffer, VK_PIPELINE_BIND_POINT_GRAPHICS, pipeline);
475 }
476
477 // Bind Vertex And Index Buffer:
478 if (draw_data->TotalVtxCount > 0)
479 {
480 VkBuffer vertex_buffers[1] = { rb->VertexBuffer };
481 VkDeviceSize vertex_offset[1] = { 0 };
482 vkCmdBindVertexBuffers(command_buffer, 0, 1, vertex_buffers, vertex_offset);
483 vkCmdBindIndexBuffer(command_buffer, rb->IndexBuffer, 0, sizeof(ImDrawIdx) == 2 ? VK_INDEX_TYPE_UINT16 : VK_INDEX_TYPE_UINT32);
484 }
485
486 // Setup viewport:
487 {
488 VkViewport viewport;
489 viewport.x = 0;
490 viewport.y = 0;
491 viewport.width = (float)fb_width;
492 viewport.height = (float)fb_height;
493 viewport.minDepth = 0.0f;
494 viewport.maxDepth = 1.0f;
495 vkCmdSetViewport(command_buffer, 0, 1, &viewport);
496 }
497
498 // Setup scale and translation:
499 // Our visible imgui space lies from draw_data->DisplayPps (top left) to draw_data->DisplayPos+data_data->DisplaySize (bottom right). DisplayPos is (0,0) for single viewport apps.
500 {
501 float scale[2];
502 scale[0] = 2.0f / draw_data->DisplaySize.x;
503 scale[1] = 2.0f / draw_data->DisplaySize.y;
504 float translate[2];
505 translate[0] = -1.0f - draw_data->DisplayPos.x * scale[0];
506 translate[1] = -1.0f - draw_data->DisplayPos.y * scale[1];
507 vkCmdPushConstants(command_buffer, bd->PipelineLayout, VK_SHADER_STAGE_VERTEX_BIT, sizeof(float) * 0, sizeof(float) * 2, scale);
508 vkCmdPushConstants(command_buffer, bd->PipelineLayout, VK_SHADER_STAGE_VERTEX_BIT, sizeof(float) * 2, sizeof(float) * 2, translate);
509 }
510}

512// Render function
513void ImGui_ImplVulkan_RenderDrawData(ImDrawData* draw_data, VkCommandBuffer command_buffer, VkPipeline pipeline)
514{
515 // Avoid rendering when minimized, scale coordinates for retina displays (screen coordinates != framebuffer coordinates)
516 int fb_width = (int)(draw_data->DisplaySize.x * draw_data->FramebufferScale.x);
517 int fb_height = (int)(draw_data->DisplaySize.y * draw_data->FramebufferScale.y);
518 if (fb_width <= 0 || fb_height <= 0)
519 return;
520
521 // Catch up with texture updates. Most of the times, the list will have 1 element with an OK status, aka nothing to do.
522 // (This almost always points to ImGui::GetPlatformIO().Textures[] but is part of ImDrawData to allow overriding or disabling texture updates).
523 if (draw_data->Textures != nullptr)
524 for (ImTextureData* tex : *draw_data->Textures)
525 if (tex->Status != ImTextureStatus_OK)
526 ImGui_ImplVulkan_UpdateTexture(tex);
527
528 ImGui_ImplVulkan_Data* bd = ImGui_ImplVulkan_GetBackendData();
529 ImGui_ImplVulkan_InitInfo* v = &bd->VulkanInitInfo;
530 if (pipeline == VK_NULL_HANDLE)
531 pipeline = bd->Pipeline;
532
533 // Allocate array to store enough vertex/index buffers
534 ImGui_ImplVulkan_WindowRenderBuffers* wrb = &bd->MainWindowRenderBuffers;
535 if (wrb->FrameRenderBuffers.Size == 0)
536 {
537 wrb->Index = 0;
538 wrb->Count = v->ImageCount;
539 wrb->FrameRenderBuffers.resize(wrb->Count);
540 memset((void*)wrb->FrameRenderBuffers.Data, 0, wrb->FrameRenderBuffers.size_in_bytes());
541 }
542 IM_ASSERT(wrb->Count == v->ImageCount);
543 wrb->Index = (wrb->Index + 1) % wrb->Count;
544 ImGui_ImplVulkan_FrameRenderBuffers* rb = &wrb->FrameRenderBuffers[wrb->Index];
545
546 if (draw_data->TotalVtxCount > 0)
547 {
548 // Create or resize the vertex/index buffers
549 VkDeviceSize vertex_size = AlignBufferSize(draw_data->TotalVtxCount * sizeof(ImDrawVert), bd->BufferMemoryAlignment);
550 VkDeviceSize index_size = AlignBufferSize(draw_data->TotalIdxCount * sizeof(ImDrawIdx), bd->BufferMemoryAlignment);
551 if (rb->VertexBuffer == VK_NULL_HANDLE || rb->VertexBufferSize < vertex_size)
552 CreateOrResizeBuffer(rb->VertexBuffer, rb->VertexBufferMemory, rb->VertexBufferSize, vertex_size, VK_BUFFER_USAGE_VERTEX_BUFFER_BIT);
553 if (rb->IndexBuffer == VK_NULL_HANDLE || rb->IndexBufferSize < index_size)
554 CreateOrResizeBuffer(rb->IndexBuffer, rb->IndexBufferMemory, rb->IndexBufferSize, index_size, VK_BUFFER_USAGE_INDEX_BUFFER_BIT);
555
556 // Upload vertex/index data into a single contiguous GPU buffer
557 ImDrawVert* vtx_dst = nullptr;
558 ImDrawIdx* idx_dst = nullptr;
559 VkResult err = vkMapMemory(v->Device, rb->VertexBufferMemory, 0, vertex_size, 0, (void**)&vtx_dst);
560 check_vk_result(err);
561 err = vkMapMemory(v->Device, rb->IndexBufferMemory, 0, index_size, 0, (void**)&idx_dst);
562 check_vk_result(err);
563 for (const ImDrawList* draw_list : draw_data->CmdLists)
564 {
565 memcpy(vtx_dst, draw_list->VtxBuffer.Data, draw_list->VtxBuffer.Size * sizeof(ImDrawVert));
566 memcpy(idx_dst, draw_list->IdxBuffer.Data, draw_list->IdxBuffer.Size * sizeof(ImDrawIdx));
567 vtx_dst += draw_list->VtxBuffer.Size;
568 idx_dst += draw_list->IdxBuffer.Size;
569 }
570 VkMappedMemoryRange range[2] = {};
571 range[0].sType = VK_STRUCTURE_TYPE_MAPPED_MEMORY_RANGE;
572 range[0].memory = rb->VertexBufferMemory;
573 range[0].size = VK_WHOLE_SIZE;
574 range[1].sType = VK_STRUCTURE_TYPE_MAPPED_MEMORY_RANGE;
575 range[1].memory = rb->IndexBufferMemory;
576 range[1].size = VK_WHOLE_SIZE;
577 err = vkFlushMappedMemoryRanges(v->Device, 2, range);
578 check_vk_result(err);
579 vkUnmapMemory(v->Device, rb->VertexBufferMemory);
580 vkUnmapMemory(v->Device, rb->IndexBufferMemory);
581 }
582
583 // Setup desired Vulkan state
584 ImGui_ImplVulkan_SetupRenderState(draw_data, pipeline, command_buffer, rb, fb_width, fb_height);
585
586 // Setup render state structure (for callbacks and custom texture bindings)
587 ImGuiPlatformIO& platform_io = ImGui::GetPlatformIO();
588 ImGui_ImplVulkan_RenderState render_state;
589 render_state.CommandBuffer = command_buffer;
590 render_state.Pipeline = pipeline;
591 render_state.PipelineLayout = bd->PipelineLayout;
592 platform_io.Renderer_RenderState = &render_state;
593
594 // Will project scissor/clipping rectangles into framebuffer space
595 ImVec2 clip_off = draw_data->DisplayPos; // (0,0) unless using multi-viewports
596 ImVec2 clip_scale = draw_data->FramebufferScale; // (1,1) unless using retina display which are often (2,2)
597
598 // Render command lists
599 // (Because we merged all buffers into a single one, we maintain our own offset into them)
600 VkDescriptorSet last_desc_set = VK_NULL_HANDLE;
601 int global_vtx_offset = 0;
602 int global_idx_offset = 0;
603 for (const ImDrawList* draw_list : draw_data->CmdLists)
604 {
605 for (int cmd_i = 0; cmd_i < draw_list->CmdBuffer.Size; cmd_i++)
606 {
607 const ImDrawCmd* pcmd = &draw_list->CmdBuffer[cmd_i];
608 if (pcmd->UserCallback != nullptr)
609 {
610 // User callback, registered via ImDrawList::AddCallback()
611 // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
612 if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
613 ImGui_ImplVulkan_SetupRenderState(draw_data, pipeline, command_buffer, rb, fb_width, fb_height);
614 else
615 pcmd->UserCallback(draw_list, pcmd);
616 last_desc_set = VK_NULL_HANDLE;
617 }
618 else
619 {
620 // Project scissor/clipping rectangles into framebuffer space
621 ImVec2 clip_min((pcmd->ClipRect.x - clip_off.x) * clip_scale.x, (pcmd->ClipRect.y - clip_off.y) * clip_scale.y);
622 ImVec2 clip_max((pcmd->ClipRect.z - clip_off.x) * clip_scale.x, (pcmd->ClipRect.w - clip_off.y) * clip_scale.y);
623
624 // Clamp to viewport as vkCmdSetScissor() won't accept values that are off bounds
625 if (clip_min.x < 0.0f) { clip_min.x = 0.0f; }
626 if (clip_min.y < 0.0f) { clip_min.y = 0.0f; }
627 if (clip_max.x > fb_width) { clip_max.x = (float)fb_width; }
628 if (clip_max.y > fb_height) { clip_max.y = (float)fb_height; }
629 if (clip_max.x <= clip_min.x || clip_max.y <= clip_min.y)
630 continue;
631
632 // Apply scissor/clipping rectangle
633 VkRect2D scissor;
634 scissor.offset.x = (int32_t)(clip_min.x);
635 scissor.offset.y = (int32_t)(clip_min.y);
636 scissor.extent.width = (uint32_t)(clip_max.x - clip_min.x);
637 scissor.extent.height = (uint32_t)(clip_max.y - clip_min.y);
638 vkCmdSetScissor(command_buffer, 0, 1, &scissor);
639
640 // Bind DescriptorSet with font or user texture
641 VkDescriptorSet desc_set = (VkDescriptorSet)pcmd->GetTexID();
642 if (desc_set != last_desc_set)
643 vkCmdBindDescriptorSets(command_buffer, VK_PIPELINE_BIND_POINT_GRAPHICS, bd->PipelineLayout, 0, 1, &desc_set, 0, nullptr);
644 last_desc_set = desc_set;
645
646 // Draw
647 vkCmdDrawIndexed(command_buffer, pcmd->ElemCount, 1, pcmd->IdxOffset + global_idx_offset, pcmd->VtxOffset + global_vtx_offset, 0);
648 }
649 }
650 global_idx_offset += draw_list->IdxBuffer.Size;
651 global_vtx_offset += draw_list->VtxBuffer.Size;
652 }
653 platform_io.Renderer_RenderState = nullptr;
654
655 // Note: at this point both vkCmdSetViewport() and vkCmdSetScissor() have been called.
656 // Our last values will leak into user/application rendering IF:
657 // - Your app uses a pipeline with VK_DYNAMIC_STATE_VIEWPORT or VK_DYNAMIC_STATE_SCISSOR dynamic state
658 // - And you forgot to call vkCmdSetViewport() and vkCmdSetScissor() yourself to explicitly set that state.
659 // If you use VK_DYNAMIC_STATE_VIEWPORT or VK_DYNAMIC_STATE_SCISSOR you are responsible for setting the values before rendering.
660 // In theory we should aim to backup/restore those values but I am not sure this is possible.
661 // We perform a call to vkCmdSetScissor() to set back a full viewport which is likely to fix things for 99% users but technically this is not perfect. (See github #4644)
662 VkRect2D scissor = { { 0, 0 }, { (uint32_t)fb_width, (uint32_t)fb_height } };
663 vkCmdSetScissor(command_buffer, 0, 1, &scissor);
664}

703 {
704 VkImageCreateInfo info = {};
705 info.sType = VK_STRUCTURE_TYPE_IMAGE_CREATE_INFO;
706 info.imageType = VK_IMAGE_TYPE_2D;
707 info.format = VK_FORMAT_R8G8B8A8_UNORM;
708 info.extent.width = tex->Width;
709 info.extent.height = tex->Height;
710 info.extent.depth = 1;
711 info.mipLevels = 1;
712 info.arrayLayers = 1;
713 info.samples = VK_SAMPLE_COUNT_1_BIT;
714 info.tiling = VK_IMAGE_TILING_OPTIMAL;
715 info.usage = VK_IMAGE_USAGE_SAMPLED_BIT | VK_IMAGE_USAGE_TRANSFER_DST_BIT;
716 info.sharingMode = VK_SHARING_MODE_EXCLUSIVE;
717 info.initialLayout = VK_IMAGE_LAYOUT_UNDEFINED;
718 err = vkCreateImage(v->Device, &info, v->Allocator, &backend_tex->Image);
719 check_vk_result(err);
720 VkMemoryRequirements req;
721 vkGetImageMemoryRequirements(v->Device, backend_tex->Image, &req);
722 VkMemoryAllocateInfo alloc_info = {};
723 alloc_info.sType = VK_STRUCTURE_TYPE_MEMORY_ALLOCATE_INFO;
724 alloc_info.allocationSize = IM_MAX(v->MinAllocationSize, req.size);
725 alloc_info.memoryTypeIndex = ImGui_ImplVulkan_MemoryType(VK_MEMORY_PROPERTY_DEVICE_LOCAL_BIT, req.memoryTypeBits);
726 err = vkAllocateMemory(v->Device, &alloc_info, v->Allocator, &backend_tex->Memory);
727 check_vk_result(err);
728 err = vkBindImageMemory(v->Device, backend_tex->Image, backend_tex->Memory, 0);
729 check_vk_result(err);
730 }
731
732 // Create the Image View:
733 {
734 VkImageViewCreateInfo info = {};
735 info.sType = VK_STRUCTURE_TYPE_IMAGE_VIEW_CREATE_INFO;
736 info.image = backend_tex->Image;
737 info.viewType = VK_IMAGE_VIEW_TYPE_2D;
738 info.format = VK_FORMAT_R8G8B8A8_UNORM;
739 info.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
740 info.subresourceRange.levelCount = 1;
741 info.subresourceRange.layerCount = 1;
742 err = vkCreateImageView(v->Device, &info, v->Allocator, &backend_tex->ImageView);
743 check_vk_result(err);
744 }
745
746 // Create the Descriptor Set
747 backend_tex->DescriptorSet = ImGui_ImplVulkan_AddTexture(bd->TexSamplerLinear, backend_tex->ImageView, VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL);
748
749 // Store identifiers
750 tex->SetTexID((ImTextureID)backend_tex->DescriptorSet);
751 tex->BackendUserData = backend_tex;
752 }
753
754 if (tex->Status == ImTextureStatus_WantCreate || tex->Status == ImTextureStatus_WantUpdates)
755 {
756 ImGui_ImplVulkan_Texture* backend_tex = (ImGui_ImplVulkan_Texture*)tex->BackendUserData;
757
758 // Update full texture or selected blocks. We only ever write to textures regions which have never been used before!
759 // This backend choose to use tex->UpdateRect but you can use tex->Updates[] to upload individual regions.
760 // We could use the smaller rect on _WantCreate but using the full rect allows us to clear the texture.
761 const int upload_x = (tex->Status == ImTextureStatus_WantCreate) ? 0 : tex->UpdateRect.x;
762 const int upload_y = (tex->Status == ImTextureStatus_WantCreate) ? 0 : tex->UpdateRect.y;
763 const int upload_w = (tex->Status == ImTextureStatus_WantCreate) ? tex->Width : tex->UpdateRect.w;
764 const int upload_h = (tex->Status == ImTextureStatus_WantCreate) ? tex->Height : tex->UpdateRect.h;
765
766 // Create the Upload Buffer:
767 VkDeviceMemory upload_buffer_memory;
768
769 VkBuffer upload_buffer;
770 VkDeviceSize upload_pitch = upload_w * tex->BytesPerPixel;
771 VkDeviceSize upload_size = AlignBufferSize(upload_h * upload_pitch, bd->NonCoherentAtomSize);
772 {
773 VkBufferCreateInfo buffer_info = {};
774 buffer_info.sType = VK_STRUCTURE_TYPE_BUFFER_CREATE_INFO;
775 buffer_info.size = upload_size;
776 buffer_info.usage = VK_BUFFER_USAGE_TRANSFER_SRC_BIT;
777 buffer_info.sharingMode = VK_SHARING_MODE_EXCLUSIVE;
778 err = vkCreateBuffer(v->Device, &buffer_info, v->Allocator, &upload_buffer);
779 check_vk_result(err);
780 VkMemoryRequirements req;
781 vkGetBufferMemoryRequirements(v->Device, upload_buffer, &req);
782 bd->BufferMemoryAlignment = (bd->BufferMemoryAlignment > req.alignment) ? bd->BufferMemoryAlignment : req.alignment;
783 VkMemoryAllocateInfo alloc_info = {};
784 alloc_info.sType = VK_STRUCTURE_TYPE_MEMORY_ALLOCATE_INFO;
785 alloc_info.allocationSize = IM_MAX(v->MinAllocationSize, req.size);
786 alloc_info.memoryTypeIndex = ImGui_ImplVulkan_MemoryType(VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT, req.memoryTypeBits);
787 err = vkAllocateMemory(v->Device, &alloc_info, v->Allocator, &upload_buffer_memory);
788 check_vk_result(err);
789 err = vkBindBufferMemory(v->Device, upload_buffer, upload_buffer_memory, 0);
790 check_vk_result(err);
791 }
792
793 // Upload to Buffer:
794 {
795 char* map = nullptr;
796 err = vkMapMemory(v->Device, upload_buffer_memory, 0, upload_size, 0, (void**)(&map));
797 check_vk_result(err);
798 for (int y = 0; y < upload_h; y++)
799 memcpy(map + upload_pitch * y, tex->GetPixelsAt(upload_x, upload_y + y), (size_t)upload_pitch);
800 VkMappedMemoryRange range[1] = {};
801 range[0].sType = VK_STRUCTURE_TYPE_MAPPED_MEMORY_RANGE;
802 range[0].memory = upload_buffer_memory;
803 range[0].size = upload_size;
804 err = vkFlushMappedMemoryRanges(v->Device, 1, range);
805 check_vk_result(err);
806 vkUnmapMemory(v->Device, upload_buffer_memory);
807 }
808
809 // Start command buffer
810 {
811 err = vkResetCommandPool(v->Device, bd->TexCommandPool, 0);
812 check_vk_result(err);
813 VkCommandBufferBeginInfo begin_info = {};
814 begin_info.sType = VK_STRUCTURE_TYPE_COMMAND_BUFFER_BEGIN_INFO;
815 begin_info.flags |= VK_COMMAND_BUFFER_USAGE_ONE_TIME_SUBMIT_BIT;
816 err = vkBeginCommandBuffer(bd->TexCommandBuffer, &begin_info);
817 check_vk_result(err);
818 }
819
820 // Copy to Image:
821 {
822 VkBufferMemoryBarrier upload_barrier[1] = {};
823 upload_barrier[0].sType = VK_STRUCTURE_TYPE_BUFFER_MEMORY_BARRIER;
824 upload_barrier[0].srcAccessMask = VK_ACCESS_HOST_WRITE_BIT;
825 upload_barrier[0].dstAccessMask = VK_ACCESS_TRANSFER_READ_BIT;
826 upload_barrier[0].srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
827 upload_barrier[0].dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
828 upload_barrier[0].buffer = upload_buffer;
829 upload_barrier[0].offset = 0;
830 upload_barrier[0].size = upload_size;
831
832 VkImageMemoryBarrier copy_barrier[1] = {};
833 copy_barrier[0].sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
834 copy_barrier[0].dstAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
835 copy_barrier[0].oldLayout = (tex->Status == ImTextureStatus_WantCreate) ? VK_IMAGE_LAYOUT_UNDEFINED : VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
836 copy_barrier[0].newLayout = VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL;
837 copy_barrier[0].srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
838 copy_barrier[0].dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
839 copy_barrier[0].image = backend_tex->Image;
840 copy_barrier[0].subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
841 copy_barrier[0].subresourceRange.levelCount = 1;
842 copy_barrier[0].subresourceRange.layerCount = 1;
843 vkCmdPipelineBarrier(bd->TexCommandBuffer, VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT | VK_PIPELINE_STAGE_HOST_BIT, VK_PIPELINE_STAGE_TRANSFER_BIT, 0, 0, nullptr, 1, upload_barrier, 1, copy_barrier);
844
845 VkBufferImageCopy region = {};
846 region.imageSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
847 region.imageSubresource.layerCount = 1;
848 region.imageExtent.width = upload_w;
849 region.imageExtent.height = upload_h;
850 region.imageExtent.depth = 1;
851 region.imageOffset.x = upload_x;
852 region.imageOffset.y = upload_y;
853 vkCmdCopyBufferToImage(bd->TexCommandBuffer, upload_buffer, backend_tex->Image, VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL, 1, &region);
854
855 VkImageMemoryBarrier use_barrier[1] = {};
856 use_barrier[0].sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
857 use_barrier[0].srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
858 use_barrier[0].dstAccessMask = VK_ACCESS_SHADER_READ_BIT;
859 use_barrier[0].oldLayout = VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL;
860 use_barrier[0].newLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
861 use_barrier[0].srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
862 use_barrier[0].dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
863 use_barrier[0].image = backend_tex->Image;
864 use_barrier[0].subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
865 use_barrier[0].subresourceRange.levelCount = 1;
866 use_barrier[0].subresourceRange.layerCount = 1;
867 vkCmdPipelineBarrier(bd->TexCommandBuffer, VK_PIPELINE_STAGE_TRANSFER_BIT, VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT, 0, 0, nullptr, 0, nullptr, 1, use_barrier);
868 }
869
870 // End command buffer
871 {
872 VkSubmitInfo end_info = {};
873 end_info.sType = VK_STRUCTURE_TYPE_SUBMIT_INFO;
874 end_info.commandBufferCount = 1;
875 end_info.pCommandBuffers = &bd->TexCommandBuffer;
876 err = vkEndCommandBuffer(bd->TexCommandBuffer);
877 check_vk_result(err);
878 err = vkQueueSubmit(v->Queue, 1, &end_info, VK_NULL_HANDLE);
879 check_vk_result(err);
880 }
881
882 err = vkQueueWaitIdle(v->Queue); // FIXME-OPT: Suboptimal!
883 check_vk_result(err);
884 vkDestroyBuffer(v->Device, upload_buffer, v->Allocator);
885 vkFreeMemory(v->Device, upload_buffer_memory, v->Allocator);
886
887 tex->SetStatus(ImTextureStatus_OK);
888 }
889
890 if (tex->Status == ImTextureStatus_WantDestroy && tex->UnusedFrames >= (int)bd->VulkanInitInfo.ImageCount)
891 ImGui_ImplVulkan_DestroyTexture(tex);
892}

1040bool ImGui_ImplVulkan_CreateDeviceObjects()
1041{
1042 ImGui_ImplVulkan_Data* bd = ImGui_ImplVulkan_GetBackendData();
1043 ImGui_ImplVulkan_InitInfo* v = &bd->VulkanInitInfo;
1044 VkResult err;
1045
1046 if (!bd->TexSamplerLinear)
1047 {
1048 // Bilinear sampling is required by default. Set 'io.Fonts->Flags |= ImFontAtlasFlags_NoBakedLines' or 'style.AntiAliasedLinesUseTex = false' to allow point/nearest sampling.
1049 VkSamplerCreateInfo info = {};
1050 info.sType = VK_STRUCTURE_TYPE_SAMPLER_CREATE_INFO;
1051 info.magFilter = VK_FILTER_LINEAR;
1052 info.minFilter = VK_FILTER_LINEAR;
1053 info.mipmapMode = VK_SAMPLER_MIPMAP_MODE_LINEAR;
1054 info.addressModeU = VK_SAMPLER_ADDRESS_MODE_CLAMP_TO_EDGE;
1055 info.addressModeV = VK_SAMPLER_ADDRESS_MODE_CLAMP_TO_EDGE;
1056 info.addressModeW = VK_SAMPLER_ADDRESS_MODE_CLAMP_TO_EDGE;
1057 info.minLod = -1000;
1058 info.maxLod = 1000;
1059 info.maxAnisotropy = 1.0f;
1060 err = vkCreateSampler(v->Device, &info, v->Allocator, &bd->TexSamplerLinear);
1061 check_vk_result(err);
1062 }
1063
1064 if (!bd->DescriptorSetLayout)
1065 {
1066 VkDescriptorSetLayoutBinding binding[1] = {};
1067 binding[0].descriptorType = VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER;
1068 binding[0].descriptorCount = 1;
1069 binding[0].stageFlags = VK_SHADER_STAGE_FRAGMENT_BIT;
1070 VkDescriptorSetLayoutCreateInfo info = {};
1071 info.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_SET_LAYOUT_CREATE_INFO;
1072 info.bindingCount = 1;
1073 info.pBindings = binding;
1074 err = vkCreateDescriptorSetLayout(v->Device, &info, v->Allocator, &bd->DescriptorSetLayout);
1075 check_vk_result(err);
1076 }
1077
1078 if (v->DescriptorPoolSize != 0)
1079 {
1080 IM_ASSERT(v->DescriptorPoolSize >= IMGUI_IMPL_VULKAN_MINIMUM_IMAGE_SAMPLER_POOL_SIZE);
1081 VkDescriptorPoolSize pool_size = { VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER, v->DescriptorPoolSize };
1082 VkDescriptorPoolCreateInfo pool_info = {};
1083 pool_info.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_POOL_CREATE_INFO;
1084 pool_info.flags = VK_DESCRIPTOR_POOL_CREATE_FREE_DESCRIPTOR_SET_BIT;
1085 pool_info.maxSets = v->DescriptorPoolSize;
1086 pool_info.poolSizeCount = 1;
1087 pool_info.pPoolSizes = &pool_size;
1088
1089 err = vkCreateDescriptorPool(v->Device, &pool_info, v->Allocator, &bd->DescriptorPool);
1090 check_vk_result(err);
1091 }
1092
1093 if (!bd->PipelineLayout)
1094 {
1095 // Constants: we are using 'vec2 offset' and 'vec2 scale' instead of a full 3d projection matrix
1096 VkPushConstantRange push_constants[1] = {};
1097 push_constants[0].stageFlags = VK_SHADER_STAGE_VERTEX_BIT;
1098 push_constants[0].offset = sizeof(float) * 0;
1099 push_constants[0].size = sizeof(float) * 4;
1100 VkDescriptorSetLayout set_layout[1] = { bd->DescriptorSetLayout };
1101 VkPipelineLayoutCreateInfo layout_info = {};
1102 layout_info.sType = VK_STRUCTURE_TYPE_PIPELINE_LAYOUT_CREATE_INFO;
1103 layout_info.setLayoutCount = 1;
1104 layout_info.pSetLayouts = set_layout;
1105 layout_info.pushConstantRangeCount = 1;
1106 layout_info.pPushConstantRanges = push_constants;
1107 err = vkCreatePipelineLayout(v->Device, &layout_info, v->Allocator, &bd->PipelineLayout);
1108 check_vk_result(err);
1109 }
1110
1111 // Create pipeline
1112 bool create_main_pipeline = (v->PipelineInfoMain.RenderPass != VK_NULL_HANDLE);
1113#ifdef IMGUI_IMPL_VULKAN_HAS_DYNAMIC_RENDERING
1114 create_main_pipeline |= (v->UseDynamicRendering && v->PipelineInfoMain.PipelineRenderingCreateInfo.sType == VK_STRUCTURE_TYPE_PIPELINE_RENDERING_CREATE_INFO_KHR);
1115#endif
1116 if (create_main_pipeline)
1117 ImGui_ImplVulkan_CreateMainPipeline(&v->PipelineInfoMain);
1118
1119 // Create command pool/buffer for texture upload
1120 if (!bd->TexCommandPool)
1121 {
1122 VkCommandPoolCreateInfo info = {};
1123 info.sType = VK_STRUCTURE_TYPE_COMMAND_POOL_CREATE_INFO;
1124 info.flags = 0;
1125 info.queueFamilyIndex = v->QueueFamily;
1126 err = vkCreateCommandPool(v->Device, &info, v->Allocator, &bd->TexCommandPool);

1425VkSurfaceFormatKHR ImGui_ImplVulkanH_SelectSurfaceFormat(VkPhysicalDevice physical_device, VkSurfaceKHR surface, const VkFormat* request_formats, int request_formats_count, VkColorSpaceKHR request_color_space)
1426{
1427 IM_ASSERT(g_FunctionsLoaded && "Need to call ImGui_ImplVulkan_LoadFunctions() if IMGUI_IMPL_VULKAN_NO_PROTOTYPES or VK_NO_PROTOTYPES are set!");
1428 IM_ASSERT(request_formats != nullptr);
1429 IM_ASSERT(request_formats_count > 0);
1430
1431 // Per Spec Format and View Format are expected to be the same unless VK_IMAGE_CREATE_MUTABLE_BIT was set at image creation
1432 // Assuming that the default behavior is without setting this bit, there is no need for separate Swapchain image and image view format
1433 // Additionally several new color spaces were introduced with Vulkan Spec v1.0.40,
1434 // hence we must make sure that a format with the mostly available color space, VK_COLOR_SPACE_SRGB_NONLINEAR_KHR, is found and used.
1435 uint32_t avail_count;
1436 vkGetPhysicalDeviceSurfaceFormatsKHR(physical_device, surface, &avail_count, nullptr);
1437 ImVector<VkSurfaceFormatKHR> avail_format;
1438 avail_format.resize((int)avail_count);
1439 vkGetPhysicalDeviceSurfaceFormatsKHR(physical_device, surface, &avail_count, avail_format.Data);
1440
1441 // First check if only one format, VK_FORMAT_UNDEFINED, is available, which would imply that any format is available
1442 if (avail_count == 1)
1443 {
1444 if (avail_format[0].format == VK_FORMAT_UNDEFINED)
1445 {
1446 VkSurfaceFormatKHR ret;
1447 ret.format = request_formats[0];
1448 ret.colorSpace = request_color_space;
1449 return ret;
1450 }
1451 else
1452 {
1453 // No point in searching another format
1454 return avail_format[0];
1455 }
1456 }
1457 else
1458 {
1459 // Request several formats, the first found will be used
1460 for (int request_i = 0; request_i < request_formats_count; request_i++)
1461 for (uint32_t avail_i = 0; avail_i < avail_count; avail_i++)
1462 if (avail_format[avail_i].format == request_formats[request_i] && avail_format[avail_i].colorSpace == request_color_space)
1463 return avail_format[avail_i];
1464
1465 // If none of the requested image formats could be found, use the first available
1466 return avail_format[0];
1467 }
