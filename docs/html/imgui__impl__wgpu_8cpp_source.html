300static void ImGui_ImplWGPU_SetupRenderState(ImDrawData* draw_data, WGPURenderPassEncoder ctx, FrameResources* fr)
301{
302 ImGui_ImplWGPU_Data* bd = ImGui_ImplWGPU_GetBackendData();
303
304 // Setup orthographic projection matrix into our constant buffer
305 // Our visible imgui space lies from draw_data->DisplayPos (top left) to draw_data->DisplayPos+data_data->DisplaySize (bottom right).
306 {
307 float L = draw_data->DisplayPos.x;
308 float R = draw_data->DisplayPos.x + draw_data->DisplaySize.x;
309 float T = draw_data->DisplayPos.y;
310 float B = draw_data->DisplayPos.y + draw_data->DisplaySize.y;
311 float mvp[4][4] =
312 {
313 { 2.0f/(R-L), 0.0f, 0.0f, 0.0f },
314 { 0.0f, 2.0f/(T-B), 0.0f, 0.0f },
315 { 0.0f, 0.0f, 0.5f, 0.0f },
316 { (R+L)/(L-R), (T+B)/(B-T), 0.5f, 1.0f },
317 };
318 wgpuQueueWriteBuffer(bd->defaultQueue, bd->renderResources.Uniforms, offsetof(Uniforms, MVP), mvp, sizeof(Uniforms::MVP));
319 float gamma;
320 switch (bd->renderTargetFormat)
321 {
322 case WGPUTextureFormat_ASTC10x10UnormSrgb:
323 case WGPUTextureFormat_ASTC10x5UnormSrgb:
324 case WGPUTextureFormat_ASTC10x6UnormSrgb:
325 case WGPUTextureFormat_ASTC10x8UnormSrgb:
326 case WGPUTextureFormat_ASTC12x10UnormSrgb:
327 case WGPUTextureFormat_ASTC12x12UnormSrgb:
328 case WGPUTextureFormat_ASTC4x4UnormSrgb:
329 case WGPUTextureFormat_ASTC5x5UnormSrgb:
330 case WGPUTextureFormat_ASTC6x5UnormSrgb:
331 case WGPUTextureFormat_ASTC6x6UnormSrgb:
332 case WGPUTextureFormat_ASTC8x5UnormSrgb:
333 case WGPUTextureFormat_ASTC8x6UnormSrgb:
334 case WGPUTextureFormat_ASTC8x8UnormSrgb:
335 case WGPUTextureFormat_BC1RGBAUnormSrgb:
336 case WGPUTextureFormat_BC2RGBAUnormSrgb:
337 case WGPUTextureFormat_BC3RGBAUnormSrgb:
338 case WGPUTextureFormat_BC7RGBAUnormSrgb:
339 case WGPUTextureFormat_BGRA8UnormSrgb:
340 case WGPUTextureFormat_ETC2RGB8A1UnormSrgb:
341 case WGPUTextureFormat_ETC2RGB8UnormSrgb:
342 case WGPUTextureFormat_ETC2RGBA8UnormSrgb:
343 case WGPUTextureFormat_RGBA8UnormSrgb:
344 gamma = 2.2f;
345 break;
346 default:
347 gamma = 1.0f;
348 }
349 wgpuQueueWriteBuffer(bd->defaultQueue, bd->renderResources.Uniforms, offsetof(Uniforms, Gamma), &gamma, sizeof(Uniforms::Gamma));
350 }
351
352 // Setup viewport
353 wgpuRenderPassEncoderSetViewport(ctx, 0, 0, draw_data->FramebufferScale.x * draw_data->DisplaySize.x, draw_data->FramebufferScale.y * draw_data->DisplaySize.y, 0, 1);
354
355 // Bind shader and vertex buffers
356 wgpuRenderPassEncoderSetVertexBuffer(ctx, 0, fr->VertexBuffer, 0, fr->VertexBufferSize * sizeof(ImDrawVert));
357 wgpuRenderPassEncoderSetIndexBuffer(ctx, fr->IndexBuffer, sizeof(ImDrawIdx) == 2 ? WGPUIndexFormat_Uint16 : WGPUIndexFormat_Uint32, 0, fr->IndexBufferSize * sizeof(ImDrawIdx));
358 wgpuRenderPassEncoderSetPipeline(ctx, bd->pipelineState);
359 wgpuRenderPassEncoderSetBindGroup(ctx, 0, bd->renderResources.CommonBindGroup, 0, nullptr);
360
361 // Setup blend factor
362 WGPUColor blend_color = { 0.f, 0.f, 0.f, 0.f };
363 wgpuRenderPassEncoderSetBlendConstant(ctx, &blend_color);
364}

366// Render function
367// (this used to be set in io.RenderDrawListsFn and called by ImGui::Render(), but you can now call this directly from your main loop)
368void ImGui_ImplWGPU_RenderDrawData(ImDrawData* draw_data, WGPURenderPassEncoder pass_encoder)
369{
370 // Avoid rendering when minimized
371 int fb_width = (int)(draw_data->DisplaySize.x * draw_data->FramebufferScale.x);
372 int fb_height = (int)(draw_data->DisplaySize.y * draw_data->FramebufferScale.y);
373 if (fb_width <= 0 || fb_height <= 0 || draw_data->CmdLists.Size == 0)
374 return;
375
376 // Catch up with texture updates. Most of the times, the list will have 1 element with an OK status, aka nothing to do.
377 // (This almost always points to ImGui::GetPlatformIO().Textures[] but is part of ImDrawData to allow overriding or disabling texture updates).
378 if (draw_data->Textures != nullptr)
379 for (ImTextureData* tex : *draw_data->Textures)
380 if (tex->Status != ImTextureStatus_OK)
381 ImGui_ImplWGPU_UpdateTexture(tex);
382
383 // FIXME: Assuming that this only gets called once per frame!
384 // If not, we can't just re-allocate the IB or VB, we'll have to do a proper allocator.
385 ImGui_ImplWGPU_Data* bd = ImGui_ImplWGPU_GetBackendData();
386 bd->frameIndex = bd->frameIndex + 1;
387 FrameResources* fr = &bd->pFrameResources[bd->frameIndex % bd->numFramesInFlight];
388
389 // Create and grow vertex/index buffers if needed
390 if (fr->VertexBuffer == nullptr || fr->VertexBufferSize < draw_data->TotalVtxCount)
391 {
392 if (fr->VertexBuffer)
393 {
394 wgpuBufferDestroy(fr->VertexBuffer);
395 wgpuBufferRelease(fr->VertexBuffer);
396 }
397 SafeRelease(fr->VertexBufferHost);
398 fr->VertexBufferSize = draw_data->TotalVtxCount + 5000;
399
400 WGPUBufferDescriptor vb_desc =
401 {
402 nullptr,
403#if !defined(IMGUI_IMPL_WEBGPU_BACKEND_WGPU_EMSCRIPTEN)
404 { "Dear ImGui Vertex buffer", WGPU_STRLEN, },
405#else
406 "Dear ImGui Vertex buffer",
407#endif
408 WGPUBufferUsage_CopyDst | WGPUBufferUsage_Vertex,
409 MEMALIGN(fr->VertexBufferSize * sizeof(ImDrawVert), 4),
410 false
411 };
412 fr->VertexBuffer = wgpuDeviceCreateBuffer(bd->wgpuDevice, &vb_desc);
413 if (!fr->VertexBuffer)
414 return;
415
416 fr->VertexBufferHost = new ImDrawVert[fr->VertexBufferSize];
417 }
418 if (fr->IndexBuffer == nullptr || fr->IndexBufferSize < draw_data->TotalIdxCount)
419 {
420 if (fr->IndexBuffer)
421 {
422 wgpuBufferDestroy(fr->IndexBuffer);
423 wgpuBufferRelease(fr->IndexBuffer);
424 }
425 SafeRelease(fr->IndexBufferHost);
426 fr->IndexBufferSize = draw_data->TotalIdxCount + 10000;
427
428 WGPUBufferDescriptor ib_desc =
429 {
430 nullptr,
431#if !defined(IMGUI_IMPL_WEBGPU_BACKEND_WGPU_EMSCRIPTEN)
432 { "Dear ImGui Index buffer", WGPU_STRLEN, },
433#else
434 "Dear ImGui Index buffer",
435#endif
436 WGPUBufferUsage_CopyDst | WGPUBufferUsage_Index,
437 MEMALIGN(fr->IndexBufferSize * sizeof(ImDrawIdx), 4),
438 false
439 };
440 fr->IndexBuffer = wgpuDeviceCreateBuffer(bd->wgpuDevice, &ib_desc);
441 if (!fr->IndexBuffer)
442 return;
443
444 fr->IndexBufferHost = new ImDrawIdx[fr->IndexBufferSize];
445 }
446
447 // Upload vertex/index data into a single contiguous GPU buffer
448 ImDrawVert* vtx_dst = (ImDrawVert*)fr->VertexBufferHost;
449 ImDrawIdx* idx_dst = (ImDrawIdx*)fr->IndexBufferHost;
450 for (const ImDrawList* draw_list : draw_data->CmdLists)
451 {
452 memcpy(vtx_dst, draw_list->VtxBuffer.Data, draw_list->VtxBuffer.Size * sizeof(ImDrawVert));
453 memcpy(idx_dst, draw_list->IdxBuffer.Data, draw_list->IdxBuffer.Size * sizeof(ImDrawIdx));
454 vtx_dst += draw_list->VtxBuffer.Size;
455 idx_dst += draw_list->IdxBuffer.Size;
456 }
457 int64_t vb_write_size = MEMALIGN((char*)vtx_dst - (char*)fr->VertexBufferHost, 4);
458 int64_t ib_write_size = MEMALIGN((char*)idx_dst - (char*)fr->IndexBufferHost, 4);
459 wgpuQueueWriteBuffer(bd->defaultQueue, fr->VertexBuffer, 0, fr->VertexBufferHost, vb_write_size);
460 wgpuQueueWriteBuffer(bd->defaultQueue, fr->IndexBuffer, 0, fr->IndexBufferHost, ib_write_size);
461
462 // Setup desired render state
463 ImGui_ImplWGPU_SetupRenderState(draw_data, pass_encoder, fr);
464
465 // Setup render state structure (for callbacks and custom texture bindings)
466 ImGuiPlatformIO& platform_io = ImGui::GetPlatformIO();
467 ImGui_ImplWGPU_RenderState render_state;
468 render_state.Device = bd->wgpuDevice;
469 render_state.RenderPassEncoder = pass_encoder;
470 platform_io.Renderer_RenderState = &render_state;
471
472 // Render command lists
473 // (Because we merged all buffers into a single one, we maintain our own offset into them)
474 int global_vtx_offset = 0;
475 int global_idx_offset = 0;
476 ImVec2 clip_scale = draw_data->FramebufferScale;
477 ImVec2 clip_off = draw_data->DisplayPos;
478 for (const ImDrawList* draw_list : draw_data->CmdLists)
479 {
480 for (int cmd_i = 0; cmd_i < draw_list->CmdBuffer.Size; cmd_i++)
481 {
482 const ImDrawCmd* pcmd = &draw_list->CmdBuffer[cmd_i];
483 if (pcmd->UserCallback != nullptr)
484 {
485 // User callback, registered via ImDrawList::AddCallback()
486 // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
487 if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
488 ImGui_ImplWGPU_SetupRenderState(draw_data, pass_encoder, fr);
489 else
490 pcmd->UserCallback(draw_list, pcmd);
491 }
492 else
493 {
494 // Bind custom texture
495 ImTextureID tex_id = pcmd->GetTexID();
496 ImGuiID tex_id_hash = ImHashData(&tex_id, sizeof(tex_id), 0);
497 WGPUBindGroup bind_group = (WGPUBindGroup)bd->renderResources.ImageBindGroups.GetVoidPtr(tex_id_hash);
498 if (!bind_group)
499 {
500 bind_group = ImGui_ImplWGPU_CreateImageBindGroup(bd->renderResources.ImageBindGroupLayout, (WGPUTextureView)tex_id);
501 bd->renderResources.ImageBindGroups.SetVoidPtr(tex_id_hash, bind_group);
502 }
503 wgpuRenderPassEncoderSetBindGroup(pass_encoder, 1, (WGPUBindGroup)bind_group, 0, nullptr);
504
505 // Project scissor/clipping rectangles into framebuffer space
506 ImVec2 clip_min((pcmd->ClipRect.x - clip_off.x) * clip_scale.x, (pcmd->ClipRect.y - clip_off.y) * clip_scale.y);
507 ImVec2 clip_max((pcmd->ClipRect.z - clip_off.x) * clip_scale.x, (pcmd->ClipRect.w - clip_off.y) * clip_scale.y);
508
509 // Clamp to viewport as wgpuRenderPassEncoderSetScissorRect() won't accept values that are off bounds
510 if (clip_min.x < 0.0f) { clip_min.x = 0.0f; }
511 if (clip_min.y < 0.0f) { clip_min.y = 0.0f; }
512 if (clip_max.x > fb_width) { clip_max.x = (float)fb_width; }
513 if (clip_max.y > fb_height) { clip_max.y = (float)fb_height; }
514 if (clip_max.x <= clip_min.x || clip_max.y <= clip_min.y)
515 continue;
516
517 // Apply scissor/clipping rectangle, Draw
518 wgpuRenderPassEncoderSetScissorRect(pass_encoder, (uint32_t)clip_min.x, (uint32_t)clip_min.y, (uint32_t)(clip_max.x - clip_min.x), (uint32_t)(clip_max.y - clip_min.y));
519 wgpuRenderPassEncoderDrawIndexed(pass_encoder, pcmd->ElemCount, 1, pcmd->IdxOffset + global_idx_offset, pcmd->VtxOffset + global_vtx_offset, 0);
520 }
521 }
522 global_idx_offset += draw_list->IdxBuffer.Size;
523 global_vtx_offset += draw_list->VtxBuffer.Size;
524 }

529 {

531 SafeRelease(bind_group);
532 }
533 image_bind_groups.Data.resize(0);
534
535 platform_io.Renderer_RenderState = nullptr;
536}

554void ImGui_ImplWGPU_UpdateTexture(ImTextureData* tex)
555{
556 ImGui_ImplWGPU_Data* bd = ImGui_ImplWGPU_GetBackendData();
557 if (tex->Status == ImTextureStatus_WantCreate)
558 {
559 // Create and upload new texture to graphics system
560 //IMGUI_DEBUG_LOG("UpdateTexture #%03d: WantCreate %dx%d\n", tex->UniqueID, tex->Width, tex->Height);
561 IM_ASSERT(tex->TexID == ImTextureID_Invalid && tex->BackendUserData == nullptr);
562 IM_ASSERT(tex->Format == ImTextureFormat_RGBA32);
563 ImGui_ImplWGPU_Texture* backend_tex = IM_NEW(ImGui_ImplWGPU_Texture)();
564
565 // Create texture
566 WGPUTextureDescriptor tex_desc = {};
567#if !defined(IMGUI_IMPL_WEBGPU_BACKEND_WGPU_EMSCRIPTEN)
568 tex_desc.label = { "Dear ImGui Texture", WGPU_STRLEN };
569#else
570 tex_desc.label = "Dear ImGui Texture";
571#endif
572 tex_desc.dimension = WGPUTextureDimension_2D;
573 tex_desc.size.width = tex->Width;
574 tex_desc.size.height = tex->Height;
575 tex_desc.size.depthOrArrayLayers = 1;
576 tex_desc.sampleCount = 1;
577 tex_desc.format = WGPUTextureFormat_RGBA8Unorm;
578 tex_desc.mipLevelCount = 1;
579 tex_desc.usage = WGPUTextureUsage_CopyDst | WGPUTextureUsage_TextureBinding;
580 backend_tex->Texture = wgpuDeviceCreateTexture(bd->wgpuDevice, &tex_desc);
581
582 // Create texture view
583 WGPUTextureViewDescriptor tex_view_desc = {};
584 tex_view_desc.format = WGPUTextureFormat_RGBA8Unorm;
585 tex_view_desc.dimension = WGPUTextureViewDimension_2D;
586 tex_view_desc.baseMipLevel = 0;
587 tex_view_desc.mipLevelCount = 1;
588 tex_view_desc.baseArrayLayer = 0;
589 tex_view_desc.arrayLayerCount = 1;
590 tex_view_desc.aspect = WGPUTextureAspect_All;
591 backend_tex->TextureView = wgpuTextureCreateView(backend_tex->Texture, &tex_view_desc);
592
593 // Store identifiers
594 tex->SetTexID((ImTextureID)(intptr_t)backend_tex->TextureView);
595 tex->BackendUserData = backend_tex;
596 // We don't set tex->Status to ImTextureStatus_OK to let the code fallthrough below.
597 }
598
599 if (tex->Status == ImTextureStatus_WantCreate || tex->Status == ImTextureStatus_WantUpdates)
600 {
601 ImGui_ImplWGPU_Texture* backend_tex = (ImGui_ImplWGPU_Texture*)tex->BackendUserData;
602 IM_ASSERT(tex->Format == ImTextureFormat_RGBA32);
603
604 // We could use the smaller rect on _WantCreate but using the full rect allows us to clear the texture.
605 const int upload_x = (tex->Status == ImTextureStatus_WantCreate) ? 0 : tex->UpdateRect.x;
606 const int upload_y = (tex->Status == ImTextureStatus_WantCreate) ? 0 : tex->UpdateRect.y;
607 const int upload_w = (tex->Status == ImTextureStatus_WantCreate) ? tex->Width : tex->UpdateRect.w;
608 const int upload_h = (tex->Status == ImTextureStatus_WantCreate) ? tex->Height : tex->UpdateRect.h;
609
610 // Update full texture or selected blocks. We only ever write to textures regions which have never been used before!
611 // This backend choose to use tex->UpdateRect but you can use tex->Updates[] to upload individual regions.
612#if !defined(IMGUI_IMPL_WEBGPU_BACKEND_WGPU_EMSCRIPTEN)
613 WGPUTexelCopyTextureInfo dst_view = {};
614#else
615 WGPUImageCopyTexture dst_view = {};
616#endif
617 dst_view.texture = backend_tex->Texture;
618 dst_view.mipLevel = 0;
619 dst_view.origin = { (uint32_t)upload_x, (uint32_t)upload_y, 0 };
620 dst_view.aspect = WGPUTextureAspect_All;
621#if !defined(IMGUI_IMPL_WEBGPU_BACKEND_WGPU_EMSCRIPTEN)
622 WGPUTexelCopyBufferLayout layout = {};
623#else
624 WGPUTextureDataLayout layout = {};
625#endif
626 layout.offset = 0;
627 layout.bytesPerRow = tex->Width * tex->BytesPerPixel;
628 layout.rowsPerImage = upload_h;
629 WGPUExtent3D write_size = { (uint32_t)upload_w, (uint32_t)upload_h, 1 };
630 wgpuQueueWriteTexture(bd->defaultQueue, &dst_view, tex->GetPixelsAt(upload_x, upload_y), (uint32_t)(tex->Width * upload_h * tex->BytesPerPixel), &layout, &write_size);
631 tex->SetStatus(ImTextureStatus_OK);
632 }
633 if (tex->Status == ImTextureStatus_WantDestroy && tex->UnusedFrames > 0)
634 ImGui_ImplWGPU_DestroyTexture(tex);
635}

655bool ImGui_ImplWGPU_CreateDeviceObjects()
656{
657 ImGui_ImplWGPU_Data* bd = ImGui_ImplWGPU_GetBackendData();
658 if (!bd->wgpuDevice)
659 return false;
660 if (bd->pipelineState)
661 ImGui_ImplWGPU_InvalidateDeviceObjects();
662
663 // Create render pipeline
664 WGPURenderPipelineDescriptor graphics_pipeline_desc = {};
665 graphics_pipeline_desc.primitive.topology = WGPUPrimitiveTopology_TriangleList;
666 graphics_pipeline_desc.primitive.stripIndexFormat = WGPUIndexFormat_Undefined;
667 graphics_pipeline_desc.primitive.frontFace = WGPUFrontFace_CW;
668 graphics_pipeline_desc.primitive.cullMode = WGPUCullMode_None;
669 graphics_pipeline_desc.multisample = bd->initInfo.PipelineMultisampleState;
670
671 // Bind group layouts
672 WGPUBindGroupLayoutEntry common_bg_layout_entries[2] = {};
673 common_bg_layout_entries[0].binding = 0;
674 common_bg_layout_entries[0].visibility = WGPUShaderStage_Vertex | WGPUShaderStage_Fragment;
675 common_bg_layout_entries[0].buffer.type = WGPUBufferBindingType_Uniform;
676 common_bg_layout_entries[1].binding = 1;
677 common_bg_layout_entries[1].visibility = WGPUShaderStage_Fragment;
678 common_bg_layout_entries[1].sampler.type = WGPUSamplerBindingType_Filtering;
679
680 WGPUBindGroupLayoutEntry image_bg_layout_entries[1] = {};
681 image_bg_layout_entries[0].binding = 0;
682 image_bg_layout_entries[0].visibility = WGPUShaderStage_Fragment;
683 image_bg_layout_entries[0].texture.sampleType = WGPUTextureSampleType_Float;
684 image_bg_layout_entries[0].texture.viewDimension = WGPUTextureViewDimension_2D;
685
686 WGPUBindGroupLayoutDescriptor common_bg_layout_desc = {};
687 common_bg_layout_desc.entryCount = 2;
688 common_bg_layout_desc.entries = common_bg_layout_entries;
689
690 WGPUBindGroupLayoutDescriptor image_bg_layout_desc = {};
691 image_bg_layout_desc.entryCount = 1;
692 image_bg_layout_desc.entries = image_bg_layout_entries;
693
694 WGPUBindGroupLayout bg_layouts[2];
695 bg_layouts[0] = wgpuDeviceCreateBindGroupLayout(bd->wgpuDevice, &common_bg_layout_desc);
696 bg_layouts[1] = wgpuDeviceCreateBindGroupLayout(bd->wgpuDevice, &image_bg_layout_desc);
697
698 WGPUPipelineLayoutDescriptor layout_desc = {};
699 layout_desc.bindGroupLayoutCount = 2;
700 layout_desc.bindGroupLayouts = bg_layouts;
701 graphics_pipeline_desc.layout = wgpuDeviceCreatePipelineLayout(bd->wgpuDevice, &layout_desc);
702
703 // Create the vertex shader
704 WGPUProgrammableStageDescriptor vertex_shader_desc = ImGui_ImplWGPU_CreateShaderModule(__shader_vert_wgsl);
705 graphics_pipeline_desc.vertex.module = vertex_shader_desc.module;
706 graphics_pipeline_desc.vertex.entryPoint = vertex_shader_desc.entryPoint;
707
708 // Vertex input configuration
709 WGPUVertexAttribute attribute_desc[] =
710 {
711#ifdef IMGUI_IMPL_WEBGPU_BACKEND_DAWN
712 { nullptr, WGPUVertexFormat_Float32x2, (uint64_t)offsetof(ImDrawVert, pos), 0 },
713 { nullptr, WGPUVertexFormat_Float32x2, (uint64_t)offsetof(ImDrawVert, uv), 1 },
714 { nullptr, WGPUVertexFormat_Unorm8x4, (uint64_t)offsetof(ImDrawVert, col), 2 },
715#else
716 { WGPUVertexFormat_Float32x2, (uint64_t)offsetof(ImDrawVert, pos), 0 },
717 { WGPUVertexFormat_Float32x2, (uint64_t)offsetof(ImDrawVert, uv), 1 },
718 { WGPUVertexFormat_Unorm8x4, (uint64_t)offsetof(ImDrawVert, col), 2 },
719#endif
720 };
721
722 WGPUVertexBufferLayout buffer_layouts[1];
723 buffer_layouts[0].arrayStride = sizeof(ImDrawVert);
724 buffer_layouts[0].stepMode = WGPUVertexStepMode_Vertex;
725 buffer_layouts[0].attributeCount = 3;
726 buffer_layouts[0].attributes = attribute_desc;
727
728 graphics_pipeline_desc.vertex.bufferCount = 1;
729 graphics_pipeline_desc.vertex.buffers = buffer_layouts;
730
731 // Create the pixel shader
732 WGPUProgrammableStageDescriptor pixel_shader_desc = ImGui_ImplWGPU_CreateShaderModule(__shader_frag_wgsl);
733
734 // Create the blending setup
735 WGPUBlendState blend_state = {};
736 blend_state.alpha.operation = WGPUBlendOperation_Add;
737 blend_state.alpha.srcFactor = WGPUBlendFactor_One;
738 blend_state.alpha.dstFactor = WGPUBlendFactor_OneMinusSrcAlpha;
739 blend_state.color.operation = WGPUBlendOperation_Add;
740 blend_state.color.srcFactor = WGPUBlendFactor_SrcAlpha;
741 blend_state.color.dstFactor = WGPUBlendFactor_OneMinusSrcAlpha;
742
743 WGPUColorTargetState color_state = {};
744 color_state.format = bd->renderTargetFormat;
745 color_state.blend = &blend_state;
746 color_state.writeMask = WGPUColorWriteMask_All;
747
748 WGPUFragmentState fragment_state = {};
749 fragment_state.module = pixel_shader_desc.module;
750 fragment_state.entryPoint = pixel_shader_desc.entryPoint;
751 fragment_state.targetCount = 1;
752 fragment_state.targets = &color_state;
753
754 graphics_pipeline_desc.fragment = &fragment_state;
755
756 // Create depth-stencil State
757 WGPUDepthStencilState depth_stencil_state = {};
758 depth_stencil_state.format = bd->depthStencilFormat;
759#if !defined(IMGUI_IMPL_WEBGPU_BACKEND_WGPU_EMSCRIPTEN)
760 depth_stencil_state.depthWriteEnabled = WGPUOptionalBool_False;
761#else
762 depth_stencil_state.depthWriteEnabled = false;
763#endif
764 depth_stencil_state.depthCompare = WGPUCompareFunction_Always;
765 depth_stencil_state.stencilFront.compare = WGPUCompareFunction_Always;
766 depth_stencil_state.stencilFront.failOp = WGPUStencilOperation_Keep;
767 depth_stencil_state.stencilFront.depthFailOp = WGPUStencilOperation_Keep;
768 depth_stencil_state.stencilFront.passOp = WGPUStencilOperation_Keep;
769 depth_stencil_state.stencilBack.compare = WGPUCompareFunction_Always;
770 depth_stencil_state.stencilBack.failOp = WGPUStencilOperation_Keep;
771 depth_stencil_state.stencilBack.depthFailOp = WGPUStencilOperation_Keep;
772 depth_stencil_state.stencilBack.passOp = WGPUStencilOperation_Keep;
773
774 // Configure disabled depth-stencil state
775 graphics_pipeline_desc.depthStencil = (bd->depthStencilFormat == WGPUTextureFormat_Undefined) ? nullptr : &depth_stencil_state;
776
777 bd->pipelineState = wgpuDeviceCreateRenderPipeline(bd->wgpuDevice, &graphics_pipeline_desc);
778
779 ImGui_ImplWGPU_CreateUniformBuffer();
780
781 // Create sampler
782 // (Bilinear sampling is required by default. Set 'io.Fonts->Flags |= ImFontAtlasFlags_NoBakedLines' or 'style.AntiAliasedLinesUseTex = false' to allow point/nearest sampling)
783 WGPUSamplerDescriptor sampler_desc = {};
784 sampler_desc.minFilter = WGPUFilterMode_Linear;
785 sampler_desc.magFilter = WGPUFilterMode_Linear;
786 sampler_desc.mipmapFilter = WGPUMipmapFilterMode_Linear;
787 sampler_desc.addressModeU = WGPUAddressMode_ClampToEdge;
788 sampler_desc.addressModeV = WGPUAddressMode_ClampToEdge;
789 sampler_desc.addressModeW = WGPUAddressMode_ClampToEdge;
790 sampler_desc.maxAnisotropy = 1;
791 bd->renderResources.Sampler = wgpuDeviceCreateSampler(bd->wgpuDevice, &sampler_desc);
792
793 // Create resource bind group
794 WGPUBindGroupEntry common_bg_entries[] =
795 {
796 { nullptr, 0, bd->renderResources.Uniforms, 0, MEMALIGN(sizeof(Uniforms), 16), 0, 0 },
797 { nullptr, 1, 0, 0, 0, bd->renderResources.Sampler, 0 },
798 };
799 WGPUBindGroupDescriptor common_bg_descriptor = {};
800 common_bg_descriptor.layout = bg_layouts[0];
801 common_bg_descriptor.entryCount = sizeof(common_bg_entries) / sizeof(WGPUBindGroupEntry);
802 common_bg_descriptor.entries = common_bg_entries;
803 bd->renderResources.CommonBindGroup = wgpuDeviceCreateBindGroup(bd->wgpuDevice, &common_bg_descriptor);
804 bd->renderResources.ImageBindGroupLayout = bg_layouts[1];
805
806 SafeRelease(vertex_shader_desc.module);
807 SafeRelease(pixel_shader_desc.module);
808 SafeRelease(graphics_pipeline_desc.layout);
809 SafeRelease(bg_layouts[0]);
810
811 return true;
812}
