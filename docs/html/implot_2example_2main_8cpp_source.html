31// Callback to handle GLFW errors
32void glfw_error_callback(int error, const char* description) { std::cerr << "GLFW Error " << error << ": " << description << std::endl; }
33
34int main() {
35 // Setup error callback
36 glfwSetErrorCallback(glfw_error_callback);
37
38 // Initialize GLFW
39 if (!glfwInit()) {
40 std::cerr << "Failed to initialize GLFW" << std::endl;
41 return -1;
42 }
43
44 // Setup OpenGL version
45#if defined(__APPLE__)
46 // GL 3.2 + GLSL 150 (MacOS)
47 const char* glsl_version = "#version 150";
48 glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 3);
49 glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 2);
50 glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE); // 3.2+ only
51 glfwWindowHint(GLFW_OPENGL_FORWARD_COMPAT, GL_TRUE); // Required on MacOS
52#else
53 // GL 3.0 + GLSL 130 (Windows and Linux)
54 const char* glsl_version = "#version 130";
55 glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 3);
56 glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 0);
57#endif
58
59 // Create window
60 GLFWwindow* window = glfwCreateWindow(1200, 800, "ImPlot Example", nullptr, nullptr);
61 if (!window) {
62 std::cerr << "Failed to create GLFW window" << std::endl;
63 glfwTerminate();
64 return -1;
65 }
66 glfwMakeContextCurrent(window);
67 glfwSwapInterval(0); // Disable vsync
68
69 // Setup context
70 IMGUI_CHECKVERSION();
71 ImGui::CreateContext();
72 ImPlot::CreateContext();
73
74 // Setup style
75 ImGui::StyleColorsDark();
76
77 // Setup backend
78 ImGui_ImplGlfw_InitForOpenGL(window, true);
79 ImGui_ImplOpenGL3_Init(glsl_version);
80
81 // Main loop
82 while (!glfwWindowShouldClose(window)) {
83 glfwPollEvents();
84
85 // Start frame
86 ImGui_ImplOpenGL3_NewFrame();
87 ImGui_ImplGlfw_NewFrame();
88 ImGui::NewFrame();
89
90 // Demo windows
91 ImGui::ShowDemoWindow();
92 ImPlot::ShowDemoWindow();
93
94 // Render
95 ImGui::Render();
96 int display_w, display_h;
97 glfwGetFramebufferSize(window, &display_w, &display_h);
98 glViewport(0, 0, display_w, display_h);
99 glClearColor(0.1f, 0.1f, 0.1f, 1.0f);
100 glClear(GL_COLOR_BUFFER_BIT);
101 ImGui_ImplOpenGL3_RenderDrawData(ImGui::GetDrawData());
102
103 // Swap buffers
104 glfwSwapBuffers(window);
105 }
106
107 // Cleanup
108 ImGui_ImplOpenGL3_Shutdown();
109 ImGui_ImplGlfw_Shutdown();
110 ImPlot::DestroyContext();
111 ImGui::DestroyContext();
112 glfwDestroyWindow(window);
113 glfwTerminate();
114
115 return 0;
116}
