81template <typename T>
82inline T RandomRange(T min, T max) {
83 T scale = rand() / (T) RAND_MAX;
84 return min + scale * ( max - min );
85}

96double RandomGauss() {
97 static double V1, V2, S;
98 static int phase = 0;
99 double X;
100 if(phase == 0) {
101 do {
102 double U1 = (double)rand() / RAND_MAX;
103 double U2 = (double)rand() / RAND_MAX;
104 V1 = 2 * U1 - 1;
105 V2 = 2 * U2 - 1;
106 S = V1 * V1 + V2 * V2;
107 } while(S >= 1 || S == 0);
108
109 X = V1 * sqrt(-2 * log(S) / S);
110 } else
111 X = V2 * sqrt(-2 * log(S) / S);
112 phase = 1 - phase;
113 return X;
114}

125// utility structure for realtime plot
126struct ScrollingBuffer {
127 int MaxSize;
128 int Offset;
129 ImVector<ImVec2> Data;
130 ScrollingBuffer(int max_size = 2000) {
131 MaxSize = max_size;
132 Offset = 0;
133 Data.reserve(MaxSize);
134 }
135 void AddPoint(float x, float y) {
136 if (Data.size() < MaxSize)
137 Data.push_back(ImVec2(x,y));
138 else {
139 Data[Offset] = ImVec2(x,y);
140 Offset = (Offset + 1) % MaxSize;
141 }
142 }
143 void Erase() {
144 if (Data.size() > 0) {
145 Data.shrink(0);
146 Offset = 0;
147 }
148 }
149};

151// utility structure for realtime plot
152struct RollingBuffer {
153 float Span;
154 ImVector<ImVec2> Data;
155 RollingBuffer() {
156 Span = 10.0f;
157 Data.reserve(2000);
158 }
159 void AddPoint(float x, float y) {
160 float xmod = fmodf(x, Span);
161 if (!Data.empty() && xmod < Data.back().x)
162 Data.shrink(0);
163 Data.push_back(ImVec2(xmod, y));
164 }

167// Huge data used by Time Formatting example (~500 MB allocation!)
168struct HugeTimeData {
169 HugeTimeData(double min) {
170 Ts = new double[Size];
171 Ys = new double[Size];
172 for (int i = 0; i < Size; ++i) {
173 Ts[i] = min + i;
174 Ys[i] = GetY(Ts[i]);
175 }
176 }
177 ~HugeTimeData() { delete[] Ts; delete[] Ys; }
178 static double GetY(double t) {
179 return 0.5 + 0.25 * sin(t/86400/12) + 0.005 * sin(t/3600);
180 }
181 double* Ts;
182 double* Ys;
183 static const int Size = 60*60*24*366;
184};

715 static int bins = 50;
716 static double mu = 5;

718 ImGui::SetNextItemWidth(200);

725 ImGui::SameLine();
726 ImGui::SetNextItemWidth(200);
727 ImGui::SliderInt("##Bins", &bins, 1, 100);
728 }
729 ImGui::CheckboxFlags("Horizontal", (unsigned int*)&hist_flags, ImPlotHistogramFlags_Horizontal);
730 ImGui::SameLine();
731 ImGui::CheckboxFlags("Density", (unsigned int*)&hist_flags, ImPlotHistogramFlags_Density);
732 ImGui::SameLine();
733 ImGui::CheckboxFlags("Cumulative", (unsigned int*)&hist_flags, ImPlotHistogramFlags_Cumulative);
734
735 static bool range = false;
736 ImGui::Checkbox("Range", &range);
737 static float rmin = -3;
738 static float rmax = 13;
739 if (range) {
740 ImGui::SameLine();
741 ImGui::SetNextItemWidth(200);
742 ImGui::DragFloat2("##Range",&rmin,0.1f,-3,13);
743 ImGui::SameLine();
744 ImGui::CheckboxFlags("Exclude Outliers", (unsigned int*)&hist_flags, ImPlotHistogramFlags_NoOutliers);
745 }
746 static NormalDistribution<10000> dist(mu, sigma);
747 static double x[100];
748 static double y[100];
749 if (hist_flags & ImPlotHistogramFlags_Density) {
750 for (int i = 0; i < 100; ++i) {
751 x[i] = -3 + 16 * (double)i/99.0;
752 y[i] = exp( - (x[i]-mu)*(x[i]-mu) / (2*sigma*sigma)) / (sigma * sqrt(2*3.141592653589793238));
753 }
754 if (hist_flags & ImPlotHistogramFlags_Cumulative) {
755 for (int i = 1; i < 100; ++i)
756 y[i] += y[i-1];
757 for (int i = 0; i < 100; ++i)
758 y[i] /= y[99];
759 }
760 }
761
762 if (ImPlot::BeginPlot("##Histograms")) {
763 ImPlot::SetupAxes(nullptr,nullptr,ImPlotAxisFlags_AutoFit,ImPlotAxisFlags_AutoFit);
764 ImPlot::SetNextFillStyle(IMPLOT_AUTO_COL,0.5f);
765 ImPlot::PlotHistogram("Empirical", dist.Data, 10000, bins, 1.0, range ? ImPlotRange(rmin,rmax) : ImPlotRange(), hist_flags);
766 if ((hist_flags & ImPlotHistogramFlags_Density) && !(hist_flags & ImPlotHistogramFlags_NoOutliers)) {
767 if (hist_flags & ImPlotHistogramFlags_Horizontal)
768 ImPlot::PlotLine("Theoretical",y,x,100);
769 else
770 ImPlot::PlotLine("Theoretical",x,y,100);
771 }
772 ImPlot::EndPlot();

1981int MetricFormatter(double value, char* buff, int size, void* data) {
1982 const char* unit = (const char*)data;
1983 static double v[] = {1000000000,1000000,1000,1,0.001,0.000001,0.000000001};
1984 static const char* p[] = {"G","M","k","","m","u","n"};
1985 if (value == 0) {
1986 return snprintf(buff,size,"0 %s", unit);
1987 }
1988 for (int i = 0; i < 7; ++i) {
1989 if (fabs(value) >= v[i]) {
1990 return snprintf(buff,size,"%g %s%s",value/v[i],p[i],unit);
1991 }
1992 }
1993 return snprintf(buff,size,"%g %s%s",value/v[6],p[6],unit);
1994}
